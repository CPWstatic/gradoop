539       .stream()
540       .filter(g -> g.getLabel().equals("Community"))
541       .filter(g -> g.getPropertyValue(PROP_INTEREST).getString().equals("Hadoop") ||
542           g.getPropertyValue(PROP_INTEREST).getString().equals("Graphs"))
543       .collect(Collectors.toList());
544 
545     List<EPGMEdge> edges = getSocialEdges()
546       .stream()
547       .filter(e -> e.getLabel().matches(PATTERN_EDGE.pattern()) ||
548         (e.hasProperty(PROP_SINCE) && e.getPropertyValue(PROP_SINCE).getInt() < 2015))
549       .collect(Collectors.toList());
550 
551     List<EPGMVertex> vertices = getSocialVertices()
552       .stream()
553       .filter(v -> v.getLabel().equals("Person"))
554       .collect(Collectors.toList())
555       .subList(1, 4);
556 
557     // Query the store
558     List<EPGMGraphHead> graphHeadResult = epgmStores[storeIndex].getGraphSpace(
559       Query.elements()
560         .fromAll()
561         .where(HBaseFilters.<EPGMGraphHead>labelIn("Community")
562           .and(HBaseFilters.<EPGMGraphHead>propEquals(PROP_INTEREST, "Hadoop")
563             .or(HBaseFilters.propEquals(PROP_INTEREST, "Graphs")))))
564       .readRemainsAndClose();
565 
566     List<EPGMEdge> edgeResult = epgmStores[storeIndex].getEdgeSpace(
567       Query.elements()
568         .fromAll()
569         // WHERE edge.label LIKE '^has.*$' OR edge.since < 2015
570         .where(HBaseFilters.<EPGMEdge>labelReg(PATTERN_EDGE)
571           .or(HBaseFilters.<EPGMEdge>propLargerThan(PROP_SINCE, 2015, true).negate())))
572       .readRemainsAndClose();
573 
574     final HBaseElementFilter<EPGMVertex> vertexFilter = HBaseFilters.<EPGMVertex>labelIn("Person")
575       .and(HBaseFilters.<EPGMVertex>propEquals(PROP_NAME, vertices.get(0).getPropertyValue("name"))
576         .or(HBaseFilters.<EPGMVertex>propEquals(PROP_NAME, vertices.get(1).getPropertyValue("name"))
577           .or(HBaseFilters.propEquals(PROP_NAME, vertices.get(2).getPropertyValue("name")))));
578 
579     List<EPGMVertex> vertexResult = epgmStores[storeIndex].getVertexSpace(
580       Query.elements()
581         .fromAll()
582         .where(vertexFilter))
583       .readRemainsAndClose();
584 
585     assertEquals(graphHeadResult.size(), 2);
586     assertEquals(edgeResult.size(), 21);
587     assertEquals(vertexResult.size(), 3);
588 
589     validateElementCollections(graphHeads, graphHeadResult);
590     validateElementCollections(vertices, vertexResult);
591     validateElementCollections(edges, edgeResult);
592   }
