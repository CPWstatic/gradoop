38  /**
39   * Test class for {@link HBasePropLargerThan}

41  @RunWith(Parameterized.class)
42  public class HBasePropLargerThanTest {
43    /**
44     * Property type
45     */
46    private final String propertyType;
47  
48    /**
49     * Property key
50     */
51    private final String propertyKey;
52  
53    /**
54     * Property value
55     */
56    private final PropertyValue propertyValue;
57  
58    /**
59     * Flag if min value should be included
60     */
61    private final boolean isInclude;
62  
63    /**
64     * Constructor for parametrized test
65     *
66     * @param propertyKey property key to test
67     * @param value property value to test
68     * @param isInclude flag if min value should be included
69     */
70    public HBasePropLargerThanTest(String propertyKey, Object value, boolean isInclude) {
71      this.propertyKey = propertyKey;
72      this.propertyValue = PropertyValue.create(value);
73      this.isInclude = isInclude;
74      this.propertyType = this.propertyValue.getType() == null ?
75        "null" : this.propertyValue.getType().toString();
76    }
77  
78    /**
79     * Test the toHBaseFilter function
80     */
81    @Test
82    public void testToHBaseFilter() {
83      HBasePropLargerThan<Vertex> vertexFilter =
84        new HBasePropLargerThan<>(propertyKey, propertyValue, isInclude);
85  
86      FilterList expectedFilter = new FilterList(FilterList.Operator.MUST_PASS_ALL);
87  
88      SingleColumnValueFilter valueFilter = new SingleColumnValueFilter(
89        Bytes.toBytesBinary(CF_PROPERTY_VALUE),
90        Bytes.toBytesBinary(propertyKey),
91        isInclude ? CompareFilter.CompareOp.GREATER_OR_EQUAL : CompareFilter.CompareOp.GREATER,
92        new BinaryComparator(PropertyValueUtils.Bytes.getRawBytesWithoutType(propertyValue)));
93  
94      // Define that the entire row will be skipped if the column is not found
95      valueFilter.setFilterIfMissing(true);
96  
97      SingleColumnValueFilter typeFilter = new SingleColumnValueFilter(
98        Bytes.toBytesBinary(CF_PROPERTY_TYPE),
99        Bytes.toBytesBinary(propertyKey),
100       CompareFilter.CompareOp.EQUAL,
101       PropertyValueUtils.Bytes.getTypeByte(propertyValue));
102 
103     // Define that the entire row will be skipped if the column is not found
104     typeFilter.setFilterIfMissing(true);
105 
106     expectedFilter.addFilter(valueFilter);
107     expectedFilter.addFilter(typeFilter);
108 
109     assertEquals("Failed during filter comparison for type [" + propertyType + "].",
110       expectedFilter.toString(), vertexFilter.toHBaseFilter(false).toString());
111   }
