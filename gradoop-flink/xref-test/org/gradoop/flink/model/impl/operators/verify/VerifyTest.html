50    /**
51     * Test the verify operator with a subgraph operator.
52     *
53     * @throws Exception when the execution in Flink fails.
54     */
55    @Test
56    public void testVerifyWithSubgraph() throws Exception {
57      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
58      loader.appendToDatabaseFromString("expected[" +
59        "(eve)-[ekb:knows {since : 2015}]->(bob)" +
60        "]");
61      LogicalGraph input = loader.getLogicalGraphByVariable("g0");
62      // Apply a subgraph operator that would result in dangling edges.
63      LogicalGraph subgraph = input.subgraph(
64        new ByProperty<EPGMVertex>("name", PropertyValue.create("Alice")).negate(),
65        new True<>(), Subgraph.Strategy.BOTH);
66      // Make sure that the graph contains dangling edges.
67      List<EPGMEdge> danglingEdges = getDanglingEdges(subgraph);
68      List<EPGMEdge> expectedDanglingEdges = Arrays.asList(loader.getEdgeByVariable("eka"),
69        loader.getEdgeByVariable("akb"), loader.getEdgeByVariable("bka"));
70      Comparator<EPGMEdge> comparator = Comparator.comparing(EPGMEdge::getId);
71      danglingEdges.sort(comparator);
72      expectedDanglingEdges.sort(comparator);
73      assertArrayEquals(expectedDanglingEdges.toArray(), danglingEdges.toArray());
74      // Now run verify and check if those edges were removed.
75      LogicalGraph verifiedSubgraph = subgraph.verify();
76      assertEquals("Verified graph contained dangling edges.",
77        0, getDanglingEdges(verifiedSubgraph).size());
78      // Check if nothing else has been removed (i.e. the result is correct)
79      collectAndAssertTrue(loader.getLogicalGraphByVariable("expected")
80        .equalsByElementData(verifiedSubgraph));
81    }

83    /**
84     * Determines all dangling edges of a graph.
85     *
86     * @throws Exception when the execution in Flink fails.
87     */
88    private List<EPGMEdge> getDanglingEdges(LogicalGraph graph) throws Exception {
89      List<EPGMVertex> vertices = new ArrayList<>();
90      List<EPGMEdge> edges = new ArrayList<>();
91      graph.getVertices().output(new LocalCollectionOutputFormat<>(vertices));
92      graph.getEdges().output(new LocalCollectionOutputFormat<>(edges));
93      getExecutionEnvironment().execute();
94      Set<GradoopId> ids = vertices.stream().map(EPGMVertex::getId).collect(Collectors.toSet());
95      return edges.stream()
96        .filter(e -> !ids.contains(e.getSourceId()) || !ids.contains(e.getTargetId()))
97        .collect(Collectors.toList());
98    }

136   /**
137    * Function to test graph element datasets for dangling graph ids.
138    *
139    * @param idSet ids of the corresponding graph or graph collection.
140    * @param elements element dataset to test for dangling grahp ids.
141    * @param <E> element type.
142    * @throws Exception when the execution in Flink fails.
143    */
144   private <E extends GraphElement> void assertGraphContainment(GradoopIdSet idSet, DataSet<E> elements)
145     throws Exception {
146     for (E element : elements.collect()) {
147       GradoopIdSet ids = element.getGraphIds();
148       assertFalse("EPGMElement has no graph ids", ids.isEmpty());
149       ids.removeAll(idSet);
150       assertTrue("EPGMElement has dangling graph ids", ids.isEmpty());
151     }
