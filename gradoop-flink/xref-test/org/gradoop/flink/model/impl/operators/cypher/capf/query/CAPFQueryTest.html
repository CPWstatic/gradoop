56      Map<String, List<PropertyMetaData>> edgePropertyMap = new HashMap<>();

60        null));

62      vertexPropertyMap.put("A", propertyList);
63      vertexPropertyMap.put("B", propertyList);
64      vertexPropertyMap.put("C", propertyList);
65      vertexPropertyMap.put("D", propertyList);
66  
67      edgePropertyMap.put("a", propertyList);
68      edgePropertyMap.put("b", propertyList);
69      edgePropertyMap.put("c", propertyList);
70  
71      MetaData metaData = new CSVMetaData(new HashMap<>(), vertexPropertyMap, edgePropertyMap);
72  
73      CAPFQueryResult result = graph.cypher("MATCH (n1)-->(n2)<--(n3) RETURN n2", metaData);

75      // because the pattern is symmetric, each result exists twice
76      GraphCollection expectedGraphs = loader.getGraphCollectionByVariables(
77        "expected1", "expected2", "expected3", "expected4",
78        "expected5", "expected6", "expected7", "expected8");

150     FlinkAsciiGraphLoader loader = getLoaderFromString(TestData.GRAPH_1);
151 
152     LogicalGraph graph = loader.getLogicalGraphByVariable(TestData.DATA_GRAPH_VARIABLE);
153 
154     Map<String, List<PropertyMetaData>> vertexPropertyMap = new HashMap<>();
155     Map<String, List<PropertyMetaData>> edgePropertyMap = new HashMap<>();
156 
157     List<PropertyMetaData> propertyList = new ArrayList<>();
158     propertyList.add(new PropertyMetaData("id", Type.INTEGER.toString(),
159       null));
160 
161     vertexPropertyMap.put("A", propertyList);
162     vertexPropertyMap.put("B", propertyList);
163     vertexPropertyMap.put("C", propertyList);
164     vertexPropertyMap.put("D", propertyList);
165 
166     edgePropertyMap.put("a", propertyList);
167     edgePropertyMap.put("b", propertyList);
168     edgePropertyMap.put("c", propertyList);
169 
170     MetaData metaData = new CSVMetaData(new HashMap<>(), vertexPropertyMap, edgePropertyMap);
171 
172     CAPFQueryResult result = graph.cypher(
173       "MATCH (n1)-->(n2)-->(n3) RETURN n1.id, n2.id, n3.id",
174       metaData);
175 
176     BatchTableEnvironment tenv = (BatchTableEnvironment) result.getTable().tableEnv();
177     DataSet<Row> resultDataSet =
178       tenv.toDataSet(result.getTable(), TypeInformation.of(Row.class)).javaSet();
179 
180     Long[][] expectedIds = {
181       {0L, 1L, 1L, 1L, 2L, 2L, 2L, 4L, 5L, 5L, 5L, 6L, 6L, 6L, 8L, 8L},
182       {1L, 6L, 6L, 6L, 6L, 6L, 6L, 1L, 4L, 4L, 9L, 2L, 5L, 5L, 5L, 5L},
183       {6L, 2L, 5L, 7L, 2L, 5L, 7L, 6L, 1L, 3L, 10L, 6L, 4L, 9L, 4L, 9L}
184     };
185 
186     List<Row> resultList = resultDataSet.collect();
187 
188     assertEquals(expectedIds[0].length, resultList.size());
189 
190     for (Row r : resultList) {
191       assertEquals(3, r.getArity());
192     }
193 
194     resultList.sort((r1, r2) -> {
195       for (int i = 0; i < r1.getArity(); i++) {
196         int comp = ((Long) r1.getField(i)).compareTo((Long) r2.getField(i));
197         if (comp != 0) {
198           return comp;
199         }
200       }
201       return 0;
202     });
203 
204     for (int i = 0; i < expectedIds.length; i++) {
205       assertEquals(expectedIds[0][i], resultList.get(i).getField(0));
206       assertEquals(expectedIds[1][i], resultList.get(i).getField(1));
207       assertEquals(expectedIds[2][i], resultList.get(i).getField(2));
208     }
209   }
