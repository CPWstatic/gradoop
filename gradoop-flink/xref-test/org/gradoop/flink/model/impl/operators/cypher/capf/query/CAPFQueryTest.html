64      vertexPropertyMap.put("D", propertyList);

68      edgePropertyMap.put("c", propertyList);

149     FlinkAsciiGraphLoader loader = getLoaderFromString(TestData.GRAPH_1);
150 
151     LogicalGraph graph = loader.getLogicalGraphByVariable(TestData.DATA_GRAPH_VARIABLE);
152 
153     Map<String, List<PropertyMetaData>> vertexPropertyMap = new HashMap<>();
154     Map<String, List<PropertyMetaData>> edgePropertyMap = new HashMap<>();
155 
156     List<PropertyMetaData> propertyList = new ArrayList<>();
157     propertyList.add(new PropertyMetaData("id", MetaData.TypeString.INTEGER.getTypeString(),
158       null));
159 
160     vertexPropertyMap.put("A", propertyList);
161     vertexPropertyMap.put("B", propertyList);
162     vertexPropertyMap.put("C", propertyList);
163     vertexPropertyMap.put("D", propertyList);
164 
165     edgePropertyMap.put("a", propertyList);
166     edgePropertyMap.put("b", propertyList);
167     edgePropertyMap.put("c", propertyList);
168 
169     MetaData metaData = new CSVMetaData(new HashMap<>(), vertexPropertyMap, edgePropertyMap);
170 
171     CAPFQueryResult result = graph.cypher(
172       "MATCH (n1)-->(n2)-->(n3) RETURN n1.id, n2.id, n3.id",
173       metaData);
174 
175     BatchTableEnvironment tenv = (BatchTableEnvironment) result.getTable().tableEnv();
176     DataSet<Row> resultDataSet =
177       tenv.toDataSet(result.getTable(), TypeInformation.of(Row.class)).javaSet();
178 
179     Long[][] expectedIds = {
180       {0L, 1L, 1L, 1L, 2L, 2L, 2L, 4L, 5L, 5L, 5L, 6L, 6L, 6L, 8L, 8L},
181       {1L, 6L, 6L, 6L, 6L, 6L, 6L, 1L, 4L, 4L, 9L, 2L, 5L, 5L, 5L, 5L},
182       {6L, 2L, 5L, 7L, 2L, 5L, 7L, 6L, 1L, 3L, 10L, 6L, 4L, 9L, 4L, 9L}
183     };
184 
185     List<Row> resultList = resultDataSet.collect();
186 
187     assertEquals(expectedIds[0].length, resultList.size());
188 
189     for (Row r : resultList) {
190       assertEquals(3, r.getArity());
191     }
192 
193     resultList.sort((r1, r2) -> {
194       for (int i = 0; i < r1.getArity(); i++) {
195         int comp = ((Long) r1.getField(i)).compareTo((Long) r2.getField(i));
196         if (comp != 0) {
197           return comp;
198         }
199       }
200       return 0;
201     });
202 
203     for (int i = 0; i < expectedIds.length; i++) {
204       assertEquals(expectedIds[0][i], resultList.get(i).getField(0));
205       assertEquals(expectedIds[1][i], resultList.get(i).getField(1));
206       assertEquals(expectedIds[2][i], resultList.get(i).getField(2));
207     }
208   }
