35  /**
36   * Parameterized test-class for {@link PageRankSampling}.
37   */
38  public class PageRankSamplingTest extends ParameterizedTestForGraphSampling {
39  
40    /**
41     * Creates a new PageRankSamplingTest instance, parsing the parameters.
42     *
43     * @param testName Name for test-case
44     * @param seed Seed-value for random number generator, e.g. 0
45     * @param sampleSize Value for sample size, e.g. 0.5
46     * @param dampeningFactor The dampening factor used by Flinks PageRank-algorithm, e.g. 0.85
47     * @param maxIteration The iteration number used by Flinks PageRank-algorithm, e.g. 20
48     * @param sampleGreaterThanThreshold Whether to sample vertices with a PageRank-score
49     *                                   greater (true) or equal/smaller (false) the threshold
50     * @param keepVerticesIfSameScore Whether to sample all vertices (true) or none of them (false)
51     *                                in case all vertices got the same PageRank-score.
52     */
53    public PageRankSamplingTest(String testName, String seed, String sampleSize,
54      String dampeningFactor, String maxIteration, String sampleGreaterThanThreshold,
55      String keepVerticesIfSameScore) {
56      super(testName, Long.parseLong(seed), Float.parseFloat(sampleSize),
57        Double.parseDouble(dampeningFactor), Integer.parseInt(maxIteration),
58        Boolean.parseBoolean(sampleGreaterThanThreshold),
59        Boolean.parseBoolean(keepVerticesIfSameScore));
60    }
61  
62    @Override
63    public SamplingAlgorithm getSamplingOperator() {
64      return new PageRankSampling(dampeningFactor, maxIteration, (double) sampleSize,
65        sampleGreaterThanThreshold, keepVerticesIfSameScore);
66    }
67  
68    @Override
69    public void validateSpecific(LogicalGraph input, LogicalGraph output) {
70  
71      try {
72        // test normal graph
73        GraphHead normalGh = output.getGraphHead().collect().get(0);
74        double minScore = normalGh.getPropertyValue(SamplingConstants.MIN_PAGE_RANK_SCORE_PROPERTY_KEY)
75          .getDouble();
76        double maxScore = normalGh.getPropertyValue(SamplingConstants.MAX_PAGE_RANK_SCORE_PROPERTY_KEY)
77          .getDouble();
78        if (minScore != maxScore) {
79          for (Vertex v : newVertices) {
80            assertTrue("vertex does not have scaled PageRank-score property (should have):" +
81              v.toString(), v.hasProperty(SamplingConstants.SCALED_PAGE_RANK_SCORE_PROPERTY_KEY));
82  
83            if (v.hasProperty(SamplingConstants.SCALED_PAGE_RANK_SCORE_PROPERTY_KEY)) {
84              double score = v.getPropertyValue(SamplingConstants.SCALED_PAGE_RANK_SCORE_PROPERTY_KEY)
85                .getDouble();
86              if (sampleGreaterThanThreshold) {
87                assertTrue("sampled vertex has PageRankScore smaller or equal than threshold",
88                  score > sampleSize);
89              } else {
90                assertTrue("sampled vertex has PageRankScore greater than threshold",
91                  score <= sampleSize);
92              }
93            }
94          }
95        } else {
96          if (keepVerticesIfSameScore) {
97            assertEquals("not all vertices got sampled (should be, all got same score)",
98              dbVertices.size(), newVertices.size());
99          } else {
100           assertTrue("some vertices got sampled (should NOT be, all got same score)",
101             newVertices.isEmpty());
102         }
103       }
104       dbEdges.removeAll(newEdges);
105       for (Edge edge : dbEdges) {
106         assertFalse("edge from original graph was not sampled but source and target were",
107           newVertexIDs.contains(edge.getSourceId()) &&
108             newVertexIDs.contains(edge.getTargetId()));
109       }
110 
111       // test special graph
112       LogicalGraph specialGraph = getLoaderFromString(
113         "(alice:Person {name : \"Alice\"})\n" +
114           "(eve:Person {name : \"Eve\"})\n" +
115           "g0:Community {interest : \"Friends\", vertexCount : 2} [\n" +
116           "(eve)-[eka:knows {since : 2013}]->(alice)\n" +
117           "(alice)-[ake:knows {since : 2013}]->(eve)]").getLogicalGraphByVariable("g0");
118 
119       LogicalGraph sampledGraph = getSamplingOperator().sample(specialGraph);
120 
121       List<Vertex> specialVertices = Lists.newArrayList();
122       List<Vertex> specialSampledVertices = Lists.newArrayList();
123 
124       specialGraph.getVertices().output(new LocalCollectionOutputFormat<>(specialVertices));
125       sampledGraph.getVertices().output(new LocalCollectionOutputFormat<>(specialSampledVertices));
126 
127       getExecutionEnvironment().execute();
128 
129       assertNotNull("graph was null", sampledGraph);
130 
131       GraphHead specialGh = sampledGraph.getGraphHead().collect().get(0);
132       double minScore1 = specialGh.getPropertyValue(
133         SamplingConstants.MIN_PAGE_RANK_SCORE_PROPERTY_KEY).getDouble();
134       double maxScore1 = specialGh.getPropertyValue(
135         SamplingConstants.MAX_PAGE_RANK_SCORE_PROPERTY_KEY).getDouble();
136 
137       assertEquals("min PageRankScore is not equal to max PageRankScore",
138         minScore1, maxScore1, 0.0);
139 
140       if (keepVerticesIfSameScore) {
141         assertEquals(
142           "special: not all vertices got sampled (should be, all got same score)",
143           specialVertices.size(), specialSampledVertices.size());
144       } else {
145         assertTrue(
146           "special: some vertices got sampled (should NOT be, all got same score)",
147           specialSampledVertices.isEmpty());
148       }
149     } catch (Exception e) {
150       e.printStackTrace();
151     }
152   }
