37    public void testVertexLabelSpecific() throws Exception {
38      FlinkAsciiGraphLoader loader = getLoaderFromString(getLabelSpecificInput());
39  
40      LogicalGraph input = loader.getLogicalGraphByVariable("input");
41  
42      loader.appendToDatabaseFromString("expected[" +
43        "(v00:Forum {topic : \"rdf\"})" +
44        "(v01:Forum {topic : \"graph\"})" +
45        "(v02:User  {gender : \"male\"})" +
46        "(v03:User  {gender : \"female\"})" +
47        "(v02)-->(v00)" +
48        "(v02)-->(v01)" +
49        "(v02)-->(v02)" +
50        "(v02)-->(v03)" +
51        "(v03)-->(v01)" +
52        "(v03)-->(v02)" +
53        "]");
54  
55      LogicalGraph output = new Grouping.GroupingBuilder()
56        .useVertexLabel(true)
57        .addVertexGroupingKey("topic")
58        .addVertexLabelGroup("User", Lists.newArrayList("gender"))
59        .setStrategy(getStrategy())
60        .<EPGMGraphHead, EPGMVertex, EPGMEdge, LogicalGraph, GraphCollection>build()
61        .execute(input);
62  
63      collectAndAssertTrue(
64        output.equalsByElementData(loader.getLogicalGraphByVariable("expected")));
65    }
