54  /**
55   * Performs the RandomJump using Gellys VertexCentricIteration (VCI).
56   * Uniformly at random picks {@link #numberOfStartVertices} starting vertices and then simulates a RandomWalk for each
57   * starting vertex on the graph, where once visited edges are not used again. For each walker,
58   * with a given {@link #jumpProbability}, or if the walk ends in a sink, or if all outgoing edges
59   * of the current vertex were visited, randomly jumps to any vertex in the graph and starts a new
60   * RandomWalk from there.
61   * Unlike the RandomWalk algorithm, RandomJump does not have problems of getting stuck or not being
62   * able to visit enough vertices. The algorithm converges when the maximum number of iterations
63   * has been reached, or enough vertices have been visited (with the percentage of vertices to
64   * visit at least given in {@link #percentageToVisit}).
65   * Returns the initial graph with vertices and edges annotated by a boolean property named
66   * "sampled", which is set to {@code true} if visited, or {@code false} if not.
67   */
68  public class KRandomJumpGellyVCI
69    extends BaseGellyAlgorithm<Long, VCIVertexValue, Long, LogicalGraph> {

71     * The graph used in {@link KRandomJumpGellyVCI#execute(LogicalGraph)}.
72     */
73    protected LogicalGraph currentGraph;

76     * Number of starting vertices.
77     */
78    private final int numberOfStartVertices;

81     * Value for maximum number of iterations for the algorithm.
82     */
83    private final int maxIterations;
84  
85    /**
86     * Probability for jumping to a random vertex instead of walking to a random neighbor.
87     */
88    private final double jumpProbability;

105   /**
106    * Creates an instance of KRandomJumpGellyVCI.
107    *
108    * @param numberOfStartVertices Number of starting vertices.
109    * @param maxIterations Value for maximum number of iterations for the algorithm.
110    * @param jumpProbability Probability for jumping to random vertex instead of walking to random
111    *                       neighbor.
112    * @param percentageToVisit Relative amount of vertices to visit at least.
113    */
114   public KRandomJumpGellyVCI(int numberOfStartVertices, int maxIterations, double jumpProbability,
115     double percentageToVisit) {
116     checkArgument(numberOfStartVertices >= 1,
117       "at least 1 starting vertex is needed, numberOfStartVertices must be equal or greater 1");
118     checkArgument(maxIterations > 0,
119       "maxIterations must be greater than 0");
120     checkArgument(jumpProbability >= 0d && jumpProbability <= 1d,
121       "jumpProbability must be equal/greater than 0.0 and smaller/equal 1.0");
122     checkArgument(percentageToVisit > 0d && percentageToVisit <= 1d,
123       "percentageToVisit must be greater than 0.0 and smaller/equal 1.0");
124     this.numberOfStartVertices = numberOfStartVertices;
125     this.maxIterations = maxIterations;
126     this.jumpProbability = jumpProbability;
127     this.percentageToVisit = percentageToVisit;
128   }

130   @Override
131   public Graph<Long, VCIVertexValue, Long> transformToGelly(LogicalGraph graph) {
132     this.currentGraph = graph;
133 
134     indexToVertexIdMap = DataSetUtils.zipWithIndex(graph.getVertices().map(new Id<>()));
135     indexToEdgeIdMap = DataSetUtils.zipWithIndex(graph.getEdges().map(new Id<>()));
136 
137     DataSet<Vertex<Long, VCIVertexValue>> vertices = indexToVertexIdMap
138       .map(new LongIdToGellyVertexWithVCIValueMap());
139 
140     DataSet<Edge<Long, Long>> edges = graph.getEdges()
141       .join(indexToVertexIdMap)
142       .where(new SourceId<>()).equalTo(1)
143       .with(new LongIdWithEdgeToTupleJoin())
144       .join(indexToVertexIdMap)
145       .where(1).equalTo(1)
146       .with(new ReplaceTargetWithLongIdJoin())
147       .join(indexToEdgeIdMap)
148       .where(2).equalTo(1)
149       .with(new LongIdTupleToGellyEdgeWithLongValueJoin());
150 
151     return Graph.fromDataSet(vertices, edges, graph.getConfig().getExecutionEnvironment());
152   }

154   @Override
155   public LogicalGraph executeInGelly(Graph<Long, VCIVertexValue, Long> gellyGraph)
156     throws Exception {
157 
158     long vertexCount = gellyGraph.numberOfVertices();
159 
160     //--------------------------------------------------------------------------
161     // pre compute
162     //--------------------------------------------------------------------------
163 
164     // define start vertices
165     Set<Long> randomStartIndices = new HashSet<>();
166     while (randomStartIndices.size() < numberOfStartVertices) {
167       long randomLongInBounds = (long) (Math.random() * (vertexCount - 1L));
168       randomStartIndices.add(randomLongInBounds);
169     }
170     DataSet<Long> startIndices = currentGraph.getConfig().getExecutionEnvironment()
171       .fromCollection(randomStartIndices);
172 
173     // define how many vertices to visit
174     long verticesToVisit = (long) Math.ceil((double) vertexCount * percentageToVisit);
175 
176     // set compute parameters
177     VertexCentricConfiguration parameters = new VertexCentricConfiguration();
178     parameters.addBroadcastSet(VCIComputeFunction.START_INDICES_BROADCAST_SET, startIndices);
179     parameters.addBroadcastSet(VCIComputeFunction.VERTEX_INDICES_BROADCAST_SET,
180       indexToVertexIdMap.map(new Value0Of2<>()));
181     parameters.registerAggregator(VCIComputeFunction.VISITED_VERTICES_AGGREGATOR_NAME,
182       new LongSumAggregator());
183 
184     // run gelly
185     Graph<Long, VCIVertexValue, Long> resultGraph = gellyGraph.runVertexCentricIteration(
186       new VCIComputeFunction(jumpProbability, verticesToVisit),
187         null, maxIterations, parameters);
188 
189     //--------------------------------------------------------------------------
190     // post compute
191     //--------------------------------------------------------------------------
192 
193     DataSet<GradoopId> visitedGellyEdgeIds = resultGraph.getVertices()
194       .flatMap(new GetVisitedGellyEdgeLongIdsFlatMap())
195       .join(indexToEdgeIdMap)
196       .where("*").equalTo(0)
197       .with(new VisitedGellyEdgesWithLongIdToGradoopIdJoin());
198 
199     // compute new visited edges
200     DataSet<EPGMEdge> visitedEdges = currentGraph.getEdges()
201       .leftOuterJoin(visitedGellyEdgeIds)
202       .where(new Id<>()).equalTo("*")
203       .with(new EPGMEdgeWithGellyEdgeIdJoin(SamplingConstants.PROPERTY_KEY_SAMPLED));
204 
205     DataSet<GradoopId> visitedSourceTargetIds = visitedEdges
206       .flatMap(new GetVisitedSourceTargetIdsFlatMap(SamplingConstants.PROPERTY_KEY_SAMPLED))
207       .distinct();
208 
209     // compute new visited vertices
210     DataSet<EPGMVertex> visitedVertices = resultGraph.getVertices()
211       .join(indexToVertexIdMap)
212       .where(0).equalTo(0)
213       .with(new GellyVertexWithLongIdToGradoopIdJoin())
214       .join(currentGraph.getVertices())
215       .where(0).equalTo(new Id<>())
216       .with(new GellyVertexWithEPGMVertexJoin(SamplingConstants.PROPERTY_KEY_SAMPLED));
217 
218     visitedVertices = visitedVertices.leftOuterJoin(visitedSourceTargetIds)
219       .where(new Id<>()).equalTo("*")
220       .with(new VertexWithVisitedSourceTargetIdJoin(SamplingConstants.PROPERTY_KEY_SAMPLED));
221 
222     // return graph
223     return currentGraph.getFactory().fromDataSets(
224       currentGraph.getGraphHead(), visitedVertices, visitedEdges);
225   }
