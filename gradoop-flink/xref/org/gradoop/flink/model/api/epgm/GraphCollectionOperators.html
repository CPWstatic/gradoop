35  /**
36   * Defines the operators that are available on a {@link GraphCollection}.
37   */
38  public interface GraphCollectionOperators extends GraphBaseOperators {
39  
40    //----------------------------------------------------------------------------
41    // Logical Graph / Graph Head Getters
42    //----------------------------------------------------------------------------
43  
44    /**
45     * Returns logical graph from collection using the given identifier. If the
46     * graph does not exist, an empty logical graph is returned.
47     *
48     * @param graphID graph identifier
49     * @return logical graph with given id or an empty logical graph
50     */
51    LogicalGraph getGraph(final GradoopId graphID);
52    /**
53     * Extracts logical graphs from collection using their identifiers.
54     *
55     * @param identifiers graph identifiers
56     * @return collection containing requested logical graphs
57     */
58    GraphCollection getGraphs(final GradoopId... identifiers);
59  
60    /**
61     * Extracts logical graphs from collection using their identifiers.
62     *
63     * @param identifiers graph identifiers
64     * @return collection containing requested logical graphs
65     */
66    GraphCollection getGraphs(GradoopIdSet identifiers);
67  
68    //----------------------------------------------------------------------------
69    // Unary operators
70    //----------------------------------------------------------------------------
71  
72    /**
73     * Filter containing graphs based on their associated graph head.
74     *
75     * @param predicateFunction predicate function for graph head
76     * @return collection with logical graphs that fulfil the predicate
77     */
78    GraphCollection select(FilterFunction<EPGMGraphHead> predicateFunction);
79  
80    /**
81     * Returns a graph collection that is sorted by a given graph property key.
82     *
83     * @param propertyKey property which is used for comparison
84     * @param order       ascending, descending
85     * @return ordered collection
86     */
87    GraphCollection sortBy(String propertyKey, Order order);
88  
89    /**
90     * Returns the first {@code n} arbitrary logical graphs contained in that
91     * collection.
92     *
93     * @param n number of graphs to return from collection
94     * @return subset of the graph collection
95     */
96    GraphCollection limit(int n);
97  
98    /**
99     * Matches a given pattern on a graph collection.
100    * The boolean flag specifies, if the return shall be the input graphs with
101    * a new property ("contains pattern"), or a new collection consisting of the
102    * constructed embeddings
103    *
104    * @param algorithm         custom pattern matching algorithm
105    * @param pattern           query pattern
106    * @param returnEmbeddings  true -> return embeddings as new collection,
107    *                          false -> return collection with new property
108    * @return  a graph collection containing either the embeddings or the input
109    * graphs with a new property ("contains pattern")
110    */
111   GraphCollection match(
112     String pattern,
113     PatternMatchingAlgorithm algorithm,
114     boolean returnEmbeddings);
115 
116   //----------------------------------------------------------------------------
117   // Binary operators
118   //----------------------------------------------------------------------------
119 
120   /**
121    * Returns a collection with all logical graphs from two input collections.
122    * Graph equality is based on their identifiers.
123    *
124    * @param otherCollection collection to build union with
125    * @return union of both collections
126    */
127   GraphCollection union(GraphCollection otherCollection);
128 
129   /**
130    * Returns a collection with all logical graphs that exist in both input
131    * collections. Graph equality is based on their identifiers.
132    *
133    * @param otherCollection collection to build intersect with
134    * @return intersection of both collections
135    */
136   GraphCollection intersect(GraphCollection otherCollection);
137 
138   /**
139    * Returns a collection with all logical graphs that exist in both input
140    * collections. Graph equality is based on their identifiers.
141    * <p>
142    * Implementation that works faster if {@code otherCollection} is small
143    * (e.g. fits in the workers main memory).
144    *
145    * @param otherCollection collection to build intersect with
146    * @return intersection of both collections
147    */
148   GraphCollection intersectWithSmallResult(
149     GraphCollection otherCollection);
150 
151   /**
152    * Returns a collection with all logical graphs that are contained in that
153    * collection but not in the other. Graph equality is based on their
154    * identifiers.
155    *
156    * @param otherCollection collection to subtract from that collection
157    * @return difference between that and the other collection
158    */
159   GraphCollection difference(GraphCollection otherCollection);
160 
161   /**
162    * Returns a collection with all logical graphs that are contained in that
163    * collection but not in the other. Graph equality is based on their
164    * identifiers.
165    * <p>
166    * Alternate implementation that works faster if the intermediate result
167    * (list of graph identifiers) fits into the workers memory.
168    *
169    * @param otherCollection collection to subtract from that collection
170    * @return difference between that and the other collection
171    */
172   GraphCollection differenceWithSmallResult(
173     GraphCollection otherCollection);
174 
175   /**
176    * Checks, if another collection contains the same graphs as this graph
177    * (by id).
178    *
179    * @param other other graph
180    * @return 1-element dataset containing true, if equal by graph ids
181    */
182   DataSet<Boolean> equalsByGraphIds(GraphCollection other);
183 
184   /**
185    * Checks, if another collection contains the same graphs as this graph
186    * (by vertex and edge ids).
187    *
188    * @param other other graph
189    * @return 1-element dataset containing true, if equal by element ids
190    */
191   DataSet<Boolean> equalsByGraphElementIds(GraphCollection other);
192 
193   /**
194    * Returns a 1-element dataset containing a {@code boolean} value which
195    * indicates if the graph collection is equal to the given graph collection.
196    *
197    * Equality is defined on the element data contained inside the collection,
198    * i.e. vertices and edges.
199    *
200    * @param other graph collection to compare with
201    * @return  1-element dataset containing {@code true} if the two collections
202    *          are equal or {@code false} if not
203    */
204   DataSet<Boolean> equalsByGraphElementData(GraphCollection other);
205 
206   /**
207    * Returns a 1-element dataset containing a {@code boolean} value which
208    * indicates if the graph collection is equal to the given graph collection.
209    *
210    * Equality is defined on the data contained inside the collection, i.e.
211    * graph heads, vertices and edges.
212    *
213    * @param other graph collection to compare with
214    * @return  1-element dataset containing {@code true} if the two collections
215    *          are equal or {@code false} if not
216    */
217   DataSet<Boolean> equalsByGraphData(GraphCollection other);
218 
219   //----------------------------------------------------------------------------
220   // Auxiliary operators
221   //----------------------------------------------------------------------------
222 
223   /**
224    * Calls the given binary collection to collection operator using that
225    * graph and the input graph.
226    *
227    * @param op              binary collection to collection operator
228    * @param otherCollection second input collection for operator
229    * @return result of given operator
230    */
231   GraphCollection callForCollection(
232     BinaryCollectionToCollectionOperator op,
233     GraphCollection otherCollection);
234 
235   /**
236    * Calls the given unary collection to graph operator for the collection.
237    *
238    * @param op unary collection to graph operator
239    * @return result of given operator
240    */
241   LogicalGraph callForGraph(
242     UnaryCollectionToGraphOperator op);
243 
244   /**
245    * Transforms a graph collection into a logical graph by applying a
246    * {@link ReducibleBinaryGraphToGraphOperator} pairwise on the elements of the
247    * collection.
248    *
249    * @param op reducible binary graph to graph operator
250    * @return logical graph
251    *
252    * @see Exclusion
253    * @see Overlap
254    * @see Combination
255    */
256   LogicalGraph reduce(ReducibleBinaryGraphToGraphOperator op);
257 
258   /**
259    * Returns a distinct collection of logical graphs.
260    * Graph equality is based on graph identifiers.
261    *
262    * @return distinct graph collection
263    */
264   GraphCollection distinctById();
265 
266   /**
267    * Groups a graph collection by isomorphism.
268    * Graph equality is based on isomorphism including labels and properties.
269    *
270    * @return distinct graph collection
271    */
272   GraphCollection distinctByIsomorphism();
273 
274   /**
275    * Groups a graph collection by isomorphism including labels and values.
276    *
277    * @param func function to reduce all graph heads of a group into a single representative one,
278    *             e.g., to count the number of group members
279    *
280    * @return grouped graph collection
281    */
282   GraphCollection groupByIsomorphism(GraphHeadReduceFunction func);
283 }
