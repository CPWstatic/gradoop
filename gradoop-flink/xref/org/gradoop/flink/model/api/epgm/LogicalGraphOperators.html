33  /**
34   * Defines the operators that are available on a {@link LogicalGraph}.
35   */
36  public interface LogicalGraphOperators
37    extends BaseGraphOperators<EPGMGraphHead, EPGMVertex, EPGMEdge, LogicalGraph, GraphCollection> {
38  
39    //----------------------------------------------------------------------------
40    // Unary Operators
41    //----------------------------------------------------------------------------
42  
43    /**
44     * Evaluates the given cypher query using CAPF (Cypher for Apache Flink). CAPF implements the
45     * default cypher morphism strategies, which is vertex homomorphism and edge isomorphism.
46     * The result is a {@link CAPFQueryResult}, containing a flink table that can be converted to a
47     * {@link GraphCollection}, if it contains vertices or edges.
48     *
49     * @param query    the query string
50     * @param metaData metaData object
51     * @return the result, containing a flink table and possibly a GraphCollection
52     * @throws Exception on failure
53     */
54    CAPFQueryResult cypher(String query, MetaData metaData) throws Exception;
55  
56    /**
57     * Evaluates the given cypher query using CAPF (Cypher for Apache Flink). CAPF implements the
58     * default cypher morphism strategies, which is vertex homomorphism and edge isomorphism.
59     * The result is a CAPFQueryResult, containing a flink table that can be converted to a GraphCollection,
60     * if it contains vertices or edges.
61     * <p>
62     * In this overloaded function, the property maps are constructed automatically. This is
63     * a lot slower and actually requires the job to be split in two parts to collect the property maps.
64     *
65     * @param query the query string
66     * @return the result, containing a flink table and possibly a GraphCollection
67     * @throws Exception on failure
68     */
69    CAPFQueryResult cypher(String query) throws Exception;
70  
71    /**
72     * Creates a new graph from a randomly chosen subset of nodes and their associated edges.
73     *
74     * @param algorithm used sampling algorithm
75     * @return logical graph with random nodes and their associated edges
76     */
77    LogicalGraph sample(SamplingAlgorithm algorithm);
78  
79    /**
80     * Generates all combinations of the supplied vertex grouping keys according to the definition of
81     * the rollUp operation in SQL and uses them together with all edge grouping keys for separate
82     * grouping operations. For example, specifying the vertex grouping keys A, B and C leads to
83     * three differently grouped graphs {A,B,C},{A,B},{A} within the resulting graph collection.
84     *
85     * @param vertexGroupingKeys       grouping keys to group vertices
86     * @param vertexAggregateFunctions aggregate functions to apply on super vertices
87     * @param edgeGroupingKeys         grouping keys to group edges
88     * @param edgeAggregateFunctions   aggregate functions to apply on super edges
89     * @return graph collection containing all resulting graphs
90     */
91    GraphCollection groupVerticesByRollUp(
92      List<String> vertexGroupingKeys, List<AggregateFunction> vertexAggregateFunctions,
93      List<String> edgeGroupingKeys, List<AggregateFunction> edgeAggregateFunctions);
94  
95    /**
96     * Generates all combinations of the supplied edge grouping keys according to the definition of
97     * the rollUp operation in SQL and uses them together with all vertex grouping keys for separate
98     * grouping operations. For example, specifying the edge grouping keys A, B and C leads to
99     * three differently grouped graphs {A,B,C},{A,B},{A} within the resulting graph collection.
100    *
101    * @param vertexGroupingKeys       grouping keys to group vertices
102    * @param vertexAggregateFunctions aggregate functions to apply on super vertices
103    * @param edgeGroupingKeys         grouping keys to group edges
104    * @param edgeAggregateFunctions   aggregate functions to apply on super edges
105    * @return graph collection containing all resulting graphs
106    */
107   GraphCollection groupEdgesByRollUp(
108     List<String> vertexGroupingKeys, List<AggregateFunction> vertexAggregateFunctions,
109     List<String> edgeGroupingKeys, List<AggregateFunction> edgeAggregateFunctions);
110 
111   //----------------------------------------------------------------------------
112   // Auxiliary Operators
113   //----------------------------------------------------------------------------
114 
115   /**
116    * Splits the graph into multiple logical graphs using the property value
117    * which is assigned to the given property key. Vertices and edges that do
118    * not have this property will be removed from the resulting collection.
119    *
120    * @param propertyKey split property key
121    * @return graph collection
122    */
123   GraphCollection splitBy(String propertyKey);
124 
125   /**
126    * Creates a logical graph from that graph and other graphs using the given operator.
127    *
128    * @param operator    multi graph to graph operator
129    * @param otherGraphs other graphs
130    * @return result of given operator
131    */
132   LogicalGraph callForGraph(GraphsToGraphOperator operator, LogicalGraph... otherGraphs);
133 
134   /**
135      * Writes the graph to given data sink.
136      *
137      * @param dataSink The data sink to which the graph should be written.
138      * @throws IOException if the graph can't be written to the sink
139      */
140   void writeTo(DataSink dataSink) throws IOException;
141 
142   /**
143      * Writes the graph to given data sink with an optional overwrite option.
144      *
145      * @param dataSink The data sink to which the graph should be written.
146      * @param overWrite determines whether existing files are overwritten
147      * @throws IOException if the graph can't be written to the sink
148      */
149   void writeTo(DataSink dataSink, boolean overWrite) throws IOException;
150 }
