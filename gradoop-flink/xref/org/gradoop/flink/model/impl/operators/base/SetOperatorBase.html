35  /**
36   * Base class for set operations that share common methods to build vertex,
37   * edge and data sets.
38   *
39   * @see Difference
40   * @see Intersection
41   * @see Union
42   */
43  public abstract class SetOperatorBase extends
44    BinaryCollectionToCollectionOperatorBase {
45  
46    /**
47     * Computes new vertices based on the new subgraphs. For each vertex, each
48     * graph is collected in a flatMap function and then joined with the new
49     * subgraph dataset.
50     *
51     * @param newGraphHeads graph dataset of the resulting graph collection
52     * @return vertex set of the resulting graph collection
53     */
54    @Override
55    protected DataSet<EPGMVertex> computeNewVertices(
56      DataSet<EPGMGraphHead> newGraphHeads) {
57  
58      DataSet<Tuple2<EPGMVertex, GradoopId>> verticesWithGraphs =
59        firstCollection.getVertices().flatMap(new PairVertexWithGraphs<>());
60  
61      return verticesWithGraphs
62        .join(newGraphHeads)
63        .where(1)
64        .equalTo(new Id<>())
65        .with(new LeftJoin0OfTuple2<>())
66        .withForwardedFieldsFirst("f0->*")
67        .distinct(new Id<>());
68    }
69  
70    /**
71     * Constructs new edges by joining the edges of the first graph with the new
72     * vertices.
73     *
74     * @param newVertices vertex set of the resulting graph collection
75     * @return edges set only connect vertices in {@code newVertices}
76     * @see Difference
77     * @see Intersection
78     */
79    @Override
80    protected DataSet<EPGMEdge> computeNewEdges(DataSet<EPGMVertex> newVertices) {
81      return firstCollection.getEdges().join(newVertices)
82        .where(new SourceId<>())
83        .equalTo(new Id<>())
84        .with(new LeftSide<>())
85        .join(newVertices)
86        .where(new TargetId<>())
87        .equalTo(new Id<>())
88        .with(new LeftSide<>())
89        .distinct(new Id<>());
90    }
91  }
