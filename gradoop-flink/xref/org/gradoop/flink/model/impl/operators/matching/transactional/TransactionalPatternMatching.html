50  /**
51   * Operator to match a given pattern on a graph collection
52   */
53  public class TransactionalPatternMatching implements UnaryCollectionToCollectionOperator {
54    /**
55     * Query Pattern
56     */
57    private String query;
58    /**
59     * Actual used algorithm
60     */
61    private PatternMatchingAlgorithm algorithm;
62  
63    /**
64     * Flag that determines what the result consists of:
65     * false: source graphs with an new property that is true iff the pattern
66     *        was found in this graph
67     * true:  one graph per found embedding
68     */
69    private boolean findEmbeddings;
70  
71    /**
72     * Constructor
73     *
74     * @param algorithm pattern matching algorithm
75     * @param query     given query-pattern
76     * @param findEmbeddings  flag that determines the type of the return
77     *                        false:  source graphs with an new property that is
78     *                                true iff the pattern was found in this graph
79     *                        true:   one graph per found embedding
80     */
81    public TransactionalPatternMatching(
82      String query,
83      PatternMatchingAlgorithm algorithm,
84      boolean findEmbeddings) {
85      this.query = query;
86      this.algorithm = algorithm;
87      this.findEmbeddings = findEmbeddings;
88    }
89  
90    @Override
91    public GraphCollection execute(GraphCollection collection) {
92  
93      //--------------------------------------------------------------------------
94      // generate graph-id set witch will be used for generating mappings
95      //--------------------------------------------------------------------------
96      DataSet<GradoopId> graphIds = collection.getGraphHeads().map(new Id<>());
97  
98      //--------------------------------------------------------------------------
99      // generate mapping from graph-id to vertex candidates
100     //--------------------------------------------------------------------------
101     DataSet<Tuple2<GradoopId, IdWithCandidates<GradoopId>>> vertexCandidatesWithGraphs =
102       collection.getVertices()
103         .filter(new MatchingVertices<>(query))
104         .map(new BuildIdWithCandidatesAndGraphs<>(query))
105         .flatMap(new ExpandFirstField<>())
106         .join(graphIds)
107         .where(0).equalTo("*")
108         .with(new LeftSide<>());
109 
110     //--------------------------------------------------------------------------
111     // generate mapping from graph-id to edge candidates
112     //--------------------------------------------------------------------------
113     DataSet<Tuple2<GradoopId, TripleWithCandidates<GradoopId>>> edgeCandidatesWithGraphs =
114       collection.getEdges()
115         .filter(new MatchingEdges<>(query))
116         .map(new BuildTripleWithCandidatesAndGraphs<>(query))
117         .flatMap(new ExpandFirstField<>())
118         .join(graphIds)
119         .where(0).equalTo("*")
120         .with(new LeftSide<>());
121 
122     //--------------------------------------------------------------------------
123     // generate graphs with the candidates for their elements
124     //--------------------------------------------------------------------------
125     DataSet<GraphWithCandidates> graphs = vertexCandidatesWithGraphs
126       .coGroup(edgeCandidatesWithGraphs)
127       .where(0).equalTo(0)
128       .with(new BuildGraphWithCandidates());
129 
130     if (findEmbeddings) {
131       return findEmbeddings(collection, graphs);
132     } else {
133       return hasEmbeddings(collection, graphs);
134     }
135   }
136 
137   /**
138    * Returns the input graph collection with a new property for each graph, that
139    * states if it contains the embedding.
140    * @param collection input graph collection
141    * @param graphs graphs with candidates of their elements
142    * @return input graph collection with new boolean property
143    */
144   private GraphCollection hasEmbeddings(GraphCollection collection,
145     DataSet<GraphWithCandidates> graphs) {
146     //--------------------------------------------------------------------------
147     // run matching algorithm
148     //--------------------------------------------------------------------------
149     DataSet<Tuple2<GradoopId, Boolean>> matches = graphs.map(new HasEmbeddings(algorithm, query));
150 
151     //--------------------------------------------------------------------------
152     // join matches to graph heads
153     //--------------------------------------------------------------------------
154     DataSet<GraphHead> newHeads = collection.getGraphHeads()
155       .coGroup(matches)
156       .where(new Id<>()).equalTo(0)
157       .with(new AddMatchesToProperties());
158 
159     //--------------------------------------------------------------------------
160     // return updated graph collection
161     //--------------------------------------------------------------------------
162     return collection.getConfig().getGraphCollectionFactory().fromDataSets(
163       newHeads, collection.getVertices(), collection.getEdges());
164   }
165 
166 
167   /**
168    * Finds all embeddings in the given graph and constructs a new graph
169    * collection consisting of these embeddings.
170    * @param collection input graph collection
171    * @param graphs graphs with candidates of their elements
172    * @return collection of found embeddings
173    */
174   private GraphCollection findEmbeddings(GraphCollection collection,
175     DataSet<GraphWithCandidates> graphs) {
176 
177     //--------------------------------------------------------------------------
178     // run the matching algorithm
179     //--------------------------------------------------------------------------
180     DataSet<Tuple4<GradoopId, GradoopId, GradoopIdSet, GradoopIdSet>> embeddings = graphs
181       .flatMap(new FindEmbeddings(algorithm, query));
182 
183     //--------------------------------------------------------------------------
184     // create new graph heads
185     //--------------------------------------------------------------------------
186     DataSet<GraphHead> newHeads = embeddings
187       .map(new Project4To0And1<>())
188       .map(new InitGraphHeadWithLineage(collection.getConfig().getGraphHeadFactory()));
189 
190     //--------------------------------------------------------------------------
191     // update vertex graphs
192     //--------------------------------------------------------------------------
193     DataSet<Tuple2<GradoopId, GradoopIdSet>> verticesWithGraphs = embeddings
194       .map(new Project4To0And2AndSwitch<>())
195       .flatMap(new ExpandFirstField<>()).groupBy(0)
196       .reduceGroup(new MergeSecondField<>());
197 
198     DataSet<Vertex> newVertices = verticesWithGraphs
199       .join(collection.getVertices())
200       .where(0).equalTo(new Id<>())
201       .with(new AddGraphsToElements<>());
202 
203     //--------------------------------------------------------------------------
204     // update edge graphs
205     //--------------------------------------------------------------------------
206     DataSet<Tuple2<GradoopId, GradoopIdSet>> edgesWithGraphs = embeddings
207       .map(new Project4To0And3AndSwitch<>())
208       .flatMap(new ExpandFirstField<>()).groupBy(0)
209       .reduceGroup(new MergeSecondField<>());
210 
211     DataSet<Edge> newEdges = edgesWithGraphs
212       .join(collection.getEdges())
213       .where(0).equalTo(new Id<>())
214       .with(new AddGraphsToElements<>());
215 
216     //--------------------------------------------------------------------------
217     // return the embeddings
218     //--------------------------------------------------------------------------
219     return collection.getConfig().getGraphCollectionFactory()
220       .fromDataSets(newHeads, newVertices, newEdges);
221   }
