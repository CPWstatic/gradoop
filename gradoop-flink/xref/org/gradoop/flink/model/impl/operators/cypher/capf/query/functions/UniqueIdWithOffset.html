16  package org.gradoop.flink.model.impl.operators.cypher.capf.query.functions;
17  
18  import org.apache.flink.api.common.functions.RichMapFunction;
19  import org.apache.flink.api.common.functions.RuntimeContext;
20  import org.apache.flink.api.java.tuple.Tuple2;
21  import org.apache.flink.configuration.Configuration;
22  
23  import java.util.List;
24  
25  /**
26   * Zip each element in the dataset with a unique Long id. Its possible to set an offset via a
27   * broadcast set.
28   *
29   * @param <E> the type of elements in the dataset
30   */
31  
32  public class UniqueIdWithOffset<E> extends RichMapFunction<E, Tuple2<Long, E>> {
33  
34    /**
35     * The parallelism this MapFunction is run at.
36     */
37    private Long parallelism;
38  
39    /**
40     * A counter to make sure each id is only given once.
41     */
42    private Long idCounter;
43  
44    /**
45     * Reduce object instantiations
46     */
47    private Tuple2<Long, E> returnTuple = new Tuple2<>();
48  
49    @Override
50    public void open(Configuration parameters) {
51  
52      idCounter = 0L;
53  
54      RuntimeContext ctx = getRuntimeContext();
55      // if an offset has been set, add it to the idCounter
56      if (ctx.hasBroadcastVariable("offset")) {
57        List<Long> offset = ctx.getBroadcastVariable("offset");
58        if (offset.size() > 0) {
59          idCounter += offset.get(0);
60        }
61      }
62      parallelism = (long) ctx.getNumberOfParallelSubtasks();
63      idCounter += (long) ctx.getIndexOfThisSubtask();
64    }
65  
66    @Override
67    public Tuple2<Long, E> map(E e) throws Exception {
68      returnTuple.f0 = idCounter;
69      returnTuple.f1 = e;
70      idCounter += parallelism;
71      return returnTuple;
72    }
73  }
