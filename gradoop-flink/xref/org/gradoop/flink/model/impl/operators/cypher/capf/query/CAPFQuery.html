138   /**
139    * Execute a cypher query on a given graph via the CAPF API.
140    *
141    * @param graph the graph that the query shall be executed on
142    * @return the result of the query, either a graph collection or a flink table
143    * @throws Exception if the execution or IO fails.
144    */
145   public CAPFQueryResult execute(LogicalGraph graph) throws Exception {
146 
147     if (metaData == null) {
148       graph = transformGraphProperties(graph);
149       metaData = new CSVMetaDataSource()
150         .fromTuples(new CSVMetaDataSource().tuplesFromGraph(graph).collect());
151     }
152     // create flink tables of nodes as required by CAPF
153     List<CAPFNodeTable> nodeTables = createNodeTables(graph);
154 
155     // create flink tables of relationships as required by CAPF
156     List<CAPFRelationshipTable> relTables = createRelationshipTables(graph);
157 
158     // if there are no nodes, no edges can exit either, so we can terminate early
159     if (nodeTables.size() > 0) {
160       List<EntityTable<FlinkCypherTable.FlinkTable>> tables = new ArrayList<>(
161         nodeTables.subList(1, nodeTables.size()));
162       tables.addAll(relTables);
163 
164       Seq<EntityTable<FlinkCypherTable.FlinkTable>> tableSeq =
165         JavaConversions.asScalaBuffer(tables);
166 
167       PropertyGraph g = session.readFrom(nodeTables.get(0), tableSeq);
168 
169       // construct a CAPFQueryResult from the CAPFResult returned by CAPF
170       return new CAPFQueryResult(
171         g.cypher(
172           query,
173           g.cypher$default$2(),
174           g.cypher$default$3(),
175           g.cypher$default$4()
176         ),
177         verticesWithIds,
178         edgesWithIds,
179         graph.getConfig()
180       );
181     }
182 
183     return null;
184   }

186   /**
187    * Transform vertex and edge properties with types not yet supported by CAPF into string
188    * representations.
189    *
190    * @param graph the graph
191    * @return a graph with transformed vertex and edge properties
192    */
193   private LogicalGraph transformGraphProperties(LogicalGraph graph) {
194     DataSet<EPGMVertex> transformedVertices = graph.getVertices()
195       .map(new PropertyEncoder<>());
196     DataSet<EPGMEdge> transformedEdges = graph.getEdges()
197       .map(new PropertyEncoder<>());
198 
199     return graph.getFactory().fromDataSets(transformedVertices, transformedEdges);
200   }

202   /**
203    * Method to transform a DataSet of vertices into the flink table format
204    * required by CAPF: Unique long ids for each vertex, one table per
205    * vertex label and each property in a unique row field.
206    *
207    * @param graph the graph whose vertices should be transformed into CAPF tables
208    * @return a list of node tables, one table per vertex label
209    */
210   private List<CAPFNodeTable> createNodeTables(LogicalGraph graph) {
211     List<CAPFNodeTable> nodeTables = new ArrayList<>();
212 
213     verticesWithIds = graph.getVertices().map(new UniqueIdWithOffset<>());
214     vertexCount = Count.count(graph.getVertices());
215 
216     // construct a table for each vertex label
217     for (String label : metaData.getVertexLabels()) {
218       List<PropertyMetaData> propertyTypes = metaData.getVertexPropertyMetaData(label);
219 
220       // list of all row field types
221       TypeInformation<?>[] types = new TypeInformation<?>[propertyTypes.size() + 1];
222       List<String> propKeys = new ArrayList<>(propertyTypes.size());
223 
224       // first field is long id
225       types[0] = TypeInformation.of(Long.class);
226 
227       for (int i = 0; i < propertyTypes.size(); i++) {
228         PropertyMetaData pmd = propertyTypes.get(i);
229         propKeys.add(pmd.getKey());
230         types[i + 1] = TypeInformation.of(MetaData.getClassFromTypeString(pmd.getTypeString()));
231       }
232 
233       RowTypeInfo info = new RowTypeInfo(types);
234 
235       // zip all vertices of one label with a globally unique id
236       DataSet<Tuple2<Long, EPGMVertex>> verticesByLabelWithIds =
237         verticesWithIds.filter(new VertexLabelFilter(label));
238 
239       // map vertices to row and wrap in scala DataSet
240       org.apache.flink.api.scala.DataSet<Row> scalaRowDataSet =
241         new org.apache.flink.api.scala.DataSet<>(
242           verticesByLabelWithIds.map(new VertexToRow(propKeys)).returns(info),
243           ClassTag$.MODULE$.apply(Row.class)
244         );
245 
246       // build table schema string, naming each field in the table
247       StringBuilder schemaStringBuilder = new StringBuilder(NODE_ID);
248       NodeMapping nodeMapping = NodeMapping.withSourceIdKey(NODE_ID)
249         .withImpliedLabel(label);
250 
251       for (String propKey : propKeys) {
252         schemaStringBuilder.append(", ").append(PROPERTY_PREFIX).append(propKey);
253 
254         nodeMapping = nodeMapping.withPropertyKey(propKey, PROPERTY_PREFIX + propKey);
255       }
256 
257       String schemaString = schemaStringBuilder.toString();
258 
259       // create table, add to node table list
260       Table vertexTable = session.tableEnv()
261         .fromDataSet(scalaRowDataSet).as(schemaString);
262 
263       nodeTables.add(CAPFNodeTable.fromMapping(nodeMapping, vertexTable));
264     }
265 
266     return nodeTables;
267   }

269   /**
270    * Method to transform a DataSet of edges into the flink table format
271    * required by CAPF: Unique long ids for each edge, source and target are long ids,
272    * one table per edge label and each property in a unique row field.
273    *
274    * @param graph the graph whose edges should be transformed into CAPF tables
275    * @return a list of edge tables, one table per edge label
276    */
277   private List<CAPFRelationshipTable> createRelationshipTables(LogicalGraph graph) {
278     List<CAPFRelationshipTable> relTables = new ArrayList<>();
279 
280     edgesWithIds = graph.getEdges().map(new UniqueIdWithOffset<>())
281       .withBroadcastSet(vertexCount, OFFSET);
282 
283     // replace source and target with long ids
284     DataSet<Tuple5<Long, Long, Long, String, Properties>> edgeTuples = edgesWithIds
285       .map(new EdgeToTuple())
286       .join(verticesWithIds)
287       .where(1).equalTo(new IdOfF1<>()).with(new ReplaceSourceId())
288       .join(verticesWithIds)
289       .where(2).equalTo(new IdOfF1<>()).with(new ReplaceTargetId());
290 
291     // construct a table for each edge label
292     for (String label : metaData.getEdgeLabels()) {
293       List<PropertyMetaData> propertyTypes = metaData.getEdgePropertyMetaData(label);
294 
295       // list of all row field types
296       TypeInformation<?>[] types = new TypeInformation<?>[propertyTypes.size() + 3];
297       List<String> propKeys = new ArrayList<>(propertyTypes.size());
298 
299       // first fields are id, source id and target id
300       types[0] = TypeInformation.of(Long.class); // id
301       types[1] = TypeInformation.of(Long.class); // source
302       types[2] = TypeInformation.of(Long.class); // target
303 
304       // other fields are properties
305       for (int i = 0; i < propertyTypes.size(); i++) {
306         PropertyMetaData pmd = propertyTypes.get(i);
307         propKeys.add(pmd.getKey());
308         types[i + 3] = TypeInformation.of(MetaData.getClassFromTypeString(pmd.getTypeString()));
309       }
310 
311       RowTypeInfo info = new RowTypeInfo(types);
312 
313       // zip all edges of one label with a globally unique id
314       DataSet<Tuple5<Long, Long, Long, String, Properties>> edgesByLabel = edgeTuples
315         .filter(new EdgeLabelFilter(label));
316 
317       // map vertices to row and wrap in scala DataSet
318       org.apache.flink.api.scala.DataSet<Row> scalaRowDataSet =
319         new org.apache.flink.api.scala.DataSet<>(
320           edgesByLabel.map(new TupleToRow(propKeys)).returns(info),
321           ClassTag$.MODULE$.apply(Row.class)
322         );
323 
324       // build table schema string, naming each field in the table
325       StringBuilder schemaStringBuilder = new StringBuilder();
326       schemaStringBuilder
327         .append(EDGE_ID).append(", ")
328         .append(START_NODE).append(", ")
329         .append(END_NODE);
330 
331       RelationshipMapping relMapping = RelationshipMapping.withSourceIdKey(EDGE_ID)
332         .withSourceStartNodeKey(START_NODE)
333         .withSourceEndNodeKey(END_NODE)
334         .withRelType(label);
335 
336       for (String propKey : propKeys) {
337         schemaStringBuilder.append(", ").append(PROPERTY_PREFIX).append(propKey);
338         relMapping = relMapping.withPropertyKey(propKey, PROPERTY_PREFIX + propKey);
339       }
340 
341       String schemaString = schemaStringBuilder.toString();
342 
343       // create table, add to relationship table list
344       Table edgeTable = session.tableEnv()
345         .fromDataSet(scalaRowDataSet).as(schemaString);
346 
347       relTables.add(CAPFRelationshipTable.fromMapping(relMapping, edgeTable));
348     }
349 
350     return relTables;
351   }
