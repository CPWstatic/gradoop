42  /**
43   * Provides methods for post-processing query results.
44   */
45  public class PostProcessor {
46  
47    /**
48     * Extracts a {@link GraphCollection} from a set of {@link EPGMElement}.
49     *
50     * @param elements  EPGM elements
51     * @param config    Gradoop Flink config
52     * @return Graph collection
53     */
54    public static GraphCollection extractGraphCollection(
55      DataSet<EPGMElement> elements, GradoopFlinkConfig config) {
56      return extractGraphCollection(elements, config, true);
57    }
58  
59    /**
60     * Extracts a {@link GraphCollection} from a set of {@link EPGMElement}.
61     *
62     * @param elements  EPGM elements
63     * @param config        Gradoop Flink config
64     * @param mayOverlap    elements may be contained in multiple graphs
65     * @return Graph collection
66     */
67    public static GraphCollection extractGraphCollection(
68      DataSet<EPGMElement> elements, GradoopFlinkConfig config, boolean mayOverlap) {
69  
70      Class<EPGMGraphHead> graphHeadType = config.getGraphHeadFactory().getType();
71      Class<EPGMVertex> vertexType = config.getVertexFactory().getType();
72      Class<EPGMEdge> edgeType = config.getEdgeFactory().getType();
73      return config.getGraphCollectionFactory().fromDataSets(
74        extractGraphHeads(elements, graphHeadType),
75        extractVertices(elements, vertexType, mayOverlap),
76        extractEdges(elements, edgeType, mayOverlap)
77      );
78    }
79  
80    /**
81     * Extracts a {@link GraphCollection} from a set of {@link EPGMElement} and
82     * attaches the original data from the input {@link LogicalGraph}.
83     *
84     * @param elements      EPGM elements
85     * @param inputGraph    original input graph
86     * @param mayOverlap    true, if elements may be contained in multiple graphs
87     * @return Graph collection
88     */
89    public static GraphCollection extractGraphCollectionWithData(
90      DataSet<EPGMElement> elements, LogicalGraph inputGraph, boolean mayOverlap) {
91  
92      GradoopFlinkConfig config = inputGraph.getConfig();
93  
94      // get result collection without data
95      GraphCollection collection = extractGraphCollection(elements, config, mayOverlap);
96  
97      // attach data by joining first and merging the graph head ids
98      DataSet<EPGMVertex> newVertices = inputGraph.getVertices()
99        .rightOuterJoin(collection.getVertices())
100       .where(new Id<>()).equalTo(new Id<>())
101       .with(new MergedGraphIds<>())
102       .withForwardedFieldsFirst("id;label;properties;");
103 
104 
105     DataSet<EPGMEdge> newEdges = inputGraph.getEdges()
106       .rightOuterJoin(collection.getEdges())
107       .where(new Id<>()).equalTo(new Id<>())
108       .with(new MergedGraphIds<>())
109       .withForwardedFieldsFirst("id;label;properties");
110 
111     return config.getGraphCollectionFactory().fromDataSets(
112       collection.getGraphHeads(), newVertices, newEdges);
113   }
114   /**
115    * Extracts vertex ids from the given pattern matching result.
116    *
117    * @param result pattern matching result
118    * @return dataset with (vertexId) tuples
119    */
120   public static DataSet<Tuple1<GradoopId>> extractVertexIds(
121     DataSet<FatVertex> result) {
122     return result.project(0);
123   }
124 
125   /**
126    * Extracts edge-source-target id triples from the given pattern matching
127    * result.
128    *
129    * @param result pattern matching result
130    * @return dataset with (edgeId, sourceId, targetId) tuples
131    */
132   public static DataSet<Tuple3<GradoopId, GradoopId, GradoopId>> extractEdgeIds(
133     DataSet<FatVertex> result) {
134     return result.flatMap(new EdgeTriple());
135   }

137   /**
138    * Filters and casts EPGM graph heads from a given set of {@link EPGMElement}
139    *
140    * @param elements      EPGM elements
141    * @param graphHeadType graph head type
142    * @return EPGM graph heads
143    */
144   public static DataSet<EPGMGraphHead> extractGraphHeads(DataSet<EPGMElement> elements,
145     Class<EPGMGraphHead> graphHeadType) {
146     return elements
147       .filter(new IsInstance<>(graphHeadType))
148       .map(new Cast<>(graphHeadType))
149       .returns(TypeExtractor.createTypeInfo(graphHeadType));
150   }
151 
152   /**
153    * Initializes EPGM vertices from the given pattern matching result.
154    *
155    * @param result        pattern matching result
156    * @param epgmVertexFactory EPGM vertex factory
157    * @return EPGM vertices
158    */
159   public static DataSet<EPGMVertex> extractVertices(DataSet<FatVertex> result,
160                                                 VertexFactory<EPGMVertex> epgmVertexFactory) {
161     return extractVertexIds(result).map(new VertexFromId(epgmVertexFactory));
162   }
163 
164   /**
165    * Filters and casts EPGM vertices from a given set of {@link EPGMElement}
166    *
167    * @param elements  EPGM elements
168    * @param vertexType    vertex type
169    * @param mayOverlap    vertices may be contained in multiple graphs
170    * @return EPGM vertices
171    */
172   public static DataSet<EPGMVertex> extractVertices(DataSet<EPGMElement> elements,
173     Class<EPGMVertex> vertexType, boolean mayOverlap) {
174     DataSet<EPGMVertex> result = elements
175       .filter(new IsInstance<>(vertexType))
176       .map(new Cast<>(vertexType))
177       .returns(TypeExtractor.createTypeInfo(vertexType));
178     return mayOverlap ? result
179       .groupBy(new Id<>())
180       .combineGroup(new MergedGraphIds<>())
181       .groupBy(new Id<>())
182       .reduceGroup(new MergedGraphIds<>()) : result;
183   }
184 
185   /**
186    * Initializes EPGM edges from the given pattern matching result.
187    *
188    * @param result      pattern matching result
189    * @param epgmEdgeFactory EPGM edge factory
190    * @return EPGM edges
191    */
192   public static DataSet<EPGMEdge> extractEdges(DataSet<FatVertex> result,
193     EdgeFactory<EPGMEdge> epgmEdgeFactory) {
194     return extractEdgeIds(result).map(new EdgeFromIds(epgmEdgeFactory));
195   }
196 
197   /**
198    * Filters and casts EPGM edges from a given set of {@link EPGMElement}
199    *
200    * @param elements      EPGM elements
201    * @param edgeType      edge type
202    * @param mayOverlap    edges may be contained in multiple graphs
203    * @return EPGM edges
204    */
205   public static DataSet<EPGMEdge> extractEdges(DataSet<EPGMElement> elements,
206     Class<EPGMEdge> edgeType, boolean mayOverlap) {
207     DataSet<EPGMEdge> result = elements
208       .filter(new IsInstance<>(edgeType))
209       .map(new Cast<>(edgeType))
210       .returns(TypeExtractor.createTypeInfo(edgeType));
211 
212     return mayOverlap ? result
213       .groupBy(new Id<>())
214       .combineGroup(new MergedGraphIds<>()).groupBy(new Id<>())
215       .reduceGroup(new MergedGraphIds<>()) : result;
216   }
217 
218   /**
219    * Initializes EPGM vertices including their original data from the given
220    * pattern matching result.
221    *
222    * @param result        pattern matching result
223    * @param inputVertices original data graph vertices
224    * @return EPGM vertices including data
225    */
226   public static DataSet<EPGMVertex> extractVerticesWithData(
227     DataSet<FatVertex> result, DataSet<EPGMVertex> inputVertices) {
228     return extractVertexIds(result)
229       .join(inputVertices)
230       .where(0).equalTo(new Id<>())
231       .with(new RightSide<>());
232   }
233 
234   /**
235    * Initializes EPGM edges including their original data from the given pattern
236    * matching result.
237    *
238    * @param result      pattern matching result
239    * @param inputEdges  original data graph edges
240    * @return EPGM edges including data
241    */
242   public static DataSet<EPGMEdge> extractEdgesWithData(DataSet<FatVertex> result,
243     DataSet<EPGMEdge> inputEdges) {
244     return extractEdgeIds(result)
245       .join(inputEdges)
246       .where(0).equalTo(new Id<>())
247       .with(new RightSide<>());
248   }
