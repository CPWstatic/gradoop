127   @Override
128   protected GraphCollection executeForVertex(LogicalGraph graph) {
129     GradoopFlinkConfig config = graph.getConfig();
130     GraphHeadFactory<EPGMGraphHead> graphHeadFactory = graph.getFactory().getGraphHeadFactory();
131     VertexFactory<EPGMVertex> vertexFactory = graph.getFactory().getVertexFactory();
132     String variable = getQueryHandler().getVertices().iterator().next().getVariable();
133 
134     DataSet<EPGMVertex> matchingVertices = graph.getVertices()
135       .filter(new MatchingVertices<>(getQuery()));
136 
137     if (!doAttachData()) {
138       matchingVertices = matchingVertices
139         .map(new Id<>())
140         .map(new ObjectTo1<>())
141         .map(new VertexFromId(vertexFactory));
142     }
143 
144     DataSet<Tuple2<EPGMVertex, EPGMGraphHead>> pairs = matchingVertices
145       .map(new AddGraphElementToNewGraph<>(graphHeadFactory, variable))
146       .returns(new TupleTypeInfo<>(
147         TypeExtractor.getForClass(vertexFactory.getType()),
148         TypeExtractor.getForClass(graphHeadFactory.getType())));
149 
150     return config.getGraphCollectionFactory().fromDataSets(
151       pairs.map(new Value1Of2<>()),
152       pairs.map(new Value0Of2<>()));
153   }

155   @Override
156   protected GraphCollection executeForPattern(LogicalGraph graph) {
157 
158     TraversalCode traversalCode = traverser.traverse();
159 
160     DataSet<Tuple1<Embedding<GradoopId>>> embeddings;
161 
162     if (traverserStrategy == TraverserStrategy.SET_PAIR_BULK_ITERATION ||
163       traverserStrategy == TraverserStrategy.SET_PAIR_FOR_LOOP_ITERATION) {
164 
165       //--------------------------------------------------------------------------
166       // Pre-processing (filter candidates)
167       //--------------------------------------------------------------------------
168 
169       DataSet<IdWithCandidates<GradoopId>> vertices = PreProcessor.filterVertices(
170         graph, getQuery());
171 
172       DataSet<TripleWithCandidates<GradoopId>> edges = PreProcessor.filterEdges(
173         graph, getQuery());
174 
175       //--------------------------------------------------------------------------
176       // Exploration via Traversal
177       //--------------------------------------------------------------------------
178 
179       SetPairTraverser<GradoopId> distributedTraverser;
180 
181       if (traverserStrategy == TraverserStrategy.SET_PAIR_BULK_ITERATION) {
182         distributedTraverser = new SetPairBulkTraverser<>(traversalCode, matchStrategy,
183           traverser.getQueryHandler().getVertexCount(), traverser.getQueryHandler().getEdgeCount(),
184           GradoopId.class, edgeStepJoinStrategy, vertexStepJoinStrategy, getVertexMapping(),
185           getEdgeMapping());
186       } else {
187         distributedTraverser = new SetPairForLoopTraverser<>(traversalCode, matchStrategy,
188           traverser.getQueryHandler().getVertexCount(), traverser.getQueryHandler().getEdgeCount(),
189           GradoopId.class, edgeStepJoinStrategy, vertexStepJoinStrategy, getVertexMapping(),
190           getEdgeMapping());
191       }
192 
193       embeddings = distributedTraverser.traverse(vertices, edges);
194     } else if (traverserStrategy == TraverserStrategy.TRIPLES_FOR_LOOP_ITERATION) {
195       DataSet<TripleWithCandidates<GradoopId>> triples = PreProcessor
196         .filterTriplets(graph, getQuery());
197 
198       TripleTraverser<GradoopId> distributedTraverser = new TripleForLoopTraverser<>(
199         traversalCode, matchStrategy,
200         traverser.getQueryHandler().getVertexCount(),
201         traverser.getQueryHandler().getEdgeCount(),
202         GradoopId.class, edgeStepJoinStrategy, getVertexMapping(), getEdgeMapping());
203 
204       embeddings = distributedTraverser.traverse(triples);
205 
206     } else {
207       throw new IllegalArgumentException("Unsupported traverser strategy: " + traverserStrategy);
208     }
209 
210     //--------------------------------------------------------------------------
211     // Post-Processing (build Graph Collection from embeddings)
212     //--------------------------------------------------------------------------
213 
214     DataSet<EPGMElement> elements = embeddings
215       .flatMap(new ElementsFromEmbedding(traversalCode,
216         graph.getFactory().getGraphHeadFactory(),
217         graph.getFactory().getVertexFactory(),
218         graph.getFactory().getEdgeFactory(),
219         getQueryHandler()
220       ));
221 
222     return doAttachData() ?
223       PostProcessor.extractGraphCollectionWithData(elements, graph, true) :
224       PostProcessor.extractGraphCollection(elements, graph.getCollectionFactory(), true);
225   }
