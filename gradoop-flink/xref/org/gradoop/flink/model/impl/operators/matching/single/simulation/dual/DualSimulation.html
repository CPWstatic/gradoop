56  /**
57   * Vertex-centric Dual-Simulation.
58   */
59  public class DualSimulation extends PatternMatching {
60  
61    /**
62     * Logger
63     */
64    private static Logger LOG = Logger.getLogger(DualSimulation.class);
65  
66    /**
67     * If true, the algorithm uses bulk iteration for the core iteration.
68     * Otherwise it uses delta iteration.
69     */
70    private final boolean useBulkIteration;
71  
72    /**
73     * Creates a new operator instance.
74     *
75     * @param query       GDL based query
76     * @param attachData  attach original data to resulting vertices/edges
77     * @param useBulk     true to use bulk, false to use delta iteration
78     */
79    public DualSimulation(String query, boolean attachData, boolean useBulk) {
80      super(query, attachData, LOG);
81      this.useBulkIteration = useBulk;
82    }
83  
84    @Override
85    protected GraphCollection executeForVertex(
86      LogicalGraph graph)  {
87      DataSet<Tuple1<GradoopId>> matchingVertexIds = PreProcessor
88        .filterVertices(graph, getQuery())
89        .project(0);
90  
91      LogicalGraphFactory graphFactory = graph.getConfig()
92        .getLogicalGraphFactory();
93      GraphCollectionFactory collectionFactory = graph.getConfig()
94        .getGraphCollectionFactory();
95  
96      if (doAttachData()) {
97        return collectionFactory.fromGraph(
98          graphFactory.fromDataSets(matchingVertexIds
99              .join(graph.getVertices())
100             .where(0).equalTo(new Id<>())
101             .with(new RightSide<>())));
102     } else {
103       return collectionFactory.fromGraph(
104         graphFactory.fromDataSets(matchingVertexIds
105             .map(new VertexFromId(graph.getConfig().getVertexFactory()))));
106     }
107   }

109   /**
110    * Performs dual simulation based on the given query.
111    *
112    * @param graph data graph
113    * @return match graph
114    */
115   protected GraphCollection executeForPattern(
116     LogicalGraph graph) {
117     //--------------------------------------------------------------------------
118     // Pre-processing (filter candidates + build initial working set)
119     //--------------------------------------------------------------------------
120 
121     DataSet<TripleWithCandidates<GradoopId>> triples = filterTriples(graph);
122     DataSet<FatVertex> fatVertices = buildInitialWorkingSet(triples);
123 
124     //--------------------------------------------------------------------------
125     // Dual Simulation
126     //--------------------------------------------------------------------------
127 
128     DataSet<FatVertex> result = useBulkIteration ?
129       simulateBulk(fatVertices) : simulateDelta(fatVertices);
130 
131     //--------------------------------------------------------------------------
132     // Post-processing (build maximum match graph)
133     //--------------------------------------------------------------------------
134 
135     return postProcess(graph, result);
136   }

138   /**
139    * Extracts valid triples from the input graph based on the query.
140    *
141    * @param g input graph
142    * @return triples that have a match in the query graph
143    */
144   private DataSet<TripleWithCandidates<GradoopId>> filterTriples(
145     LogicalGraph g) {
146     // filter vertex-edge-vertex triples by query predicates
147     return PreProcessor.filterTriplets(g, getQuery());
148   }

150   /**
151    * Prepares the initial working set for the bulk iteration.
152    *
153    * @param triples matching triples from the input graph
154    * @return data set containing fat vertices
155    */
156   private DataSet<FatVertex> buildInitialWorkingSet(
157     DataSet<TripleWithCandidates<GradoopId>> triples) {
158     return triples.flatMap(new CloneAndReverse())
159       .groupBy(1) // sourceId
160       .combineGroup(new BuildFatVertex(getQuery()))
161       .groupBy(0) // vertexId
162       .reduceGroup(new GroupedFatVertices());
163   }

165   /**
166    * Performs dual simulation using bulk iteration.
167    *
168    * @param vertices fat vertices
169    * @return remaining fat vertices after dual simulation
170    */
171   private DataSet<FatVertex> simulateBulk(DataSet<FatVertex> vertices) {
172 
173     vertices = log(vertices, new PrintFatVertex(false, "iteration start"),
174       getVertexMapping(), getEdgeMapping());
175 
176     // ITERATION HEAD
177     IterativeDataSet<FatVertex> workSet = vertices.iterate(Integer.MAX_VALUE);
178 
179     // ITERATION BODY
180 
181     // validate neighborhood of each vertex and create deletions
182     DataSet<Deletion> deletions = workSet
183       .filter(new UpdatedFatVertices())
184       .flatMap(new ValidateNeighborhood(getQuery()));
185 
186     deletions = log(deletions, new PrintDeletion(true, "deletion"),
187       getVertexMapping(), getEdgeMapping());
188 
189     // combine deletions to message
190     DataSet<Message> combinedMessages = deletions
191       .groupBy(0)
192       .combineGroup(new CombinedMessages());
193 
194     combinedMessages = log(combinedMessages, new PrintMessage(true, "combined"),
195       getVertexMapping(), getEdgeMapping());
196 
197     // group messages to final message
198     DataSet<Message> messages = combinedMessages
199       .groupBy(0)
200       .reduceGroup(new GroupedMessages());
201 
202     messages = log(messages, new PrintMessage(true, "grouped"),
203       getVertexMapping(), getEdgeMapping());
204 
205     // update candidates and build next working set
206     DataSet<FatVertex> nextWorkingSet = workSet
207       .leftOuterJoin(messages)
208       .where(0).equalTo(0) // vertexId == recipientId
209       .with(new UpdateVertexState(getQuery()))
210       .filter(new ValidFatVertices());
211 
212     nextWorkingSet = log(nextWorkingSet,
213       new PrintFatVertex(true, "next workset"),
214       getVertexMapping(), getEdgeMapping());
215 
216     // ITERATION FOOTER
217     return workSet.closeWith(nextWorkingSet, deletions);
218   }

220   /**
221    * Performs dual simulation using delta iteration.
222    *
223    * @param vertices fat vertices
224    * @return remaining fat vertices after dual simulation
225    */
226   private DataSet<FatVertex> simulateDelta(DataSet<FatVertex> vertices) {
227     // prepare initial working set
228     DataSet<Message> initialWorkingSet = vertices
229       .flatMap(new ValidateNeighborhood(getQuery()))
230       .groupBy(0)
231       .combineGroup(new CombinedMessages())
232       .groupBy(0)
233       .reduceGroup(new GroupedMessages());
234 
235     vertices = log(vertices, new PrintFatVertex(false, "initial solution set"),
236       getVertexMapping(), getEdgeMapping());
237 
238     initialWorkingSet = log(initialWorkingSet,
239       new PrintMessage(false, "initial working set"),
240       getVertexMapping(), getEdgeMapping());
241 
242     // ITERATION HEAD
243     DeltaIteration<FatVertex, Message> iteration = vertices
244       .iterateDelta(initialWorkingSet, Integer.MAX_VALUE, 0);
245 
246     // ITERATION BODY
247 
248     // get updated vertices
249     DataSet<FatVertex> deltas = iteration.getSolutionSet()
250       .join(iteration.getWorkset())
251       .where(0).equalTo(0)
252       .with(new UpdateVertexState(getQuery()));
253 
254     deltas = log(deltas, new PrintFatVertex(true, "solution set delta"),
255       getVertexMapping(), getEdgeMapping());
256 
257     // prepare new messages for the next round from updates
258     DataSet<Message> updates = deltas
259       .filter(new ValidFatVertices())
260       .flatMap(new ValidateNeighborhood(getQuery()))
261       .groupBy(0)
262       .combineGroup(new CombinedMessages())
263       .groupBy(0)
264       .reduceGroup(new GroupedMessages());
265 
266     updates = log(updates, new PrintMessage(true, "next working set"),
267       getVertexMapping(), getEdgeMapping());
268 
269     // ITERATION FOOTER
270     // filter vertices with no candidates after iteration
271     return iteration.closeWith(deltas, updates).filter(new ValidFatVertices());
272   }

274   /**
275    * Extracts vertices and edges from the query result and constructs a
276    * maximum match graph.
277    *
278    * @param graph    input graph
279    * @param vertices valid vertices after simulation
280    * @return maximum match graph
281    */
282   private GraphCollection postProcess(LogicalGraph graph,
283     DataSet<FatVertex> vertices) {
284     GradoopFlinkConfig config = graph.getConfig();
285 
286     DataSet<Vertex> matchVertices = doAttachData() ?
287       PostProcessor.extractVerticesWithData(vertices, graph.getVertices()) :
288       PostProcessor.extractVertices(vertices, config.getVertexFactory());
289 
290     DataSet<Edge> matchEdges = doAttachData() ?
291       PostProcessor.extractEdgesWithData(vertices, graph.getEdges()) :
292       PostProcessor.extractEdges(vertices, config.getEdgeFactory());
293 
294     return config.getGraphCollectionFactory().fromGraph(
295       config.getLogicalGraphFactory().fromDataSets(matchVertices, matchEdges));
296   }
