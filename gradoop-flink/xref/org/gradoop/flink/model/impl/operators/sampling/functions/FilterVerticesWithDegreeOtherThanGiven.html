16  package org.gradoop.flink.model.impl.operators.sampling.functions;
17  
18  import org.apache.flink.api.java.DataSet;
19  import org.gradoop.common.model.impl.pojo.EPGMVertex;
20  import org.gradoop.flink.algorithms.gelly.vertexdegrees.DistinctVertexDegrees;
21  import org.gradoop.flink.model.api.operators.UnaryGraphToGraphOperator;
22  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
23  import org.gradoop.flink.model.impl.functions.epgm.PropertyRemover;
24  import org.gradoop.flink.model.impl.operators.sampling.common.SamplingConstants;
25  
26  /**
27   * Retains all vertices which do not have the given degree.
28   */
29  public class FilterVerticesWithDegreeOtherThanGiven implements UnaryGraphToGraphOperator {
30  
31    /**
32     * the given degree
33     */
34    private long degree;
35  
36    /**
37     * Constructor
38     *
39     * @param degree the given degree
40     */
41    public FilterVerticesWithDegreeOtherThanGiven(long degree) {
42      this.degree = degree;
43    }
44  
45    @Override
46    public LogicalGraph execute(LogicalGraph graph) {
47  
48      DistinctVertexDegrees distinctVertexDegrees = new DistinctVertexDegrees(
49        SamplingConstants.DEGREE_PROPERTY_KEY,
50        SamplingConstants.IN_DEGREE_PROPERTY_KEY,
51        SamplingConstants.OUT_DEGREE_PROPERTY_KEY,
52        true);
53  
54      DataSet<EPGMVertex> newVertices = distinctVertexDegrees.execute(graph).getVertices()
55        .filter(new VertexWithDegreeFilter<>(degree, SamplingConstants.DEGREE_PROPERTY_KEY))
56        .map(new PropertyRemover<>(SamplingConstants.DEGREE_PROPERTY_KEY))
57        .map(new PropertyRemover<>(SamplingConstants.IN_DEGREE_PROPERTY_KEY))
58        .map(new PropertyRemover<>(SamplingConstants.OUT_DEGREE_PROPERTY_KEY));
59  
60      return graph.getFactory().fromDataSets(
61        graph.getGraphHead(), newVertices, graph.getEdges());
62    }
63  }
