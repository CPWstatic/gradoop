39  /**
40   * This is an implementation of a very straight-forward depth-first pattern
41   * matching algorithm.
42   *
43   * The depth-first search is implemented by popping an embedding from
44   * a stack, growing it by one vertex and its corresponding edges, and adding
45   * the resulting new edges to the top of the stack. If it is not possible to
46   * apply a step to an existing embedding, the embedding is removed.
47   * This is repeated until all embeddings are either completed
48   * (all steps have been applied).
49   */
50  public class DepthSearchMatching implements PatternMatchingAlgorithm {

60     * source-id -> edge-ids

63    /**
64     * target-id -> edge-ids
65     */
66    private Map<GradoopId, Set<GradoopId>> targetDict;
67    /**
68     * edge id -> source-id, target-id, edge candidates

86    @Override
87    public List<Embedding<GradoopId>> findEmbeddings(GraphWithCandidates graph,
88      String query) {
89      // construct a handler for the query
90      handler = new QueryHandler(query);
91  
92      // build the plan that determines in which order vertices should added
93      int[] plan = buildQueryPlan();
94  
95      // by using a stack (first in - last out) we implement a depth-first search
96      Stack<Embedding<GradoopId>> embeddings = new Stack<>();
97  
98      // construct an empty embedding
99      Embedding<GradoopId> firstEmbedding = new Embedding<>();
100     firstEmbedding.setVertexMapping(new GradoopId[handler.getVertexCount()]);
101     firstEmbedding.setEdgeMapping(new GradoopId[handler.getEdgeCount()]);
102     embeddings.push(firstEmbedding);
103 
104     // initialize the vertex, edge, source and target maps
105     initializeMaps(graph);
106 
107     List<Embedding<GradoopId>> results = new ArrayList<>();
108 
109     // grow the embeddings until there are none left
110     while (!embeddings.isEmpty()) {
111       Embedding<GradoopId> embedding = embeddings.pop();
112 
113       // find the first step in the plan that has not been applied yet
114       int nextStep = -1;
115       for (int step : plan) {
116         if (embedding.getVertexMapping()[step] == null) {
117           nextStep = step;
118           break;
119         }
120       }
121 
122       // if all steps have been applied, add the embedding to the results
123       if (nextStep == -1) {
124         results.add(embedding);
125         continue;
126       }
127 
128       // compute new embeddings by applying the step on the current embedding
129       List<Embedding<GradoopId>> newEmbeddings = executeStep(embedding, nextStep);
130 
131       // add the new embeddings on top of the stack
132       embeddings.addAll(newEmbeddings);
133     }
134 
135     // reset the maps for the next graph
136     resetMaps();
137 
138     return results;
139   }

142   public Boolean hasEmbedding(GraphWithCandidates graph, String query) {
143 
144     // construct a handler for the query
145     handler = new QueryHandler(query);
146 
147     // build the plan that determines in which order vertices should added
148     int[] plan = buildQueryPlan();
149 
150     // by using a stack (first in - last out) we implement a depth-first search
151     Stack<Embedding<GradoopId>> embeddings = new Stack<>();
152 
153     // construct an empty embedding
154     Embedding<GradoopId> firstEmbedding = new Embedding<>();
155     firstEmbedding.setVertexMapping(new GradoopId[handler.getVertexCount()]);
156     firstEmbedding.setEdgeMapping(new GradoopId[handler.getEdgeCount()]);
157     embeddings.push(firstEmbedding);
158 
159     // initialize the vertex, edge, source and target maps
160     initializeMaps(graph);
161 
162     // grow the embeddings until there are none left
163     while (!embeddings.isEmpty()) {
164       Embedding<GradoopId> embedding = embeddings.pop();
165 
166       // find the first step in the plan that has not been applied yet
167       int nextStep = -1;
168       for (int step : plan) {
169         if (embedding.getVertexMapping()[step] == null) {
170           nextStep = step;
171           break;
172         }
173       }
174 
175       // if there is a complete embedding, return true
176       if (nextStep == -1) {
177         resetMaps();
178         return true;
179       }
180 
181       // compute new embeddings by applying the step on the current embedding
182       List<Embedding<GradoopId>> newEmbeddings = executeStep(embedding, nextStep);
183 
184       // add the new embeddings on top of the stack
185       embeddings.addAll(newEmbeddings);
186     }
187 
188     // reset the maps for the next graph
189     resetMaps();
190 
191     // if no complete embedding could be constructed, return false
192     return false;
193   }

205   /**
206    * Execute a step. A step corresponds to the position of the next vertex in
207    * the vertex mappings of the embedding that shall be matched.
208    *
209    * @param embedding the embedding that has been constructed so far
210    * @param step      number of the next vertex to be matched
211    * @return list of newly constructed embeddings
212    */
213   private List<Embedding<GradoopId>> executeStep(Embedding<GradoopId> embedding, int step) {
214 
215     // map containing the found matches for all pattern edges
216     // this is necessary to be able to construct all valid permutations of edges
217     Map<Long, Set<GradoopId>> edgeMatches = new HashMap<>();
218 
219     List<Embedding<GradoopId>> results = new ArrayList<>();
220 
221     // get the possible vertex matches
222     for (GradoopId id : getCandidates(step)) {
223 
224       // flag to recognize failed vertex matchings
225       boolean failed = false;
226 
227       // if the vertex is already in the embedding, skip it
228       if (Arrays.asList(embedding.getVertexMapping()).contains(id)) {
229         continue;
230       }
231 
232       // get all outgoing edges of the next step vertex
233       Collection<Edge> edges = handler.getEdgesBySourceVertexId((long) step);
234       List<Edge> patternEdges =
235         edges != null ? new ArrayList<>(edges) : new ArrayList<>();
236 
237       // only keep those edges that have at least one vertex already in the
238       // embedding, or are direct loops
239       filterPatternEdges(patternEdges, embedding);
240 
241       // find the matches for each pattern edge
242       for (Edge patternEdge : patternEdges) {
243         edgeMatches.put(patternEdge.getId(), new HashSet<>());
244 
245         // find all edge candidates
246         Set<GradoopId> edgeCandidateIds =
247           sourceDict.get(id) != null ? sourceDict.get(id) : new HashSet<>();
248 
249         // for each candidate, check if it matches the pattern edge
250         for (GradoopId edgeCandidateId : edgeCandidateIds) {
251 
252           Tuple3<GradoopId, GradoopId, boolean[]> edgeCandidate =
253             edgeDict.get(edgeCandidateId);
254 
255           // get the target vertex of the edge
256           GradoopId target = embedding.getVertexMapping()[
257             Math.toIntExact(patternEdge.getTargetVertexId())];
258 
259           // if the pattern edge and the edge candidate are both loops
260           // and the candidate matches, add it to the embedding
261           if (isLoop(patternEdge, edgeCandidate)) {
262             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
263 
264             // else, if the candidate matches the pattern edge,
265             // add it to the embedding
266           } else if (matchOutgoingEdge(patternEdge, edgeCandidateId, target)) {
267             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
268           }
269         }
270 
271         // if there was no matching edge for a pattern edge, the candidate can be discarded
272         if (edgeMatches.get(patternEdge.getId()).isEmpty()) {
273           failed = true;
274           break;
275         }
276       }
277 
278       if (failed) {
279         continue;
280       }
281 
282       // get all incoming edges of the next step vertex
283       edges = handler.getEdgesByTargetVertexId((long) step);
284       patternEdges = edges != null ? new ArrayList<>(edges) : new ArrayList<>();
285 
286       // only keep those edges that have at least one vertex already in the
287       // embedding, or are direct loops
288       filterPatternEdges(patternEdges, embedding);
289 
290       // find the matches for each pattern edge
291       for (Edge patternEdge : patternEdges) {
292         edgeMatches.put(patternEdge.getId(), new HashSet<>());
293 
294         // find all edge candidates
295         Set<GradoopId> edgeCandidateIds =
296           targetDict.get(id) != null ? targetDict.get(id) : new HashSet<>();
297 
298         // for each candidate, check if it matches the pattern edge
299         for (GradoopId edgeCandidateId : edgeCandidateIds) {
300 
301           Tuple3<GradoopId, GradoopId, boolean[]> edgeCandidate =
302             edgeDict.get(edgeCandidateId);
303 
304           // get the target vertex of the edge
305           GradoopId source = embedding.getVertexMapping()[
306             Math.toIntExact(patternEdge.getSourceVertexId())];
307 
308           // if the pattern edge and the edge candidate are both loops
309           // and the candidate matches, add it to the embedding
310           if (isLoop(patternEdge, edgeCandidate)) {
311             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
312 
313             // else, if the candidate matches the pattern edge,
314             // add it to the embedding
315           } else if (matchIncomingEdge(patternEdge, edgeCandidateId, source)) {
316             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
317           }
318         }
319 
320         // if there was no matching edge for a pattern edge, the candidate
321         // can be discarded
322         if (edgeMatches.get(patternEdge.getId()).isEmpty()) {
323           failed = true;
324           break;
325         }
326       }
327 
328       if (failed) {
329         continue;
330       }
331       // add all grown embeddings to the results
332       results.addAll(buildNewEmbeddings(embedding, step, id, edgeMatches));
333     }
334     return results;
335   }

337   /**
338    * If both the pattern edge and the edge candidate are loops and the candidate
339    * matches the pattern, return true.
340    *
341    * @param patternEdge pattern edge
342    * @param edgeCandidate edge candidate
343    * @return true iff both are loops and the candidate matches the pattern edge
344    */
345   private boolean isLoop(Edge patternEdge,
346     Tuple3<GradoopId, GradoopId, boolean[]> edgeCandidate) {
347     if (patternEdge.getSourceVertexId()
348       .equals(patternEdge.getTargetVertexId())) {
349       if (edgeCandidate.f0.equals(edgeCandidate.f1)) {
350         return edgeCandidate.f2[(int) patternEdge.getId()];
351       }
352     }
353     return false;
354   }

356   /**
357    * Remove all edges that are neither loops nor have source or target vertex
358    * that is in the embedding.
359    *
360    * @param patternEdges edges to be filtered
361    * @param embedding embedding with possible source and target vertices
362    * @return filtered edge list
363    */
364   private List<Edge> filterPatternEdges(
365     List<Edge> patternEdges,
366     Embedding<GradoopId> embedding) {
367     for (int i = 0; i < patternEdges.size(); i++) {
368       long sourceId = patternEdges.get(i).getSourceVertexId();
369       long targetId = patternEdges.get(i).getTargetVertexId();
370       GradoopId[] vertexMapping = embedding.getVertexMapping();
371       if ((vertexMapping[(int) sourceId] == null) &&
372         (vertexMapping[(int) targetId] == null)) {
373         if (sourceId != targetId) {
374           patternEdges.remove(i);
375           i--;
376         }
377       }
378     }
379     return patternEdges;
380   }

382   /**
383    * Extend the given embedding by found matches
384    *
385    * @param startEmbedding embedding that is to be extended
386    * @param step next step
387    * @param vertexMatch vertex chosen for next step
388    * @param edgeMatches found match
389    * @return extended embedding
390    */
391   private List<Embedding<GradoopId>> buildNewEmbeddings(
392     Embedding<GradoopId> startEmbedding,
393     int step,
394     GradoopId vertexMatch,
395     Map<Long, Set<GradoopId>> edgeMatches) {
396 
397     // begin with the start embedding
398     List<Embedding<GradoopId>> result = new ArrayList<>();
399     result.add(startEmbedding);
400 
401     // list of embeddings in next growing step
402     List<Embedding<GradoopId>> temporaryEmbeddings;
403 
404     // get edge matches
405     List<Map.Entry<Long, Set<GradoopId>>> entries =
406       new ArrayList<>(edgeMatches.entrySet());
407 
408     // if no edge matches have been found, the result is an embedding with only
409     // one vertex
410     if (entries.isEmpty()) {
411       Embedding<GradoopId> newEmbedding = copyEmbedding(startEmbedding);
412       newEmbedding.getVertexMapping()[step] = vertexMatch;
413       result.clear();
414       result.add(newEmbedding);
415     }
416 
417     // for each edge with matches
418     for (Map.Entry<Long, Set<GradoopId>> entry : entries) {
419       temporaryEmbeddings = new ArrayList<>();
420       // for each embedding in the result so far
421       for (Embedding<GradoopId> embedding : result) {
422         // for each edge match
423         for (GradoopId id : entry.getValue()) {
424           // check if the edge match already occurs in the embedding
425           boolean contains = false;
426           for (GradoopId edgeId : embedding.getEdgeMapping()) {
427             if (edgeId == null) {
428               continue;
429             }
430             if (edgeId.equals(id)) {
431               contains = true;
432               break;
433             }
434           }
435           // if yes, skip it
436           if (contains) {
437             continue;
438           }
439           // else, copy the embedding, add the edge match and add it to the
440           // results in the next iteration
441           Embedding<GradoopId> newEmbedding = copyEmbedding(embedding);
442           newEmbedding.getVertexMapping()[step] = vertexMatch;
443           newEmbedding.getEdgeMapping()[Math.toIntExact(entry.getKey())] = id;
444           temporaryEmbeddings.add(newEmbedding);
445         }
446       }
447       result = new ArrayList<>(temporaryEmbeddings);
448     }
449     return result;
450   }

452   /**
453    * Creates a copy of an existing embedding.
454    *
455    * @param existing existing embedding
456    * @return copy of the existing embedding
457    */
458   private Embedding<GradoopId> copyEmbedding(Embedding<GradoopId> existing) {
459     Embedding<GradoopId> newEmbedding = new Embedding<>();
460 
461     GradoopId[] vertexCopy = new GradoopId[existing.getVertexMapping().length];
462     System.arraycopy(existing.getVertexMapping(), 0, vertexCopy, 0,
463       existing.getVertexMapping().length);
464 
465     GradoopId[] edgeCopy = new GradoopId[existing.getEdgeMapping().length];
466     System.arraycopy(existing.getEdgeMapping(), 0, edgeCopy, 0,
467       existing.getEdgeMapping().length);
468 
469     newEmbedding.setVertexMapping(vertexCopy);
470     newEmbedding.setEdgeMapping(edgeCopy);
471     return newEmbedding;
472   }

474   /**
475    * Get all vertices that match the pattern of a step.
476    *
477    * @param step next step in the plan
478    * @return list containing the ids of the matching vertices
479    */
480   public List<GradoopId> getCandidates(int step) {
481     List<GradoopId> possibilities = new ArrayList<>(vertexDict.keySet());
482     for (int i = 0; i < possibilities.size(); i++) {
483       GradoopId vertexId = possibilities.get(i);
484       boolean[] candidates = vertexDict.get(vertexId);
485       if (!candidates[step]) {
486         possibilities.remove(i);
487         --i;
488       }
489     }
490     return possibilities;
491   }

493   /**
494    * Returns true, if the edge candidate matches the outgoing pattern edge.
495    *
496    * @param patternEdge outgoing pattern edge
497    * @param edgeCandidate id of the edge candidate
498    * @param target target id of the edge candidate
499    * @return true, iff edge matches pattern edge
500    */
501   private boolean matchOutgoingEdge(Edge patternEdge, GradoopId edgeCandidate,
502     GradoopId target) {
503     Tuple3<GradoopId, GradoopId, boolean[]> edge = edgeDict.get(edgeCandidate);
504     if (edge.f2[(int) patternEdge.getId()]) {
505       GradoopId possibleVertex = edge.f1;
506       return possibleVertex.equals(target);
507     }
508     return false;
509   }

515    * @param edgeCandidate id of the edge candidate
516    * @param source source id of the edge candidate
517    * @return true, iff edge matches pattern edge

529   /**
530    * Build HashMaps for faster access.
531    *
532    * @param graph graphs with candidate sets
533    */
534   private void initializeMaps(GraphWithCandidates graph) {
535     for (IdWithCandidates<GradoopId> vertex : graph.getVertexCandidates()) {
536       vertexDict.put(vertex.getId(), vertex.getCandidates());
537     }
538 
539     for (TripleWithCandidates<GradoopId> edge : graph.getEdgeCandidates()) {
540       edgeDict.put(edge.getEdgeId(),
541         new Tuple3<>(edge.getSourceId(), edge.getTargetId(),
542           edge.getCandidates()));
543     }
544 
545     for (TripleWithCandidates<GradoopId> edge : graph.getEdgeCandidates()) {
546       if (!sourceDict.containsKey(edge.getSourceId())) {
547         sourceDict.put(edge.getSourceId(), new HashSet<>());
548       }
549       sourceDict.get(edge.getSourceId()).add(edge.getEdgeId());
550       if (!targetDict.containsKey(edge.getTargetId())) {
551         targetDict.put(edge.getTargetId(), new HashSet<>());
552       }
553       targetDict.get(edge.getTargetId()).add(edge.getEdgeId());
554     }
555   }

557   /**
558    * Method to create query plan, determining how patterns are grown.
559    *
560    * @return plan for walk through
561    */
562   private int[] buildQueryPlan() {
563     int[] queryPlan = new int[handler.getVertices().size()];
564     int step = 0;
565     Set<Long> alreadyVisited = new HashSet<>();
566     Stack<Vertex> stack = new Stack<>();
567     stack.push(handler.getVertexById(0L));
568     alreadyVisited.add(0L);
569     while (!stack.isEmpty()) {
570       Vertex current = stack.pop();
571       Collection<Vertex> neighbors = handler.getNeighbors(current.getId());
572       queryPlan[step] = (int) current.getId();
573       step++;
574       neighbors.stream()
575         .filter(neighbor -> !alreadyVisited.contains(neighbor.getId()))
576         .forEach(neighbor -> {
577             alreadyVisited.add(neighbor.getId());
578             stack.push(neighbor);
579           }
580         );
581     }
582     return queryPlan;
583   }
