48  /**
49   * Takes a collection of logical graphs and a user defined aggregate function as
50   * input. The aggregate function is applied on each logical graph contained in
51   * the collection and the aggregate is stored as an additional property at the
52   * graphs.
53   */
54  public class ApplySubgraph implements ApplicableUnaryGraphToGraphOperator {
55    /**
56     * Used to filter vertices from the logical graph.
57     */
58    private final FilterFunction<Vertex> vertexFilterFunction;
59    /**
60     * Used to filter edges from the logical graph.
61     */
62    private final FilterFunction<Edge> edgeFilterFunction;
63  
64    /**
65     * Creates a new sub graph operator instance.
66     * <p/>
67     * If both parameters are not {@code null}, the operator returns the subgraph
68     * defined by filtered vertices and edges.
69     * <p/>
70     * If the {@code edgeFilterFunction} is {@code null}, the operator returns the
71     * vertex-induced subgraph.
72     * <p/>
73     * If the {@code vertexFilterFunction} is {@code null}, the operator returns
74     * the edge-induced subgraph.
75     *
76     * @param vertexFilterFunction vertex filter function
77     * @param edgeFilterFunction   edge filter function
78     */
79    public ApplySubgraph(FilterFunction<Vertex> vertexFilterFunction,
80      FilterFunction<Edge> edgeFilterFunction) {
81      if (vertexFilterFunction == null && edgeFilterFunction == null) {
82        throw new IllegalArgumentException("No filter functions was given.");
83      }
84      this.vertexFilterFunction = vertexFilterFunction;
85      this.edgeFilterFunction = edgeFilterFunction;
86    }
87  
88    @Override
89    public GraphCollection executeForGVELayout(GraphCollection collection) {
90      return vertexFilterFunction != null && edgeFilterFunction != null ?
91        subgraph(collection) :
92        vertexFilterFunction != null ? vertexInducedSubgraph(collection) :
93          edgeInducedSubgraph(collection);
94    }
95  
96    @Override
97    public GraphCollection executeForTxLayout(GraphCollection collection) {
98      return executeForGVELayout(collection);
99    }

101   /**
102    * Returns one subgraph for each of the given super graphs.
103    * The subgraphs are defined by the vertices that fulfil the vertex filter
104    * function.
105    *
106    * @param collection collection of supergraphs
107    * @return collection of vertex-induced subgraphs
108    */
109   private GraphCollection vertexInducedSubgraph(
110     GraphCollection collection) {
111     //--------------------------------------------------------------------------
112     // compute a dictionary that maps the old graph ids to new ones
113     //--------------------------------------------------------------------------
114 
115     DataSet<Tuple2<GradoopId, GradoopId>> graphIdDictionary = collection
116       .getGraphHeads()
117       .map(new Id<>())
118       .map(new PairElementWithNewId<>());
119 
120     //--------------------------------------------------------------------------
121     // compute new graphs
122     //--------------------------------------------------------------------------
123 
124     EPGMGraphHeadFactory<GraphHead> graphFactory = collection.getFactory().getGraphHeadFactory();
125 
126     DataSet<GraphHead> newGraphHeads = graphIdDictionary
127       .map(new Project2To1<>())
128       .map(new InitGraphHead(graphFactory));
129 
130     //--------------------------------------------------------------------------
131     // compute pairs of a vertex id and the set of new graphs this vertex is
132     // contained in
133     // filter function is applied first to improve performance
134     //--------------------------------------------------------------------------
135 
136     DataSet<Tuple2<GradoopId, GradoopIdSet>> vertexIdsWithNewGraphs =
137       collection.getVertices()
138         .filter(vertexFilterFunction)
139         .flatMap(new ElementIdGraphIdTuple<>())
140         .join(graphIdDictionary)
141         .where(1)
142         .equalTo(0)
143         .with(new JoinTuplesWithNewGraphs())
144         .groupBy(0)
145         .reduceGroup(new MergeTupleGraphs());
146 
147     //--------------------------------------------------------------------------
148     // compute new vertices
149     //--------------------------------------------------------------------------
150 
151     DataSet<Vertex> newVertices = vertexIdsWithNewGraphs
152         .join(collection.getVertices())
153         .where(0)
154         .equalTo(new Id<>())
155         .with(new AddGraphsToElements<>());
156 
157     //--------------------------------------------------------------------------
158     // build tuples4 for each edge, containing
159     // edge id, source id, target id, set of new graph ids
160     //--------------------------------------------------------------------------
161 
162     DataSet<Tuple4<GradoopId, GradoopId, GradoopId, GradoopIdSet>> edgeTuple =
163       collection.getEdges()
164         .flatMap(new IdSourceTargetGraphTuple<>())
165         .join(graphIdDictionary)
166         .where(3).equalTo(0)
167         .with(new EdgesWithNewGraphsTuple())
168         .groupBy(new Value0Of4<>())
169         .reduceGroup(new MergeEdgeGraphs());
170 
171     //--------------------------------------------------------------------------
172     // join the tuple4 with vertex tuples, keeping the graph sets
173     // apply a "filter" by using a flat map function to check for each edge
174     // if a graph exists that this edge, its source and its target are contained
175     // in
176     // the result tuple consists of
177     // edge id, new edge graphs
178     //--------------------------------------------------------------------------
179 
180     DataSet<Tuple2<GradoopId, GradoopIdSet>> edgeIdsWithNewGraphs =
181       edgeTuple
182         .join(vertexIdsWithNewGraphs)
183         .where(1).equalTo(0)
184         .with(new JoinWithSourceGraphIdSet())
185         .join(vertexIdsWithNewGraphs)
186         .where(2).equalTo(0)
187         .with(new JoinWithTargetGraphIdSet())
188         .flatMap(new FilterEdgeGraphs());
189 
190     //--------------------------------------------------------------------------
191     // compute the new edges
192     //--------------------------------------------------------------------------
193 
194     DataSet<Edge> newEdges = edgeIdsWithNewGraphs
195       .join(collection.getEdges())
196       .where(0)
197       .equalTo(new Id<>())
198       .with(new AddGraphsToElements<>());
199 
200     return collection.getConfig().getGraphCollectionFactory()
201       .fromDataSets(newGraphHeads, newVertices, newEdges);
202   }

204   /**
205    * Returns one subgraph for each of the given supergraphs.
206    * The subgraphs are defined by the edges that fulfil the vertex filter
207    * function.
208    *
209    * @param collection collection of supergraphs
210    * @return collection of edge-induced subgraphs
211    */
212   private GraphCollection edgeInducedSubgraph(
213     GraphCollection collection) {
214     //--------------------------------------------------------------------------
215     // compute a dictionary that maps the old graph ids to new ones
216     //--------------------------------------------------------------------------
217 
218     DataSet<Tuple2<GradoopId, GradoopId>> graphIdDictionary = collection
219       .getGraphHeads()
220       .map(new Id<>())
221       .map(new PairElementWithNewId<>());
222 
223     //--------------------------------------------------------------------------
224     // compute new graphs
225     //--------------------------------------------------------------------------
226 
227     EPGMGraphHeadFactory<GraphHead> graphFactory = collection.getFactory().getGraphHeadFactory();
228 
229     DataSet<GraphHead> newGraphHeads = graphIdDictionary
230       .map(new Project2To1<>())
231       .map(new InitGraphHead(graphFactory));
232 
233     //--------------------------------------------------------------------------
234     // compute new edges by applying the filter function,
235     // building tuples containing the edge id and the new graphs this edge is
236     // contained in and finally adding the new graphs to the edges
237     //--------------------------------------------------------------------------
238 
239     DataSet<Edge> newEdges = collection
240       .getEdges()
241       .filter(edgeFilterFunction)
242       .flatMap(new ElementIdGraphIdTuple<>())
243       .join(graphIdDictionary)
244       .where(1)
245       .equalTo(0)
246       .with(new JoinTuplesWithNewGraphs())
247       .groupBy(0)
248       .reduceGroup(new MergeTupleGraphs())
249       .join(collection.getEdges())
250       .where(0)
251       .equalTo(new Id<>())
252       .with(new AddGraphsToElements<>());
253 
254     //--------------------------------------------------------------------------
255     // compute the new vertices
256     // first, tuples 2 containing the sources and targets of all new edges in
257     // the first field and the new graphs this edge is contained in is created
258     // this is then joined with the input vertices and the new graphs
259     // are added to the vertices
260     //--------------------------------------------------------------------------
261 
262     DataSet<Vertex> newVertices = newEdges
263       .flatMap(new SourceTargetIdGraphsTuple<>())
264       .distinct(0)
265       .coGroup(collection.getVertices())
266       .where(0)
267       .equalTo(new Id<>())
268       .with(new AddGraphsToElementsCoGroup<>());
269 
270     return collection.getConfig().getGraphCollectionFactory()
271       .fromDataSets(newGraphHeads, newVertices, newEdges);
272   }

274   /**
275    * Returns one subgraph for each of the given supergraphs.
276    * The subgraphs are defined by the vertices that fulfil the vertex filter
277    * function and edges that fulfill the edge filter function.
278    *
279    * Note, that the operator does not verify the consistency of the resulting
280    * graph.
281    *
282    * @param collection collection of supergraphs
283    * @return collection of subgraphs
284    */
285   private GraphCollection subgraph(
286     GraphCollection collection) {
287 
288     //--------------------------------------------------------------------------
289     // compute a dictionary that maps the old graph ids to new ones
290     //--------------------------------------------------------------------------
291 
292     DataSet<Tuple2<GradoopId, GradoopId>> graphIdDictionary = collection
293       .getGraphHeads()
294       .map(new Id<>())
295       .map(new PairElementWithNewId<>());
296 
297     //--------------------------------------------------------------------------
298     // compute new graphs
299     //--------------------------------------------------------------------------
300 
301     EPGMGraphHeadFactory<GraphHead> graphFactory = collection.getFactory().getGraphHeadFactory();
302 
303     DataSet<GraphHead> newGraphHeads = graphIdDictionary
304       .map(new Project2To1<>())
305       .map(new InitGraphHead(graphFactory));
306 
307     //--------------------------------------------------------------------------
308     // compute pairs of a vertex and the set of new graphs this vertex is
309     // contained in
310     // filter function is applied first to improve performance
311     //--------------------------------------------------------------------------
312 
313     DataSet<Vertex> newVertices =
314       collection.getVertices()
315         .filter(vertexFilterFunction)
316         .flatMap(new ElementIdGraphIdTuple<>())
317         .join(graphIdDictionary)
318         .where(1).equalTo(0)
319         .with(new JoinTuplesWithNewGraphs())
320         .groupBy(0)
321         .reduceGroup(new MergeTupleGraphs())
322         .join(collection.getVertices())
323         .where(0)
324         .equalTo(new Id<>())
325         .with(new AddGraphsToElements<>());
326 
327     //--------------------------------------------------------------------------
328     // compute pairs of an edge and the set of new graphs this edge is
329     // contained in
330     // filter function is applied first to improve performance
331     //--------------------------------------------------------------------------
332 
333     DataSet<Edge> newEdges = collection.getEdges()
334       .filter(edgeFilterFunction)
335       .flatMap(new ElementIdGraphIdTuple<>())
336       .join(graphIdDictionary)
337       .where(1)
338       .equalTo(0)
339       .with(new JoinTuplesWithNewGraphs())
340       .groupBy(0)
341       .reduceGroup(new MergeTupleGraphs())
342       .join(collection.getEdges())
343       .where(0)
344       .equalTo(new Id<>())
345       .with(new AddGraphsToElements<>());
346 
347     return collection.getConfig().getGraphCollectionFactory()
348       .fromDataSets(newGraphHeads, newVertices, newEdges);
349   }
