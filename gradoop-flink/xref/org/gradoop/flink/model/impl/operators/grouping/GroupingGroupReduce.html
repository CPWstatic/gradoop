35  /**
36   * Grouping implementation that uses group + groupReduce for building super
37   * vertices and updating the original vertices.
38   *
39   * Algorithmic idea:
40   *
41   * 1) Map vertices to a minimal representation, i.e. {@link VertexGroupItem}.
42   * 2) Group vertices on label and/or property.
43   * 3) Create a super vertex id for each group and collect a non-candidate
44   *    {@link VertexGroupItem} for each group element and one additional
45   *    super vertex tuple that holds the group aggregate.
46   * 4) Filter output of 3)
47   *    a) non-candidate tuples are mapped to {@link VertexWithSuperVertex}
48   *    b) super vertex tuples are used to build final super vertices
49   * 5) Map edges to a minimal representation, i.e. {@link EdgeGroupItem}
50   * 6) Join edges with output of 4a) and replace source/target id with super
51   *    vertex id.
52   * 7) Updated edges are grouped by source and target id and optionally by label
53   *    and/or edge property.
54   * 8) Group combine on the workers and compute aggregate.
55   * 9) Group reduce globally and create final super edges.
56   */
57  public class GroupingGroupReduce extends Grouping {
58    /**
59     * Creates grouping operator instance.
60     *
61     * @param useVertexLabels   group on vertex label true/false
62     * @param useEdgeLabels     group on edge label true/false
63     * @param vertexLabelGroups stores grouping properties for vertex labels
64     * @param edgeLabelGroups   stores grouping properties for edge labels
65     */
66    GroupingGroupReduce(
67      boolean useVertexLabels,
68      boolean useEdgeLabels,
69      List<LabelGroup> vertexLabelGroups,
70      List<LabelGroup> edgeLabelGroups) {
71      super(
72        useVertexLabels, useEdgeLabels, vertexLabelGroups, edgeLabelGroups);
73    }
74  
75    @Override
76    protected LogicalGraph groupInternal(LogicalGraph graph) {
77  
78      DataSet<VertexGroupItem> verticesForGrouping = graph.getVertices()
79        // map vertex to vertex group item
80        .flatMap(new BuildVertexGroupItem(useVertexLabels(), getVertexLabelGroups()));
81  
82      // group vertices by label / properties / both
83      DataSet<VertexGroupItem> vertexGroupItems = groupVertices(verticesForGrouping)
84        // apply aggregate function
85        .reduceGroup(new ReduceVertexGroupItems(useVertexLabels()));
86  
87      DataSet<Vertex> superVertices = vertexGroupItems
88        // filter group representative tuples
89        .filter(new FilterSuperVertices())
90        // build super vertices
91        .map(new BuildSuperVertex(useVertexLabels(), config.getVertexFactory()));
92  
93      DataSet<VertexWithSuperVertex> vertexToRepresentativeMap = vertexGroupItems
94        // filter group element tuples
95        .filter(new FilterRegularVertices())
96        // build vertex to group representative tuple
97        .map(new BuildVertexWithSuperVertex());
98  
99      // build super edges
100     DataSet<Edge> superEdges = buildSuperEdges(graph, vertexToRepresentativeMap);
101 
102     return config.getLogicalGraphFactory().fromDataSets(superVertices, superEdges);
103   }
