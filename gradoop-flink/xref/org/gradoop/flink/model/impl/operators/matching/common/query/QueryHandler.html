106     vertexCache = gdlHandler.getVertexCache(true, true);

460   /**
461    * Returns all vertices that are adjacent to the given vertex (without
462    * considering edge direction).
463    *
464    * @param vertexId vertex id
465    * @return adjacent vertices
466    */
467   public Collection<Vertex> getNeighbors(Long vertexId) {
468     Set<Vertex> neighbors = Sets.newHashSet();
469     Collection<Edge> outgoingEdges = getEdgesBySourceVertexId(vertexId);
470     if (outgoingEdges != null) {
471       for (Edge edge : outgoingEdges) {
472         neighbors.add(getVertexById(edge.getTargetVertexId()));
473       }
474     }
475     Collection<Edge> incomingEdges = getEdgesByTargetVertexId(vertexId);
476     if (incomingEdges != null) {
477       for (Edge edge : incomingEdges) {
478         neighbors.add(getVertexById(edge.getSourceVertexId()));
479       }
480     }
481     return neighbors;
482   }

499   /**
500    * Returns a mapping between the given variables (if existent) and the corresponding element
501    * label.
502    *
503    * @param variables query variables
504    * @return mapping between existing variables and their corresponding label
505    */
506   public Map<String, String> getLabelsForVariables(Collection<String> variables) {
507     return variables.stream()
508       .filter(var -> isEdge(var) || isVertex(var))
509       .map(var -> {
510         if (isEdge(var)) {
511           return Pair.of(var, getEdgeByVariable(var).getLabel());
512         } else {
513           return Pair.of(var, getVertexByVariable(var).getLabel());
514         }
515       }).collect(Collectors.toMap(Pair::getLeft, Pair::getRight));
516   }

580     List<Long> result = Lists.newArrayList();

589   /**
590    * Initializes a cache for the given data where every key maps to exactly one element (injective).
591    * Key selector will be called on every element to extract the caches key.
592    * Value selector will be called on every element to extract the value.
593    * Returns a cache of type
594    * KT -> VT
595    *
596    * @param elements elements the cache will be build from
597    * @param keySelector key selector function extraction cache keys from elements
598    * @param valueSelector value selector function extraction cache values from elements
599    * @param <EL> the element type
600    * @param <KT> the cache key type
601    * @param <VT> the cache value type
602    * @return cache KT -> VT
603    */
604   private <EL, KT, VT> Map<KT, VT> initCache(Collection<EL> elements,
605     Function<EL, KT> keySelector, Function<EL, VT> valueSelector) {
606 
607     return elements.stream().collect(Collectors.toMap(keySelector, valueSelector));
608   }
609 
610   /**
611    * Initializes a cache for the given elements where every key maps to multiple elements.
612    * Key selector will be called on every element to extract the caches key.
613    * Value selector will be called on every element to extract the value.
614    * Returns a cache of the form
615    * KT -> Set<VT>
616    *
617    * @param elements elements the cache will be build from
618    * @param keySelector key selector function extraction cache keys from elements
619    * @param valueSelector value selector function extraction cache values from elements
620    * @param <EL> the element type
621    * @param <KT> the cache key type
622    * @param <VT> the cache value type
623    * @return cache KT -> Set<VT>
624    */
625   private <EL, KT, VT> Map<KT, Set<VT>> initSetCache(Collection<EL> elements,
626     Function<EL, KT> keySelector, Function<EL, VT> valueSelector) {
627 
628     return elements.stream()
629       .collect(
630         Collectors.groupingBy(
631           keySelector,
632           Collectors.mapping(valueSelector, Collectors.toSet())
633       ));
634   }

636   /**
637    * Update variable names of vertices and edges with a generated variable name.
638    * This will also update the vertex- and edge-caches of this handler.
639    *
640    * @param renameFunction The renaming function, mapping old to new variable names.
641    */
642   public void updateGeneratedVariableNames(Function<String, String> renameFunction) {
643     Set<String> generatedEdgeVariables = gdlHandler.getEdgeCache(false, true).keySet();
644     Set<String> generatedVertexVariables = gdlHandler.getVertexCache(false, true).keySet();
645     Map<String, Vertex> newVertexCache = new HashMap<>();
646     Map<String, Edge> newEdgeCache = new HashMap<>();
647     // Update vertices.
648     for (Vertex vertex : getVertices()) {
649       final String variable = vertex.getVariable();
650       if (generatedVertexVariables.contains(variable)) {
651         vertex.setVariable(renameFunction.apply(variable));
652       }
653       newVertexCache.put(vertex.getVariable(), vertex);
654     }
655     // Update edges.
656     for (Edge edge : getEdges()) {
657       final String variable = edge.getVariable();
658       if (generatedEdgeVariables.contains(variable)) {
659         edge.setVariable(renameFunction.apply(variable));
660       }
661       newEdgeCache.put(edge.getVariable(), edge);
662     }
663     vertexCache = Collections.unmodifiableMap(newVertexCache);
664     edgeCache = Collections.unmodifiableMap(newEdgeCache);
665   }
