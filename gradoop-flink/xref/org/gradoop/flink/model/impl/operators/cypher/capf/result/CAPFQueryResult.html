52  /**
53   * Wrapper containing the results of a CAPF query.
54   */

57    /**
58     * Logical optimizer rules to be removed for better optimizer performance.
59     *
60     * {@code ProjectMergeRule:force_mode} would result in long optimization times.
61     */
62    private static final List<String> DISABLED_RULES =
63      Collections.singletonList("ProjectMergeRule:force_mode");

96     * Constructor;

99     * @param verticesWithIds map between long id and original vertex
100    * @param edgesWithIds    map between long id and original edge
101    * @param config          the gradoop config

103   public CAPFQueryResult(
104     CypherResult result,
105     DataSet<Tuple2<Long, Vertex>> verticesWithIds,
106     DataSet<Tuple2<Long, Edge>> edgesWithIds,
107     GradoopFlinkConfig config) {
108     this.records = (CAPFRecords) result.records();
109     this.verticesWithIds = verticesWithIds;
110     this.edgesWithIds = edgesWithIds;
111     this.config = config;
112 
113     this.session = ((CAPFRecords) result.records()).capf();
114     this.isGraph = !records.header().entityVars().isEmpty();

127   /**
128    * Get the graphs contained in the CAPF query result.
129    * Returns null if the result contains no graphs.
130    *
131    * @return graphs contained in CAPF query iff there are any, else null
132    */
133   public GraphCollection getGraphs() {
134 
135     if (!isGraph) {
136       return null;
137     }
138 
139     Set<Var> nodeVars = new HashSet<>();
140     Set<Var> relVars = new HashSet<>();
141     Set<Var> otherVars = new HashSet<>();
142 
143     Iterator<Var> varIt = records.header().vars().iterator();
144     while (varIt.hasNext()) {
145       otherVars.add(varIt.next());
146     }
147 
148     Iterator<Var> nodeVarIt = records.header().nodeEntities().iterator();
149     while (nodeVarIt.hasNext()) {
150       Var nodeVar = nodeVarIt.next();
151       nodeVars.add(nodeVar);
152       otherVars.remove(nodeVar);
153     }
154 
155     Iterator<Var> relVarIt = records.header().relationshipEntities().iterator();
156     while (relVarIt.hasNext()) {
157       Var relVar = relVarIt.next();
158       relVars.add(relVar);
159       otherVars.remove(relVar);
160     }
161 
162     StringBuilder entityFieldsBuilder = new StringBuilder();
163     for (Var var : nodeVars) {
164       entityFieldsBuilder.append(records.header().column((Expr) var)).append(",");
165     }
166 
167     for (Var var : relVars) {
168       entityFieldsBuilder.append(records.header().column((Expr) var)).append(",");
169     }
170 
171     StringBuilder otherFieldsBuilder = new StringBuilder();
172     List<String> otherVarNames = new ArrayList<>();
173 
174     for (Var var : otherVars) {
175       otherVarNames.add(var.name());
176       otherFieldsBuilder.append(
177         records.header().getColumn((Expr) var).get()).append(", ");
178     }
179 
180     String fieldString = entityFieldsBuilder.toString() + otherFieldsBuilder.toString();
181     if (fieldString.length() > 0) {
182       fieldString = fieldString.substring(0, fieldString.length() - 1);
183     }
184 
185     TypeInformation<Row> rowTypeInfo = TypeInformation.of(Row.class);
186 
187     // Workaround for usable optimizer times
188     removeSlowOptimizationRule();
189 
190     // entities, others, id
191     org.apache.flink.api.scala.DataSet<Row> scalarowsWithNewIds = session.tableEnv()
192       .toDataSet(records.table().table().select(fieldString), rowTypeInfo);
193 
194     DataSet<Row> rowsWithNewIds = scalarowsWithNewIds.javaSet()
195       .map(new AddGradoopIdToRow());
196 
197     int entityFieldsCount = nodeVars.size() + relVars.size();
198     int otherFieldsCount = otherVars.size();
199 
200     DataSet<GraphHead> graphHeads = rowsWithNewIds
201       .map(new CreateGraphHeadWithProperties(
202         entityFieldsCount,
203         entityFieldsCount + otherFieldsCount,
204         config.getGraphHeadFactory(),
205         otherVarNames)
206       );
207 
208     DataSet<Tuple2<Long, GradoopIdSet>> rowsWithGraphIdSets = rowsWithNewIds
209       .flatMap(new SplitRow(0, entityFieldsCount))
210       .groupBy(0)
211       .reduceGroup(new AggregateGraphs<>());
212 
213     DataSet<Vertex> vertices =
214       rowsWithGraphIdSets
215         .join(verticesWithIds)
216         .where(0)
217         .equalTo(0)
218         .with(new AddNewGraphs<>());
219 
220     DataSet<Edge> edges =
221       rowsWithGraphIdSets
222         .join(edgesWithIds)
223         .where(0)
224         .equalTo(0)
225         .with(new AddNewGraphs<>());
226 
227     vertices = vertices.map(new PropertyDecoder<>());
228     edges = edges.map(new PropertyDecoder<>());
229 
230     return config.getGraphCollectionFactory().fromDataSets(graphHeads, vertices, edges);
231   }

233   /**
234    * Workaround to remove slow logical optimization rules listed in
235    * {@link CAPFQueryResult#DISABLED_RULES}.
236    *
237    * This method accesses protected scala flink functions. See Issue #1221
238    * (https://github.com/dbs-leipzig/gradoop/issues/1221).
239    */
240   private void removeSlowOptimizationRule() {
241     List<RelOptRule> ruleList = new ArrayList<>();
242     for (RelOptRule rule : session.tableEnv().getLogicalOptRuleSet()) {
243       if (!DISABLED_RULES.contains(rule.toString())) {
244         ruleList.add(rule);
245       }
246     }
247     CalciteConfigBuilder builder = new CalciteConfigBuilder()
248       .replaceLogicalOptRuleSet(RuleSets.ofList(ruleList));
249 
250     // rebuild old calcite config
251     CalciteConfig calciteConfig = session.tableEnv().config().getCalciteConfig();
252     if (calciteConfig.replacesDecoRuleSet()) {
253       builder.replaceDecoRuleSet(calciteConfig.getDecoRuleSet().get());
254     }
255     if (calciteConfig.replacesNormRuleSet()) {
256       builder.replaceNormRuleSet(calciteConfig.getNormRuleSet().get());
257     }
258     if (calciteConfig.replacesPhysicalOptRuleSet()) {
259       builder.replacePhysicalOptRuleSet(calciteConfig.getPhysicalOptRuleSet().get());
260     }
261     if (calciteConfig.replacesSqlOperatorTable()) {
262       builder.replaceSqlOperatorTable(calciteConfig.getSqlOperatorTable().get());
263     }
264     if (calciteConfig.getSqlParserConfig().isDefined()) {
265       builder.replaceSqlParserConfig(calciteConfig.getSqlParserConfig().get());
266     }
267     if (calciteConfig.getSqlToRelConverterConfig().isDefined()) {
268       builder.replaceSqlToRelConverterConfig(calciteConfig.getSqlToRelConverterConfig().get());
269     }
270 
271     session.tableEnv().config().setCalciteConfig(builder.build());
272   }
