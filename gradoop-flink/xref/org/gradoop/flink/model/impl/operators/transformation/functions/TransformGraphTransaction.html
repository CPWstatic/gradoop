16  package org.gradoop.flink.model.impl.operators.transformation.functions;
17  
18  import org.apache.flink.api.common.functions.MapFunction;
19  import org.gradoop.common.model.api.entities.EPGMEdgeFactory;
20  import org.gradoop.common.model.api.entities.EPGMGraphHeadFactory;
21  import org.gradoop.common.model.api.entities.EPGMVertexFactory;
22  import org.gradoop.common.model.impl.pojo.Edge;
23  import org.gradoop.common.model.impl.pojo.GraphHead;
24  import org.gradoop.common.model.impl.pojo.Vertex;
25  import org.gradoop.common.util.GradoopConstants;
26  import org.gradoop.flink.model.api.functions.TransformationFunction;
27  import org.gradoop.flink.model.impl.layouts.transactional.tuples.GraphTransaction;
28  
29  import java.util.stream.Collectors;
30  
31  /**
32   * Transformation map function for graph transactions.
33   */
34  public class TransformGraphTransaction implements MapFunction<GraphTransaction, GraphTransaction> {
35  
36    /**
37     * Factory to init modified graph head.
38     */
39    private final EPGMGraphHeadFactory<GraphHead> graphHeadFactory;
40    /**
41     * Graph head modification function
42     */
43    private final TransformationFunction<GraphHead> graphHeadTransFunc;
44  
45    /**
46     * Factory to init modified vertex.
47     */
48    private final EPGMVertexFactory<Vertex> vertexFactory;
49  
50    /**
51     * Vertex modification function
52     */
53    private final TransformationFunction<Vertex> vertexTransFunc;
54  
55    /**
56     * Factory to init modified edge.
57     */
58    private final EPGMEdgeFactory<Edge> edgeFactory;
59    /**
60     * Edge modification function
61     */
62    private final TransformationFunction<Edge> edgeTransFunc;
63  
64    /**
65     * Constructor
66     *
67     * @param graphHeadFactory graph head factory
68     * @param graphHeadTransFunc graph head transformation function
69     * @param vertexFactory vertex factory
70     * @param vertexTransFunc vertex transformation function
71     * @param edgeFactory edge factory
72     * @param edgeTransFunc edge transformation function
73     */
74    public TransformGraphTransaction(EPGMGraphHeadFactory<GraphHead> graphHeadFactory,
75      TransformationFunction<GraphHead> graphHeadTransFunc, EPGMVertexFactory<Vertex> vertexFactory,
76      TransformationFunction<Vertex> vertexTransFunc, EPGMEdgeFactory<Edge> edgeFactory,
77      TransformationFunction<Edge> edgeTransFunc) {
78      this.graphHeadFactory = graphHeadFactory;
79      this.graphHeadTransFunc = graphHeadTransFunc;
80      this.vertexFactory = vertexFactory;
81      this.vertexTransFunc = vertexTransFunc;
82      this.edgeFactory = edgeFactory;
83      this.edgeTransFunc = edgeTransFunc;
84    }
85  
86    @Override
87    public GraphTransaction map(GraphTransaction transaction) throws Exception {
88      if (graphHeadTransFunc != null) {
89        transaction.setGraphHead(graphHeadTransFunc.apply(
90          transaction.getGraphHead(),
91          graphHeadFactory.initGraphHead(
92            transaction.getGraphHead().getId(), GradoopConstants.DEFAULT_GRAPH_LABEL)));
93      }
94  
95      if (vertexTransFunc != null) {
96        transaction.setVertices(transaction.getVertices().stream()
97          .map(vertex -> vertexTransFunc.apply(
98            vertex,
99            vertexFactory.initVertex(
100             vertex.getId(), GradoopConstants.DEFAULT_VERTEX_LABEL, vertex.getGraphIds())))
101         .collect(Collectors.toSet()));
102     }
103 
104     if (edgeTransFunc != null) {
105       transaction.setEdges(transaction.getEdges().stream()
106         .map(edge -> edgeTransFunc.apply(
107           edge,
108           edgeFactory.initEdge(
109             edge.getId(), GradoopConstants.DEFAULT_EDGE_LABEL,
110             edge.getSourceId(), edge.getTargetId(), edge.getGraphIds())))
111         .collect(Collectors.toSet()));
112     }
113     return transaction;
114   }
115 }
