44  /**
45   * Extracts a subgraph from a logical graph using the given filter functions.
46   * The graph head stays unchanged for the subgraph.
47   *
48   * The operator is able to:
49   * 1) extract vertex-induced subgraph
50   * 2) extract edge-induced subgraph via join + union strategy
51   * 2) extract edge-induced subgraph via project + union + join strategy
52   * 3) extract subgraph based on vertex and edge filter function
53   * 4) extract subgraph based on vertex and edge filter function without verification (no joins)
54   *
55   * @param <G> type of the graph head
56   * @param <V> the vertex type
57   * @param <E> the edge type
58   * @param <LG> type of the logical graph instance
59   * @param <GC> type of the graph collection
60   */
61  public class Subgraph<
62    G extends EPGMGraphHead,
63    V extends EPGMVertex,
64    E extends EPGMEdge,
65    LG extends BaseGraph<G, V, E, LG, GC>,
66    GC extends BaseGraphCollection<G, V, E, LG, GC>>
67    implements UnaryBaseGraphToBaseGraphOperator<LG> {
68  
69    /**
70     * Used to filter vertices from the logical graph.
71     */
72    private final FilterFunction<V> vertexFilterFunction;
73  
74    /**
75     * Used to filter edges from the logical graph.
76     */
77    private final FilterFunction<E> edgeFilterFunction;
78  
79    /**
80     * Execution strategy for the operator.
81     */
82    private final Strategy strategy;
83  
84    /**
85     * Available execution strategies.
86     */
87    public enum Strategy {
88      /**
89       * Applies both filter functions on the input vertex and edge data set.
90       */
91      BOTH,
92      /**
93       * Applies both filter functions on the input vertex and edge data set. In addition, this
94       * strategy verifies the consistency of the output graph by triplifying it projecting the
95       * vertices and edges.
96       */
97      BOTH_VERIFIED,
98      /**
99       * Only applies the vertex filter function and adds the incident edges connecting those
100      * vertices via a join.
101      */
102     VERTEX_INDUCED,
103     /**
104      * Only applies the edge filter function and computes the resulting vertices via:
105      * (E |><| V ON e.source = v.id) U (E |><| V on e.target = v.id)
106      */
107     EDGE_INDUCED,
108     /**
109      * Only applies the edge filter function and computes the resulting vertices via:
110      * DISTINCT((Ï_source(E) U Ï_target(E))) |><| V
111      */
112     EDGE_INDUCED_PROJECT_FIRST
113   }
114 
115   /**
116    * Creates a new sub graph operator instance.
117    *
118    * If both parameters are not {@code null}, the operator returns the subgraph
119    * defined by filtered vertices and edges.
120    *
121    * If the {@code edgeFilterFunction} is {@code null}, the operator returns the
122    * vertex-induced subgraph.
123    *
124    * If the {@code vertexFilterFunction} is {@code null}, the operator returns
125    * the edge-induced subgraph.
126    *
127    * @param vertexFilterFunction  vertex filter function
128    * @param edgeFilterFunction    edge filter function
129    * @param strategy              sets the execution strategy for the operator
130    */
131   public Subgraph(FilterFunction<V> vertexFilterFunction,
132     FilterFunction<E> edgeFilterFunction, Strategy strategy) {
133 
134     if ((strategy == BOTH || strategy == BOTH_VERIFIED) &&
135       vertexFilterFunction == null && edgeFilterFunction == null) {
136       throw new IllegalArgumentException("No filter functions was given.");
137     }
138 
139     if (strategy == VERTEX_INDUCED && vertexFilterFunction == null) {
140       throw new IllegalArgumentException("No vertex filter functions was given.");
141     }
142 
143     if ((strategy == EDGE_INDUCED || strategy == EDGE_INDUCED_PROJECT_FIRST) &&
144       edgeFilterFunction == null) {
145       throw new IllegalArgumentException("No vertex edge functions was given.");
146     }
147 
148     this.strategy = strategy;
149 
150     this.vertexFilterFunction = vertexFilterFunction;
151     this.edgeFilterFunction = edgeFilterFunction;
152   }
153 
154   @Override
155   public LG execute(LG superGraph) {
156 
157     LG result;
158     switch (strategy) {
159     case BOTH:
160       result = subgraph(superGraph);
161       break;
162     case BOTH_VERIFIED:
163       result = verify(subgraph(superGraph));
164       break;
165     case VERTEX_INDUCED:
166       result = vertexInducedSubgraph(superGraph);
167       break;
168     case EDGE_INDUCED:
169       result = edgeInducedSubgraph(superGraph);
170       break;
171     case EDGE_INDUCED_PROJECT_FIRST:
172       result = edgeInducedSubgraphProjectFirst(superGraph);
173       break;
174     default:
175       throw new IllegalArgumentException("Strategy " + strategy + " is not supported");
176     }
177 
178     return result;
179   }

181   /**
182    * Returns the subgraph of the given supergraph that is induced by the
183    * vertices that fulfil the given filter function.
184    *
185    * @param superGraph supergraph
186    * @return vertex-induced subgraph
187    */
188   private LG vertexInducedSubgraph(LG superGraph) {
189     DataSet<V> filteredVertices = superGraph.getVertices().filter(vertexFilterFunction);
190 
191     DataSet<E> newEdges = superGraph.getEdges()
192       .join(filteredVertices)
193       .where(new SourceId<>()).equalTo(new Id<>())
194       .with(new LeftSide<>())
195       .join(filteredVertices)
196       .where(new TargetId<>()).equalTo(new Id<>())
197       .with(new LeftSide<>());
198 
199     return superGraph.getFactory()
200       .fromDataSets(superGraph.getGraphHead(), filteredVertices, newEdges);
201   }
202 
203   /**
204    * Returns the subgraph of the given supergraph that is induced by the
205    * edges that fulfil the given filter function.
206    *
207    * @param superGraph supergraph
208    * @return edge-induced subgraph
209    */
210   private LG edgeInducedSubgraph(LG superGraph) {
211     DataSet<E> filteredEdges = superGraph.getEdges().filter(edgeFilterFunction);
212 
213     DataSet<V> filteredVertices = filteredEdges
214       .join(superGraph.getVertices())
215       .where(new SourceId<>()).equalTo(new Id<>())
216       .with(new RightSide<>())
217       .union(filteredEdges
218         .join(superGraph.getVertices())
219         .where(new TargetId<>()).equalTo(new Id<>())
220         .with(new RightSide<>()))
221       .distinct(new Id<>());
222 
223     return superGraph.getFactory()
224       .fromDataSets(superGraph.getGraphHead(), filteredVertices, filteredEdges);
225   }
226 
227   /**
228    * Returns the subgraph of the given supergraph that is induced by the
229    * edges that fulfil the given filter function.
230    *
231    * @param superGraph supergraph
232    * @return edge-induced subgraph
233    */
234   private LG edgeInducedSubgraphProjectFirst(LG superGraph) {
235     DataSet<E> filteredEdges = superGraph.getEdges().filter(edgeFilterFunction);
236 
237     DataSet<Tuple1<GradoopId>> vertexIdentifiers = filteredEdges
238       .map(new SourceId<>())
239       .map(new ObjectTo1<>())
240       .union(filteredEdges
241         .map(new TargetId<>())
242         .map(new ObjectTo1<>()))
243       .distinct();
244 
245     DataSet<V> filteredVertices = vertexIdentifiers
246       .join(superGraph.getVertices())
247       .where(0).equalTo(new Id<>())
248       .with(new RightSide<>());
249 
250     return superGraph.getFactory()
251       .fromDataSets(superGraph.getGraphHead(), filteredVertices, filteredEdges);
252   }
253 
254   /**
255    * Returns the subgraph of the given supergraph that is defined by the
256    * vertices that fulfil the vertex filter function and edges that fulfill
257    * the edge filter function.
258    *
259    * Note, that the operator does not verify the consistency of the resulting
260    * graph.
261    *
262    * @param superGraph supergraph
263    * @return subgraph
264    */
265   private LG subgraph(LG superGraph) {
266     return superGraph.getFactory().fromDataSets(
267       superGraph.getGraphHead(),
268       superGraph.getVertices().filter(vertexFilterFunction),
269       superGraph.getEdges().filter(edgeFilterFunction));
270   }
271 
272   /**
273    * Verifies that the given graph is consistent, contains only edges that connect to vertices
274    * within the subgraph.
275    *
276    * @param subgraph supergraph
277    * @return verified subgraph
278    */
279   private LG verify(LG subgraph) {
280 
281     DataSet<Tuple2<Tuple2<E, V>, V>> verifiedTriples = subgraph.getEdges()
282       .join(subgraph.getVertices())
283       .where(new SourceId<>()).equalTo(new Id<>())
284       .join(subgraph.getVertices())
285       .where("0.targetId").equalTo(new Id<>());
286 
287     DataSet<E> verifiedEdges = verifiedTriples
288       .map(new Value0Of2<>())
289       .map(new Value0Of2<>());
290 
291     DataSet<V> verifiedVertices = verifiedTriples
292       .map(new Value0Of2<>())
293       .map(new Value1Of2<>())
294       .union(verifiedTriples.map(new Value1Of2<>()))
295       .distinct(new Id<>());
296 
297     return subgraph.getFactory()
298       .fromDataSets(subgraph.getGraphHead(), verifiedVertices, verifiedEdges);
299   }
