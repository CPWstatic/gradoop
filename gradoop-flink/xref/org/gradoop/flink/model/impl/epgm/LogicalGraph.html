80  /**
81   * A logical graph is one of the base concepts of the Extended Property Graph Model. A logical graph
82   * encapsulates three concepts:
83   * <p>
84   * - a so-called graph head, that stores information about the graph (i.e. label and properties)
85   * - a set of vertices assigned to the graph
86   * - a set of directed, possibly parallel edges assigned to the graph
87   * <p>
88   * Furthermore, a logical graph provides operations that are performed on the underlying data. These
89   * operations result in either another logical graph or in a {@link GraphCollection}.
90   * <p>
91   * A logical graph is wrapping a {@link LogicalGraphLayout} which defines, how the graph is
92   * represented in Apache Flink. Note that the LogicalGraph also implements that interface and
93   * just forward the calls to the layout. This is just for convenience and API synchronicity.
94   */
95  public class LogicalGraph implements
96    BaseGraph<EPGMGraphHead, EPGMVertex, EPGMEdge, LogicalGraph, GraphCollection>, LogicalGraphOperators {

310   @Override
311   public LogicalGraph groupBy(
312     List<String> vertexGroupingKeys, List<AggregateFunction> vertexAggregateFunctions,
313     List<String> edgeGroupingKeys, List<AggregateFunction> edgeAggregateFunctions,
314     GroupingStrategy groupingStrategy) {
315 
316     Objects.requireNonNull(vertexGroupingKeys, "missing vertex grouping key(s)");
317     Objects.requireNonNull(groupingStrategy, "missing vertex grouping strategy");
318 
319     Grouping.GroupingBuilder builder = new Grouping.GroupingBuilder();
320 
321     builder.addVertexGroupingKeys(vertexGroupingKeys);
322     builder.setStrategy(groupingStrategy);
323 
324     if (edgeGroupingKeys != null) {
325       builder.addEdgeGroupingKeys(edgeGroupingKeys);
326     }
327     if (vertexAggregateFunctions != null) {
328       vertexAggregateFunctions.forEach(builder::addVertexAggregateFunction);
329     }
330     if (edgeAggregateFunctions != null) {
331       edgeAggregateFunctions.forEach(builder::addEdgeAggregateFunction);
332     }
333     return callForGraph(builder.build());
334   }
