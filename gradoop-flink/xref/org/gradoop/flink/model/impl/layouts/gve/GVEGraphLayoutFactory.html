35  /**
36   * Responsible for creating a {@link GVELayout} from given data.
37   */
38  public class GVEGraphLayoutFactory extends GVEBaseFactory
39    implements LogicalGraphLayoutFactory<GraphHead, Vertex, Edge> {

41    @Override
42    public GVELayout fromDataSets(DataSet<Vertex> vertices) {
43      return fromDataSets(vertices,
44        createEdgeDataSet(Lists.newArrayListWithCapacity(0)));
45    }

47    @Override
48    public GVELayout fromDataSets(DataSet<Vertex> vertices, DataSet<Edge> edges) {
49      Objects.requireNonNull(vertices, "Vertex DataSet was null");
50      Objects.requireNonNull(edges, "Edge DataSet was null");
51      GraphHead graphHead = getConfig()
52        .getGraphHeadFactory()
53        .createGraphHead();
54  
55      DataSet<GraphHead> graphHeadSet = getConfig().getExecutionEnvironment()
56        .fromElements(graphHead);
57  
58      // update vertices and edges with new graph head id
59      vertices = vertices
60        .map(new AddToGraph<>(graphHead))
61        .withForwardedFields("id;label;properties");
62      edges = edges
63        .map(new AddToGraph<>(graphHead))
64        .withForwardedFields("id;sourceId;targetId;label;properties");
65  
66      return new GVELayout(graphHeadSet, vertices, edges);
67    }
68  
69    @Override
70    public LogicalGraphLayout<GraphHead, Vertex, Edge> fromDataSets(DataSet<GraphHead> graphHead,
71      DataSet<Vertex> vertices, DataSet<Edge> edges) {
72      return create(graphHead, vertices, edges);
73    }
74  
75    @Override
76    public LogicalGraphLayout<GraphHead, Vertex, Edge> fromIndexedDataSets(
77      Map<String, DataSet<Vertex>> vertices,
78      Map<String, DataSet<Edge>> edges) {
79      GraphHead graphHead = getConfig()
80        .getGraphHeadFactory()
81        .createGraphHead();
82  
83      DataSet<GraphHead> graphHeadSet = getConfig().getExecutionEnvironment()
84        .fromElements(graphHead);
85  
86      Map<String, DataSet<GraphHead>> graphHeads = Maps.newHashMap();
87      graphHeads.put(graphHead.getLabel(), graphHeadSet);
88  
89      // update vertices and edges with new graph head id
90      vertices = vertices.entrySet().stream()
91        .collect(Collectors.toMap(
92          Map.Entry::getKey, e -> e.getValue().map(new AddToGraph<>(graphHead))
93            .withForwardedFields("id;label;properties")));
94  
95      edges = edges.entrySet().stream()
96        .collect(Collectors.toMap(
97          Map.Entry::getKey, e -> e.getValue().map(new AddToGraph<>(graphHead))
98            .withForwardedFields("id;sourceId;targetId;label;properties")));
99  
100     return create(graphHeads, vertices, edges);
101   }

105     Map<String, DataSet<GraphHead>> graphHeads,

111   @Override
112   public LogicalGraphLayout<GraphHead, Vertex, Edge> fromCollections(GraphHead graphHead,
113     Collection<Vertex> vertices,
114     Collection<Edge> edges) {
115 
116     Objects.requireNonNull(vertices, "Vertex collection was null");
117 
118     List<GraphHead> graphHeads;
119     if (graphHead == null) {
120       graphHeads = Lists.newArrayListWithCapacity(0);
121     } else {
122       graphHeads = Lists.newArrayList(graphHead);
123     }
124     if (edges == null) {
125       edges = Lists.newArrayListWithCapacity(0);
126     }
127 
128     return fromDataSets(
129       createGraphHeadDataSet(graphHeads),
130       createVertexDataSet(vertices),
131       createEdgeDataSet(edges));
132   }
133 
134   @Override
135   public LogicalGraphLayout<GraphHead, Vertex, Edge> fromCollections(Collection<Vertex> vertices,
136     Collection<Edge> edges) {
137     Objects.requireNonNull(vertices, "Vertex collection was null");
138     Objects.requireNonNull(edges, "Edge collection was null");
139     return fromDataSets(createVertexDataSet(vertices), createEdgeDataSet(edges));
140   }
141 
142   @Override
143   public LogicalGraphLayout<GraphHead, Vertex, Edge> createEmptyGraph() {
144     Collection<Vertex> vertices = new ArrayList<>(0);
145     Collection<Edge> edges = new ArrayList<>(0);
146     return fromCollections(null, vertices, edges);
147   }
