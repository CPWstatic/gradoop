33  /**
34   * Converts a GraphTransaction to the following .dot format:
35   * <p>
36   *   digraph 0
37   *   {
38   *   gradoopId1 [label="person",name="Bob",age="20",...];
39   *   gradoopId2 [label="person",name="Alice",age="20",...];
40   *   gradoopID3;
41   *   gradoopID4;
42   *   gradoopId1->gradoopId2 [label="knows",since="2003",...];
43   *   gradoopId2->gradoopId1 [label="knows",since="2003",...];
44   *   gradoopId3->gradoopId4;
45   *   }
46   * </p>
47   */
48  public class DOTFileFormat implements TextOutputFormat.TextFormatter<GraphTransaction> {
49    /**
50     * id that needs to be update upon changes to this class' structure.
51     */
52    private static final long serialVersionUID = 1L;
53    /**
54     * .DOT vertex identifier prefix
55     */
56    private static final String VERTEX_ID_PREFIX = "v";
57    /**
58     * flag to print graph head information to dot
59     */
60    private boolean printGraphHead;
61  
62    /**
63     * Constructor
64     *
65     * @param printGraphHead true, iff graph head data shall be attached to the output
66     */
67    public DOTFileFormat(Boolean printGraphHead) {
68      this.printGraphHead = printGraphHead;
69    }
70  
71    @Override
72    public String format(GraphTransaction transaction) {
73  
74      StringBuilder builder = new StringBuilder();
75  
76      //--------------------------------------------------------------------------
77      // write DOT head lines and open block
78      //--------------------------------------------------------------------------
79  
80      GradoopId id = transaction.getGraphHead().getId();
81      // writes for each graph:
82      // digraph graphHeadId
83      // {
84      builder.append("subgraph cluster_g")
85        .append(id)
86        .append("{\n");
87  
88      //--------------------------------------------------------------------------
89      // write DOT body
90      //--------------------------------------------------------------------------
91  
92      if (printGraphHead) {
93        writeGraphHead(transaction, builder);
94      }
95  
96      writeVertices(transaction, builder, id.toString());
97  
98      writeEdges(transaction, builder, id.toString());
99  
100     //--------------------------------------------------------------------------
101     // close DOT block
102     //--------------------------------------------------------------------------
103 
104     builder.append("}\n");
105 
106     return builder.toString();
107   }
108 
109   /**
110    * Adds graph head information to the specified builder.
111    *
112    * Output: label="label";
113    * @param transaction graph transaction
114    * @param builder string builder to append
115    */
116   private void writeGraphHead(GraphTransaction transaction, StringBuilder builder) {
117     GraphHead graphHead = transaction.getGraphHead();
118 
119     writeLabel(builder, graphHead, "#AAAAAA");
120 
121     builder.append(";\n");
122   }
123 
124   /**
125    * Adds vertex information to the specified builder.
126    *
127    * vertexId [label="label", property1="value1", ...];
128    *
129    * @param transaction graph transaction
130    * @param builder string builder to append
131    * @param suffix id suffix specific for the current {@link GraphTransaction}
132    */
133   private void writeVertices(GraphTransaction transaction, StringBuilder builder, String suffix) {
134     for (Vertex vertex: transaction.getVertices()) {
135       // writes for each vertex:
136       // "v1234",
137       builder.append(VERTEX_ID_PREFIX)
138         .append(vertex.getId())
139         .append(suffix)
140         .append(" [ shape=Mrecord, ");
141 
142       writeLabel(builder, vertex, "#000000");
143 
144       // writes:
145       // ";"
146       builder.append("];\n");
147     }
148   }
149 
150   /**
151    * Adds edge information to the specified builder
152    *
153    * sourceId->targetId [label="label", property1="value1", ...];
154    *
155    * @param transaction graph transaction
156    * @param builder string builder to append
157    * @param suffix id suffix specific for the current {@link GraphTransaction}
158    */
159   private void writeEdges(GraphTransaction transaction, StringBuilder builder, String suffix) {
160     for (Edge edge: transaction.getEdges()) {
161 
162       builder.append(VERTEX_ID_PREFIX)
163         .append(edge.getSourceId())
164         .append(suffix)
165         .append("->")
166         .append(VERTEX_ID_PREFIX)
167         .append(edge.getTargetId())
168         .append(suffix)
169         .append(" [");
170       // write dot attributes if existent
171       writeLabel(builder, edge, "#666666");
172       builder.append("];\n");
173     }
174   }
175 
176   /**
177    * Writes the specified label and properties as DOT HTML label string (table)
178    *
179    * output: label=<<table>...</table>>
180    *
181    * @param builder string builder to append
182    * @param elem graph element with id, label and properties
183    * @param color color for header background and properties text
184    */
185   private void writeLabel(StringBuilder builder, EPGMElement elem, String color) {
186     String label = elem.getLabel();
187     String id = elem.getId().toString();
188     Properties properties = elem.getProperties();
189     String lbl = StringUtils.isEmpty(label) ? id : label;
190 
191     if (properties != null && properties.size() > 0) {
192       // writes properties as rows in html table
193       //write white on black label/id as header
194       builder.append("label=<")
195         .append("<font color=\"").append(color).append("\">")
196         .append("<table border=\"0\" cellborder=\"0\" cellpadding=\"3\">")
197         .append("<tr><td colspan=\"2\" bgcolor=\"")
198         .append(color)
199         .append("\"><font color=\"white\">")
200         .append(escapeHtml4(lbl))
201         .append("</font></td></tr>");
202 
203       Iterator<Property> iterator = properties.iterator();
204       while (iterator.hasNext()) {
205         Property property = iterator.next();
206         builder.append("<tr><td>")
207           .append(escapeHtml4(property.getKey()))
208           .append("</td><td>")
209           .append(escapeHtml4(property.getValue().toString()))
210           .append("</td></tr>");
211       }
212       builder.append("</table></font>>");
213     } else {
214       //write id/label as node label in dot
215       builder.append("label=\"")
216         .append(escapeHtml4(lbl))
217         .append("\"");
218     }
219   }
