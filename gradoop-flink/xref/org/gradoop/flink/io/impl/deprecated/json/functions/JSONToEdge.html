18  import org.apache.flink.api.common.functions.MapFunction;
19  import org.codehaus.jettison.json.JSONException;
20  import org.codehaus.jettison.json.JSONObject;
21  import org.gradoop.common.model.impl.pojo.Edge;
22  import org.gradoop.flink.io.impl.deprecated.json.JSONConstants;
23  import org.gradoop.common.model.api.entities.EPGMEdgeFactory;
24  import org.gradoop.common.model.impl.id.GradoopId;
25  import org.gradoop.common.model.impl.id.GradoopIdSet;
26  import org.gradoop.common.model.impl.properties.Properties;

28  /**
29   * Reads edge data from a json document. The document contains at least
30   * the edge id, the source vertex id, the target vertex id, an embedded
31   * data document and an embedded meta document.
32   * The data document contains all key-value pairs stored at the edge, the
33   * meta document contains the edge label and an optional list of graph
34   * identifiers the edge is contained in.
35   * <p>
36   * Example:
37   * <p>
38   * {
39   * "id":0,"start":15,"end":12,
40   * "data":{"since":2015},
41   * "meta":{"label":"worksFor","graphs":[1,2,3,4]}
42   * }
43   */
44  public class JSONToEdge extends JSONToEntity
45    implements MapFunction<String, Edge> {
46    /**
47     * Edge data factory.
48     */
49    private final EPGMEdgeFactory<Edge> edgeFactory;
50  
51    /**
52     * Creates map function.
53     *
54     * @param epgmEdgeFactory edge data factory
55     */
56    public JSONToEdge(EPGMEdgeFactory<Edge> epgmEdgeFactory) {
57      this.edgeFactory = epgmEdgeFactory;
58    }
59  
60    /**
61     * Creates an edge from JSON string representation.
62     *
63     * @param s json string
64     * @return Gelly edge storing gradoop edge data
65     * @throws Exception on failure
66     */
67    @Override
68    public Edge map(String s) throws Exception {
69      JSONObject jsonEdge = new JSONObject(s);
70      GradoopId edgeID = getID(jsonEdge);
71      String edgeLabel = getLabel(jsonEdge);
72      GradoopId sourceID = getSourceId(jsonEdge);
73      GradoopId targetID = getTargetId(jsonEdge);
74      Properties properties = Properties.createFromMap(
75        getProperties(jsonEdge));
76      GradoopIdSet graphs = getGraphs(jsonEdge);
77  
78      return edgeFactory.initEdge(edgeID, edgeLabel, sourceID, targetID,
79        properties, graphs);
80    }
81  
82    /**
83     * Reads the source vertex identifier from the json object.
84     *
85     * @param jsonEdge json string representation
86     * @return source vertex identifier
87     * @throws JSONException if failed to parse JSON
88     */
89    private GradoopId getSourceId(JSONObject jsonEdge) throws JSONException {
90      return GradoopId.fromString(jsonEdge.getString(JSONConstants.EDGE_SOURCE));
91    }
92  
93    /**
94     * Reads the target vertex identifier from the json object.
95     *
96     * @param jsonEdge json string representation
97     * @return target vertex identifier
98     * @throws JSONException if failed to parse JSON
99     */
100   private GradoopId getTargetId(JSONObject jsonEdge) throws JSONException {
101     return GradoopId.fromString(jsonEdge.getString(JSONConstants.EDGE_TARGET));
102   }
103 }
