37  /**
38   * A graph data source for CSV files indexed by label.
39   *
40   * The datasource expects files separated by label, e.g. in the following directory structure:
41   *
42   * csvRoot
43   *   |- Person.csv     # contains all vertices with label 'Person'
44   *   |- University.csv # contains all vertices with label 'University'
45   *   |- knows.csv      # contains all edges with label 'knows'
46   *   |- studyAy.csv    # contains all edges with label 'studyAt'
47   *   |- metadata.csv   # Meta data for all data contained in the graph
48   */
49  @Deprecated
50  public class LogicalGraphIndexedCSVDataSource extends LogicalGraphCSVBase implements DataSource {
51    /**
52     * HDFS Configuration
53     */
54    private final Configuration hdfsConfig;
55  
56    /**
57     * Creates a new data source. The constructor creates a default HDFS configuration.
58     *
59     * @param csvPath root path of csv files
60     * @param config gradoop configuration
61     */
62    public LogicalGraphIndexedCSVDataSource(String csvPath, GradoopFlinkConfig config) {
63      this(csvPath, config, new Configuration());
64    }
65  
66    /**
67     * Creates a new data source.
68     *
69     * @param csvPath root path of csv files
70     * @param conf gradoop configuration
71     * @param hdfsConf HDFS configuration
72     */
73    public LogicalGraphIndexedCSVDataSource(
74      String csvPath, GradoopFlinkConfig conf, Configuration hdfsConf) {
75      super(csvPath, conf);
76      Objects.requireNonNull(hdfsConf);
77      this.hdfsConfig = hdfsConf;
78    }
79  
80    @Override
81    public LogicalGraph getLogicalGraph() throws IOException {
82      MetaData metaData = MetaData.fromFile(getMetaDataPath(), hdfsConfig);
83  
84      ExecutionEnvironment env = getConfig().getExecutionEnvironment();
85      LogicalGraphFactory factory = getConfig().getLogicalGraphFactory();
86      VertexFactory<EPGMVertex> vertexFactory = factory.getVertexFactory();
87      EdgeFactory<EPGMEdge> edgeFactory = factory.getEdgeFactory();
88  
89      Map<String, DataSet<EPGMVertex>> vertices = metaData.getVertexLabels().stream()
90        .map(l -> Tuple2.of(l, env.readTextFile(getVertexCSVPath(l))
91          .map(new CSVLineToVertex(vertexFactory))
92          .withBroadcastSet(MetaData.fromFile(getMetaDataPath(), getConfig()), BC_METADATA)))
93        .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
94  
95      Map<String, DataSet<EPGMEdge>> edges = metaData.getEdgeLabels().stream()
96        .map(l -> Tuple2.of(l, env.readTextFile(getEdgeCSVPath(l))
97          .map(new CSVLineToEdge(edgeFactory))
98          .withBroadcastSet(MetaData.fromFile(getMetaDataPath(), getConfig()), BC_METADATA)))
99        .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
100 
101     return factory.fromIndexedDataSets(vertices, edges);
102   }
103 
104   @Override
105   public GraphCollection getGraphCollection() throws IOException {
106     LogicalGraph logicalGraph = getLogicalGraph();
107     return logicalGraph.getCollectionFactory().fromGraph(logicalGraph);
108   }
