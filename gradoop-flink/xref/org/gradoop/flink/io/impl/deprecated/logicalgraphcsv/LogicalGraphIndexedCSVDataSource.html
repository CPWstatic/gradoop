36  /**
37   * A graph data source for CSV files indexed by label.
38   *
39   * The datasource expects files separated by label, e.g. in the following directory structure:
40   *
41   * csvRoot
42   *   |- Person.csv     # contains all vertices with label 'Person'
43   *   |- University.csv # contains all vertices with label 'University'
44   *   |- knows.csv      # contains all edges with label 'knows'
45   *   |- studyAy.csv    # contains all edges with label 'studyAt'
46   *   |- metadata.csv   # Meta data for all data contained in the graph
47   */
48  @Deprecated
49  public class LogicalGraphIndexedCSVDataSource extends LogicalGraphCSVBase implements DataSource {
50    /**
51     * HDFS Configuration
52     */
53    private final Configuration hdfsConfig;
54  
55    /**
56     * Creates a new data source. The constructor creates a default HDFS configuration.
57     *
58     * @param csvPath root path of csv files
59     * @param config gradoop configuration
60     */
61    public LogicalGraphIndexedCSVDataSource(String csvPath, GradoopFlinkConfig config) {
62      this(csvPath, config, new Configuration());
63    }
64  
65    /**
66     * Creates a new data source.
67     *
68     * @param csvPath root path of csv files
69     * @param conf gradoop configuration
70     * @param hdfsConf HDFS configuration
71     */
72    public LogicalGraphIndexedCSVDataSource(
73      String csvPath, GradoopFlinkConfig conf, Configuration hdfsConf) {
74      super(csvPath, conf);
75      Objects.requireNonNull(hdfsConf);
76      this.hdfsConfig = hdfsConf;
77    }
78  
79    @Override
80    public LogicalGraph getLogicalGraph() throws IOException {
81      MetaData metaData = MetaData.fromFile(getMetaDataPath(), hdfsConfig);
82  
83      ExecutionEnvironment env = getConfig().getExecutionEnvironment();
84      VertexFactory vertexFactory = getConfig().getVertexFactory();
85      EdgeFactory edgeFactory = getConfig().getEdgeFactory();
86  
87      Map<String, DataSet<Vertex>> vertices = metaData.getVertexLabels().stream()
88        .map(l -> Tuple2.of(l, env.readTextFile(getVertexCSVPath(l))
89          .map(new CSVLineToVertex(vertexFactory))
90          .withBroadcastSet(MetaData.fromFile(getMetaDataPath(), getConfig()), BC_METADATA)))
91        .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
92  
93      Map<String, DataSet<Edge>> edges = metaData.getEdgeLabels().stream()
94        .map(l -> Tuple2.of(l, env.readTextFile(getEdgeCSVPath(l))
95          .map(new CSVLineToEdge(edgeFactory))
96          .withBroadcastSet(MetaData.fromFile(getMetaDataPath(), getConfig()), BC_METADATA)))
97        .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
98  
99      return getConfig().getLogicalGraphFactory().fromIndexedDataSets(vertices, edges);
100   }
101 
102   @Override
103   public GraphCollection getGraphCollection() throws IOException {
104     return getConfig().getGraphCollectionFactory().fromGraph(getLogicalGraph());
105   }
106 }
