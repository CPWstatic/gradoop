16  package org.gradoop.flink.io.impl.deprecated.json.functions;
17  
18  import org.codehaus.jettison.json.JSONArray;
19  import org.codehaus.jettison.json.JSONException;
20  import org.codehaus.jettison.json.JSONObject;
21  import org.gradoop.common.model.api.entities.EPGMGraphElement;
22  import org.gradoop.common.model.impl.pojo.GraphHead;
23  import org.gradoop.flink.io.impl.deprecated.json.JSONConstants;
24  import org.gradoop.common.model.api.entities.EPGMAttributed;
25  import org.gradoop.common.model.api.entities.EPGMLabeled;
26  import org.gradoop.common.model.impl.id.GradoopId;
27  import org.gradoop.common.model.impl.id.GradoopIdSet;
28  
29  /**
30   * Contains methods used by all entity writers (e.g. write meta, data).
31   */
32  public class EntityToJSON {
33  
34    /**
35     * Writes all key-value properties to a json object and returns it.
36     *
37     * @param entity entity with key-value properties
38     * @return json object containing the properties
39     * @throws JSONException if failed to create JSON
40     */
41    protected JSONObject writeProperties(EPGMAttributed entity) throws
42      JSONException {
43      JSONObject data = new JSONObject();
44      if (entity.getPropertyCount() > 0) {
45        for (String propertyKey : entity.getPropertyKeys()) {
46          data.put(
47            propertyKey, entity.getPropertyValue(propertyKey).getObject());
48        }
49      }
50      return data;
51    }
52  
53    /**
54     * Writes all meta data regarding a labeled graph element to a json
55     * object and returns it.
56     *
57     * @param entity labeled graph element (e.g., vertex and edge)
58     * @param <T>    input element type
59     * @return json object containing meta information
60     * @throws JSONException if failed to create JSON
61     */
62    protected <T extends EPGMLabeled & EPGMGraphElement> JSONObject
63    writeGraphElementMeta(
64      T entity) throws JSONException {
65      JSONObject meta = writeMeta(entity);
66      if (entity.getGraphCount() > 0) {
67        meta.put(JSONConstants.GRAPHS, writeJsonArray(entity.getGraphIds()));
68      }
69      return meta;
70    }
71  
72    /**
73     * Writes all meta data regarding a logical graph to a json object and
74     * returns it.
75     *
76     * @param entity logical graph data
77     * @param <T>    graph data type
78     * @return json object with graph meta data
79     * @throws JSONException if failed to create JSON
80     */
81    protected <T extends GraphHead> JSONObject
82    writeGraphMeta(T entity) throws JSONException {
83      return writeMeta(entity);
84    }
85  
86    /**
87     * Writes meta data (e.g., label) to a json object and returns it.
88     *
89     * @param entity labeled entity
90     * @return json object with meta data containing the label
91     * @throws JSONException if failed to create JSON
92     */
93    private JSONObject writeMeta(EPGMLabeled entity) throws JSONException {
94      JSONObject meta = new JSONObject();
95      meta.put(JSONConstants.LABEL, entity.getLabel());
96      return meta;
97    }
98  
99    /**
100    * Creates a json array from a given set of identifiers.
101    *
102    * @param values identifier set
103    * @return json array containing the identifiers
104    */
105   private JSONArray writeJsonArray(final GradoopIdSet values) {
106     JSONArray jsonArray = new JSONArray();
107     for (GradoopId val : values) {
108       jsonArray.put(val);
109     }
110     return jsonArray;
111   }
112 }
