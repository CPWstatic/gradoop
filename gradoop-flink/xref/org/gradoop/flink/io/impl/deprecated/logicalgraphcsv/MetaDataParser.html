45     * Used to separate property meta data.

47    private static final String PROPERTY_DELIMITER = ",";

49     * Used to separate list items

51    private static final String LIST_DELIMITER = ", ";

53     * Used to separate property tokens (property-key, property-type)

55    private static final String PROPERTY_TOKEN_DELIMITER = ":";

57     * Used to map a type string to its corresponding parsing function

59    private static final Map<String, Function<String, Object>> TYPE_PARSER_MAP = getTypeParserMap();

61    /**
62     * Creates the type - parser function mapping of static property TYPE_PARSER_MAP
63     *
64     * @return a HashMap containing the mapping of a type string to its corresponding parsing function
65     */
66    private static Map<String, Function<String, Object>> getTypeParserMap() {
67      Map<String, Function<String, Object>> map = new HashMap<>();
68      map.put(TypeString.SHORT.getTypeString(), Short::parseShort);
69      map.put(TypeString.INTEGER.getTypeString(), Integer::parseInt);
70      map.put(TypeString.LONG.getTypeString(), Long::parseLong);
71      map.put(TypeString.FLOAT.getTypeString(), Float::parseFloat);
72      map.put(TypeString.DOUBLE.getTypeString(), Double::parseDouble);
73      map.put(TypeString.BOOLEAN.getTypeString(), Boolean::parseBoolean);
74      map.put(TypeString.STRING.getTypeString(), s -> s);
75      map.put(TypeString.BIGDECIMAL.getTypeString(), BigDecimal::new);
76      map.put(TypeString.GRADOOPID.getTypeString(), GradoopId::fromString);
77      map.put(TypeString.MAP.getTypeString(), MetaDataParser::parseMapProperty);
78      map.put(TypeString.LIST.getTypeString(), MetaDataParser::parseListProperty);
79      map.put(TypeString.LOCALDATE.getTypeString(), LocalDate::parse);
80      map.put(TypeString.LOCALTIME.getTypeString(), LocalTime::parse);
81      map.put(TypeString.LOCALDATETIME.getTypeString(), LocalDateTime::parse);
82      map.put(TypeString.NULL.getTypeString(), MetaDataParser::parseNullProperty);
83      map.put(TypeString.SET.getTypeString(), MetaDataParser::parseSetProperty);
84      return Collections.unmodifiableMap(map);
85    }

87    /**
88     * Creates a {@link MetaData} object from the specified lines. The specified tuple is already
89     * separated into the label and the
90     *
91     * @param metaDataStrings (element prefix (g,v,e), label, meta-data) tuples
92     * @return Meta Data object
93     */
94    public static MetaData create(List<Tuple3<String, String, String>> metaDataStrings) {
95      Map<Tuple2<String, String>, List<PropertyMetaData>> metaDataMap
96        = new HashMap<>(metaDataStrings.size());
97  
98      for (Tuple3<String, String, String> tuple : metaDataStrings) {
99        List<PropertyMetaData> propertyMetaDataList;
100       if (tuple.f2.length() > 0) {
101         String[] propertyStrings = tuple.f2.split(PROPERTY_DELIMITER);
102         propertyMetaDataList = new ArrayList<>(propertyStrings.length);
103         for (String propertyString : propertyStrings) {
104           String[] propertyMetadata = propertyString.split(PROPERTY_TOKEN_DELIMITER, 2);
105           String[] propertyTypeTokens = propertyMetadata[1].split(PROPERTY_TOKEN_DELIMITER);
106           if (propertyTypeTokens.length == 2 &&
107             propertyTypeTokens[0].equals(TypeString.LIST.getTypeString())) {
108             // it's a list with one additional data type (type of list items)
109             propertyMetaDataList.add(new PropertyMetaData(
110               propertyMetadata[0], propertyMetadata[1], getListValueParser(propertyTypeTokens[1])));
111           } else if (propertyTypeTokens.length == 2 &&
112             propertyTypeTokens[0].equals(TypeString.SET.getTypeString())) {
113             // it's a set with one additional data type (type of set items)
114             propertyMetaDataList.add(new PropertyMetaData(
115               propertyMetadata[0], propertyMetadata[1], getSetValueParser(propertyTypeTokens[1])));
116           } else if (propertyTypeTokens.length == 3 &&
117             propertyTypeTokens[0].equals(TypeString.MAP.getTypeString())) {
118             // it's a map with two additional data types (key type + value type)
119             propertyMetaDataList.add(
120               new PropertyMetaData(
121                 propertyMetadata[0],
122                 propertyMetadata[1],
123                 getMapValueParser(propertyTypeTokens[1], propertyTypeTokens[2])
124               )
125             );
126           } else {
127             propertyMetaDataList.add(new PropertyMetaData(
128               propertyMetadata[0], propertyMetadata[1], getValueParser(propertyMetadata[1])));
129           }
130         }
131       } else {
132         propertyMetaDataList = new ArrayList<>(0);
133       }
134       metaDataMap.put(new Tuple2<>(tuple.f0, tuple.f1), propertyMetaDataList);
135     }
136 
137     return new MetaData(metaDataMap);
138   }

142    * property key and the property value type, e.g. "foo:int".

144    * @param property property
145    * @return property meta data

147   public static String getPropertyMetaData(Property property) {
148     return String.format("%s%s%s",
149       property.getKey(), PROPERTY_TOKEN_DELIMITER, getTypeString(property.getValue()));

164   /**
165    * Creates a parsing function for the given property type.
166    *
167    * @param type property type
168    * @return parsing function
169    */
170   private static Function<String, Object> getValueParser(String type) {
171     type = type.toLowerCase();
172     if (TYPE_PARSER_MAP.containsKey(type)) {
173       return TYPE_PARSER_MAP.get(type);
174     } else {
175       throw new IllegalArgumentException("Type " + type + " is not supported");
176     }
177   }

179   /**
180    * Creates a parsing function for list property type.
181    *
182    * @param listItemType string representation of the list item type, e.g. "String"
183    * @return parsing function
184    */
185   private static Function<String, Object> getListValueParser(String listItemType) {
186     final String itemType = listItemType.toLowerCase();
187     // check the validity of the list item type
188     if (!TYPE_PARSER_MAP.containsKey(itemType)) {
189       throw new TypeNotPresentException(itemType, null);
190     }
191 
192     return s -> parseListProperty(s, TYPE_PARSER_MAP.get(itemType));
193   }

195   /**
196    * Creates a parsing function for map property type.
197    *
198    * @param keyType string representation of the map key type, e.g. "String"
199    * @param valueType string representation of the map value type, e.g. "Double"
200    * @return parsing function
201    */
202   private static Function<String, Object> getMapValueParser(String keyType, String valueType) {
203     final String keyTypeLowerCase = keyType.toLowerCase();
204     // check the validity of the map key type
205     if (!TYPE_PARSER_MAP.containsKey(keyTypeLowerCase)) {
206       throw new TypeNotPresentException(keyTypeLowerCase, null);
207     }
208 
209     final String valueTypeLowerCase = valueType.toLowerCase();
210     // check the validity of the map value type
211     if (!TYPE_PARSER_MAP.containsKey(valueTypeLowerCase)) {
212       throw new TypeNotPresentException(keyTypeLowerCase, null);
213     }
214 
215     return s -> parseMapProperty(
216       s,
217       TYPE_PARSER_MAP.get(keyTypeLowerCase),
218       TYPE_PARSER_MAP.get(valueTypeLowerCase)
219     );
220   }

237   /**
238    * Parse function to translate string representation of a List to a list of PropertyValues
239    * Every PropertyValue has the type "string", because there is no parser for the items given
240    * Use {@link #parseListProperty(String, Function)} to specify a parsing function
241    *
242    * @param s the string to parse as list, e.g. "[myString1, myString2]"
243    * @return the list represented by the argument
244    */
245   private static Object parseListProperty(String s) {
246     // no item type given, so use string as type
247     s = s.replace("[", "").replace("]", "");
248     return Arrays.stream(s.split(LIST_DELIMITER))
249       .map(PropertyValue::create)
250       .collect(Collectors.toList());
251   }

253   /**
254    * Parse function to translate string representation of a List to a list of PropertyValues

256    * @param s the string to parse as list, e.g. "[myString1, myString2]"
257    * @param itemParser the function to parse the list items
258    * @return the list represented by the argument

260   private static Object parseListProperty(String s, Function<String, Object> itemParser) {
261     s = s.replace("[", "").replace("]", "");

264       .map(PropertyValue::create)
265       .collect(Collectors.toList());

268   /**
269    * Parse function to translate string representation of a Map to a Map with key and value of
270    * type PropertyValue. Every PropertyValue (key and value) has the type "string", because there
271    * are no parsers for the keys and values given. Use
272    * {@link #parseMapProperty(String, Function, Function)} to specify both parsing functions.
273    *
274    * @param s the string to parse as map, e.g. "{myString1=myValue1, myString2=myValue2}"
275    * @return the map represented by the argument
276    */
277   private static Object parseMapProperty(String s) {
278     // no key type and value type given, so use string as types
279     s = s.replace("{", "").replace("}", "");
280     return Arrays.stream(s.split(LIST_DELIMITER))
281       .map(st -> st.split("="))
282       .collect(Collectors.toMap(e -> PropertyValue.create(e[0]), e -> PropertyValue.create(e[1])));
283   }

285   /**
286    * Parse function to translate string representation of a Map to a Map with
287    * key and value of type PropertyValue
288    *
289    * @param s the string to parse as map, e.g. "{myString1=myValue1, myString2=myValue2}"
290    * @param keyParser the function to parse the keys
291    * @param valueParser the function to parse the values
292    * @return the map represented by the argument
293    */
294   private static Object parseMapProperty(
295     String s,
296     Function<String, Object> keyParser,
297     Function<String, Object> valueParser
298   ) {
299     s = s.replace("{", "").replace("}", "");
300     return Arrays.stream(s.split(LIST_DELIMITER))
301       .map(st -> st.split("="))
302       .map(strings -> {
303         Object[] objects = new Object[2];
304         objects[0] = keyParser.apply(strings[0]);
305         objects[1] = valueParser.apply(strings[1]);
306         return objects;
307       })
308       .collect(Collectors.toMap(e -> PropertyValue.create(e[0]), e -> PropertyValue.create(e[1])));
309   }

341   /**
342    * Parse function to create null from the null string representation.
343    *
344    * @param nullString The string representing null.
345    * @throws IllegalArgumentException The string that is passed has to represent null.
346    * @return Returns null
347    */
348   private static Object parseNullProperty(String nullString) throws IllegalArgumentException {
349     if (nullString != null && nullString.equalsIgnoreCase(TypeString.NULL.getTypeString())) {
350       return null;
351     } else {
352       throw new IllegalArgumentException("Only null represents a null string.");
353     }

356   /**
357    * Returns the type string for the specified property value.
358    *
359    * @param propertyValue property value
360    * @return property type string
361    */
362   public static String getTypeString(PropertyValue propertyValue) {
363     if (propertyValue.isNull()) {
364       return TypeString.NULL.getTypeString();
365     } else if (propertyValue.isShort()) {
366       return TypeString.SHORT.getTypeString();
367     } else if (propertyValue.isInt()) {
368       return TypeString.INTEGER.getTypeString();
369     } else if (propertyValue.isLong()) {
370       return TypeString.LONG.getTypeString();
371     } else if (propertyValue.isFloat()) {
372       return TypeString.FLOAT.getTypeString();
373     } else if (propertyValue.isDouble()) {
374       return TypeString.DOUBLE.getTypeString();
375     } else if (propertyValue.isBoolean()) {
376       return TypeString.BOOLEAN.getTypeString();
377     } else if (propertyValue.isString()) {
378       return TypeString.STRING.getTypeString();
379     } else if (propertyValue.isBigDecimal()) {
380       return TypeString.BIGDECIMAL.getTypeString();
381     } else if (propertyValue.isGradoopId()) {
382       return TypeString.GRADOOPID.getTypeString();
383     } else if (propertyValue.isMap()) {
384       // map type string is map:{keyType}:{valueType}
385       return TypeString.MAP.getTypeString() +
386         PROPERTY_TOKEN_DELIMITER +
387         getTypeString(propertyValue.getMap().keySet().iterator().next()) +
388         PROPERTY_TOKEN_DELIMITER +
389         getTypeString(propertyValue.getMap().values().iterator().next());
390     } else if (propertyValue.isList()) {
391       // list type string is list:{itemType}
392       return TypeString.LIST.getTypeString() +
393         PROPERTY_TOKEN_DELIMITER +
394         getTypeString(propertyValue.getList().get(0));
395     } else if (propertyValue.isDate()) {
396       return TypeString.LOCALDATE.getTypeString();
397     } else if (propertyValue.isTime()) {
398       return TypeString.LOCALTIME.getTypeString();
399     } else if (propertyValue.isDateTime()) {
400       return TypeString.LOCALDATETIME.getTypeString();
401     } else if (propertyValue.isSet()) {
402       // set type string is set:{itemType}
403       return TypeString.SET.getTypeString() +
404         PROPERTY_TOKEN_DELIMITER +
405         getTypeString(propertyValue.getSet().iterator().next());
406     } else {
407       throw new IllegalArgumentException("Type " + propertyValue.getType() + " is not supported");
408     }
409   }
