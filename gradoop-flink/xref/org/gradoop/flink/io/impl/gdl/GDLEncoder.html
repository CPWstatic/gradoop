111   /**

118   public GDLEncoder(List<GraphHead> graphHeads, List<Vertex> vertices, List<Edge> edges) {

124   /**
125    * Creates a GDL formatted string from the graph heads, vertices and edges.
126    *
127    * @return GDL formatted string
128    */
129   public String getGDLString() {
130     Map<GradoopId, String> idToGraphHeadName = getGraphHeadNameMapping(graphHeads);
131     Map<GradoopId, String> idToVertexName = getVertexNameMapping(vertices);
132     Map<GradoopId, String> idToEdgeName = getEdgeNameMapping(edges);
133 
134     Set<GradoopId> usedVertexIds = new HashSet<>();
135     Set<GradoopId> usedEdgeIds = new HashSet<>();
136 
137     StringBuilder result = new StringBuilder();
138 
139     for (GraphHead gh : graphHeads) {
140       StringBuilder verticesString = new StringBuilder();
141       StringBuilder edgesString = new StringBuilder();
142 
143       for (Vertex v : vertices) {
144         boolean containedInGraph = v.getGraphIds().contains(gh.getId());
145         boolean firstOccurrence = !usedVertexIds.contains(v.getId());
146 
147         if (containedInGraph) {
148           String vertexString = vertexToGDLString(v, idToVertexName, firstOccurrence);
149           usedVertexIds.add(v.getId());
150           verticesString.append(vertexString).append(System.lineSeparator());
151         }
152       }
153 
154       for (Edge e : edges) {
155         if (e.getGraphIds().contains(gh.getId())) {
156           boolean firstOccurrence = !usedEdgeIds.contains(e.getId());
157           String edgeString = edgeToGDLString(e, idToVertexName, idToEdgeName, firstOccurrence);
158           usedEdgeIds.add(e.getId());
159           edgesString.append(edgeString).append(System.lineSeparator());
160         }
161       }
162 
163       result
164         .append(graphHeadToGDLString(gh, idToGraphHeadName))
165         .append(GRAPH_ELEMENTS_DEFINITION_START).append(System.lineSeparator())
166         .append(verticesString)
167         .append(edgesString.length() > 0 ? System.lineSeparator() : "")
168         .append(edgesString)
169         .append(GRAPH_ELEMENTS_DEFINITION_END)
170         .append(System.lineSeparator()).append(System.lineSeparator());
171     }
172     return result.toString();
173   }

175   /**
176    * Returns a mapping between the graph heads gradoop ids and the GDL variable names.
177    *
178    * @param graphHeads The graph heads.
179    * @return Mapping between graph head and GDL variable name.
180    */
181   private Map<GradoopId, String> getGraphHeadNameMapping(List<GraphHead> graphHeads) {
182     Map<GradoopId, String> idToGraphHeadName = new HashMap<>(graphHeads.size());
183     for (int i = 0; i < graphHeads.size(); i++) {
184       GraphHead g = graphHeads.get(i);
185       String gName = String.format("%s%s", GRAPH_VARIABLE_PREFIX, i);
186       idToGraphHeadName.put(g.getId(), gName);
187     }
188     return idToGraphHeadName;
189   }

191   /**
192    * Returns a mapping between the vertex GradoopID and the GDL variable name.
193    *
194    * @param vertices The graph vertices.
195    * @return Mapping between vertex and GDL variable name.
196    */
197   private Map<GradoopId, String> getVertexNameMapping(List<Vertex> vertices) {
198     Map<GradoopId, String> idToVertexName = new HashMap<>(vertices.size());
199     for (int i = 0; i < vertices.size(); i++) {
200       Vertex v = vertices.get(i);
201       String vName = String.format("%s_%s_%s", VERTEX_VARIABLE_PREFIX, v.getLabel(), i);
202       idToVertexName.put(v.getId(), vName);
203     }
204     return idToVertexName;
205   }

207   /**
208    * Returns a mapping between the edge GradoopId and the GDL variable name.
209    *
210    * @param edges The graph edges.
211    * @return Mapping between edge and GDL variable name.
212    */
213   private Map<GradoopId, String> getEdgeNameMapping(List<Edge> edges) {
214     Map<GradoopId, String> idToEdgeName = new HashMap<>(edges.size());
215     for (int i = 0; i < edges.size(); i++) {
216       Edge e = edges.get(i);
217       String eName = String.format("%s_%s_%s", EDGE_VARIABLE_PREFIX, e.getLabel(), i);
218       idToEdgeName.put(e.getId(), eName);
219     }
220     return idToEdgeName;
221   }

223   /**
224    * Returns a GDL formatted graph head string.
225    *
226    * @param g graph head
227    * @param idToGraphHeadName mapping from graph head id to its GDL variable name
228    * @return GDL formatted string
229    */
230   private String graphHeadToGDLString(GraphHead g, Map<GradoopId, String> idToGraphHeadName) {
231     return String.format("%s:%s %s",
232       idToGraphHeadName.get(g.getId()),
233       g.getLabel(),
234       propertiesToGDLString(g.getProperties()));
235   }

237   /**
238    * Returns the gdl formatted vertex including the properties and the label on first occurrence
239    * or otherwise just the variable name.
240    *
241    * @param vertex The vertex that should be formatted.
242    * @param idToVertexName Maps GradoopId of a vertex to a string that represents the gdl
243    *                       variable name
244    * @param firstOccurrence Is it the first occurrence of the vertex in all graphs?
245    * @return A GDL formatted vertex string.
246    */
247   private String vertexToGDLString(
248     Vertex vertex,
249     Map<GradoopId, String> idToVertexName,
250     boolean firstOccurrence) {
251     if (firstOccurrence) {
252       return String.format("(%s:%s %s)",
253         idToVertexName.get(vertex.getId()),
254         vertex.getLabel(),
255         propertiesToGDLString(vertex.getProperties()));
256     } else {
257       return String.format("(%s)", idToVertexName.get(vertex.getId()));
258     }
259   }
260 
261   /**
262    * Returns the GDL formatted edge, including the properties and the label on first occurrence
263    * or otherwise just the variable name.
264    *
265    * @param edge The edge to be formatted.
266    * @param idToVertexName Maps GradoopId of a vertex to a string that represents the GDL
267    *                       variable name
268    * @param idToEdgeName Maps GradoopId of an edge to a string that represents the GDL variable
269    *                     name.
270    * @param firstOccurrence Is it the first occurrence of the edge in all graphs?
271    * @return A GDL formatted edge string.
272    */
273   private String edgeToGDLString(
274     Edge edge,
275     Map<GradoopId, String> idToVertexName,
276     Map<GradoopId, String> idToEdgeName,
277     boolean firstOccurrence) {
278     String result;
279     if (firstOccurrence) {
280       result =  String.format("(%s)-[%s:%s%s]->(%s)",
281         idToVertexName.get(edge.getSourceId()),
282         idToEdgeName.get(edge.getId()),
283         edge.getLabel(),
284         propertiesToGDLString(edge.getProperties()),
285         idToVertexName.get(edge.getTargetId()));
286     } else {
287       result = String.format("(%s)-[%s]->(%s)",
288         idToVertexName.get(edge.getSourceId()),
289         idToEdgeName.get(edge.getId()),
290         idToVertexName.get(edge.getTargetId()));
291     }
292     return result;
293   }

295   /**
296    * Returns the properties as a GDL formatted String.
297    *
298    * @param properties The properties to be formatted.
299    * @return A GDL formatted string that represents the properties.
300    */
301   private String propertiesToGDLString(Properties properties) {
302     if (properties == null || properties.isEmpty()) {
303       return "";
304     } else {
305       return properties.toList().stream()
306         .map(this::propertyToGDLString)
307         .collect(Collectors.joining(PROPERTIES_SEPARATOR, PROPERTIES_PREFIX, PROPERTIES_SUFFIX));
308     }
309   }
310 
311   /**
312    * Returns this property as a GDL formatted String.
313    *
314    * @param property The property.
315    * @return A GDL formatted string that represents the property.
316    */
317   private String propertyToGDLString(Property property) {
318     StringBuilder result = new StringBuilder()
319       .append(property.getKey())
320       .append(KEY_VALUE_SEPARATOR);
321 
322     PropertyValue value = property.getValue();
323 
324     if (value.isString()) {
325       result.append(STRING_PREFIX).append(value.toString()).append(STRING_SUFFIX);
326     } else if (value.isNull()) {
327       result.append(NULL_STRING);
328     } else if (value.isDouble()) {
329       result.append(value.toString()).append(DOUBLE_SUFFIX);
330     } else if (value.isFloat()) {
331       result.append(value.toString()).append(FLOAT_SUFFIX);
332     } else if (value.isLong()) {
333       result.append(value.toString()).append(LONG_SUFFIX);
334     } else {
335       result.append(value.toString());
336     }
337 
338     return result.toString();
339   }
