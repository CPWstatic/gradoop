35  /**
36   * The program executes the following workflow:
37   *
38   * 1) Extract subgraph with:
39   *    - vertex predicate: must be of type 'Person'
40   *    - edge predicate: must be of type 'knows'
41   * 2) Transform vertices and edges to necessary information
42   * 3) Compute communities using Gelly label propagation
43   * 4) Compute vertex count per community
44   * 5) Select communities with a vertex count greater than a given threshold
45   * 6) Combine the remaining graphs to a single graph
46   * 7) Group the graph using the vertex attributes 'city' and 'gender' and
47   *    - count the number of vertices represented by each super vertex
48   *    - count the number of edges represented by each super edge
49   * 8) Aggregate the grouped graph:
50   *    - add the total vertex count as new graph property
51   *    - add the total edge count as new graph property
52   *
53   * The program can be either executed using external data (for benchmarking) or
54   * demo data ({@link #main(String[])}).
55   */
56  public class SNABenchmark2 extends AbstractRunner implements
57    ProgramDescription {

66     *

69     * 1) (possibly HDFS) input directory that contains a EPGM graph

153   /**
154    * The actual computation.
155    *
156    * @param socialNetwork social network graph
157    * @param threshold     used in community selection predicate
158    * @return summarized, aggregated graph
159    */
160   private static LogicalGraph
161   execute(LogicalGraph socialNetwork,
162     final int threshold) {
163 
164     final int maxIterations   = 4;
165     final String vertexCount  = "vertexCount";
166     final String edgeCount    = "edgeCount";
167     final String person       = "person";
168     final String knows        = "knows";
169     final String city         = "city";
170     final String gender       = "gender";
171     final String birthday     = "birthday";
172     final String label        = "label";
173 
174     return socialNetwork
175       // 1) extract subgraph
176       .subgraph(
177         vertex -> vertex.getLabel().toLowerCase().equals(person),
178         edge -> edge.getLabel().toLowerCase().equals(knows))
179       // project to necessary information
180       .transform(
181         // keep graph heads
182         TransformationFunction.keep(),
183         // keep necessary vertex properties
184         (current, transformed) -> {
185           transformed.setLabel(current.getLabel());
186           transformed.setProperty(city, current.getPropertyValue(city));
187           transformed.setProperty(gender, current.getPropertyValue(gender));
188           transformed.setProperty(label, current.getPropertyValue(birthday));
189           return transformed;
190         },
191         // keep only edge label
192         (current, transformed) -> {
193           transformed.setLabel(current.getLabel());
194           return transformed;
195         })
196       // 3a) compute communities
197       .callForGraph(new GellyLabelPropagation(maxIterations, label))
198       // 3b) separate communities
199       .splitBy(label)
200       // 4) compute vertex count per community
201       .apply(new ApplyAggregation(new VertexCount()))
202       // 5) select graphs with more than minClusterSize vertices
203       .select(g -> g.getPropertyValue(vertexCount).getLong() > threshold)
204       // 6) reduce filtered graphs to a single graph using combination
205       .reduce(new ReduceCombination())
206       // 7) group that graph by vertex properties
207       .groupBy(Lists.newArrayList(city, gender))
208       // 8) count vertices and edges of grouped graph
209       .aggregate(new VertexCount(), new EdgeCount());
210   }
