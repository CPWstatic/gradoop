122   /**
123    * Runs the simulation using the given arguments.
124    * <p>
125    * -i, --input-path (path to data graph)<br />
126    * -f, --input-format (format of data graph [csv, indexed, json])<br />
127    * -o, --output-path (path to output directory)<br />
128    * -q, --query (GDL/Cypher query string)<br />
129    * -a, --algorithm [dual-bulk,dual-delta,iso-exp,iso-exp-bc-hf,cypher]<br />
130    * -d, --attach-data (default: false)<br />
131    * -s, --statistics (path to input graph statistics, used for cypher engine)<br />
132    *
133    * @param args option line
134    * @throws Exception on failure
135    */
136   @SuppressWarnings("unchecked")
137   public static void main(String[] args) throws Exception {
138     CommandLine cmd = parseArguments(args,
139       PatternMatchingRunner.class.getName());
140     if (cmd == null) {
141       return;
142     }
143     performSanityCheck(cmd);
144 
145     String inputDir       = cmd.getOptionValue(OPTION_INPUT_PATH);
146     String inputFormat    = cmd.getOptionValue(OPTION_INPUT_FORMAT).toLowerCase();
147     String outputDir      = cmd.getOptionValue(OPTION_OUTPUT_PATH);
148     String query          = cmd.getOptionValue(OPTION_QUERY_GRAPH);
149     String algorithm      = cmd.getOptionValue(OPTION_ALGORITHM);
150     boolean attachData    = cmd.hasOption(OPTION_ATTACH_DATA);
151     boolean hasStatistics = cmd.hasOption(OPTION_STATISTICS);
152 
153     GraphStatistics statistics = null;
154     if (hasStatistics) {
155       String statisticsPath = cmd.getOptionValue(OPTION_STATISTICS);
156       if (statisticsPath.startsWith("hdfs://")) {
157         statistics = GraphStatisticsHDFSReader.read(statisticsPath, new Configuration());
158       } else {
159         statistics = GraphStatisticsLocalFSReader.read(statisticsPath);
160       }
161     }
162 
163     LogicalGraph epgmDatabase = readLogicalGraph(inputDir, inputFormat);
164 
165     GraphCollection result = execute(epgmDatabase, query, attachData, algorithm, statistics);
166 
167     writeGraphCollection(result, outputDir);
168 
169     System.out.println(String.format("Net runtime [s]: %d",
170       getExecutionEnvironment()
171         .getLastJobExecutionResult()
172         .getNetRuntime(TimeUnit.SECONDS)));
173   }

175   /**
176    * Checks if the minimum of arguments is provided
177    *
178    * @param cmd command line
179    */
180   private static void performSanityCheck(final CommandLine cmd) {
181     if (!cmd.hasOption(OPTION_INPUT_PATH)) {
182       throw new IllegalArgumentException("Define a graph input directory.");
183     }
184     if (!cmd.hasOption(OPTION_INPUT_FORMAT)) {
185       throw new IllegalArgumentException("Define an input format");
186     }
187     if (!cmd.hasOption(OPTION_OUTPUT_PATH)) {
188       throw new IllegalArgumentException("Define an graph output directory.");
189     }
190     if (!cmd.hasOption(OPTION_QUERY_GRAPH)) {
191       throw new IllegalArgumentException("Define a graph query.");
192     }
193     if (!cmd.hasOption(OPTION_ALGORITHM)) {
194       throw new IllegalArgumentException("Chose an algorithm.");
195     }
196     if (cmd.getOptionValue(OPTION_ALGORITHM).equals(ALGO_CYPHER)) {
197       if (!cmd.hasOption(OPTION_STATISTICS)) {
198         throw new IllegalArgumentException("Provide graph statistics when using Cypher engine");
199       }
200     }
201   }

203   /**
204    * Executes dual simulation on the given logical graph.
205    *
206    * @param databaseGraph data graph
207    * @param query         query graph
208    * @param attachData    attach vertex and edge data to the match graph
209    * @param algorithm     algorithm to use for pattern matching
210    * @param statistics    statistics about the input graph (needed for cypher)
211    * @return matching subgraphs
212    */
213   private static GraphCollection execute(
214     LogicalGraph databaseGraph,
215     String query, boolean attachData, String algorithm, GraphStatistics statistics) {
216 
217     PatternMatching op;
218 
219     switch (algorithm) {
220     case ALGO_DUAL_BULK:
221       op = new DualSimulation(query, attachData, true);
222       break;
223     case ALGO_DUAL_DELTA:
224       op = new DualSimulation(query, attachData, false);
225       break;
226     case ALGO_ISO_EXP:
227       op = new ExplorativePatternMatching.Builder()
228         .setQuery(query)
229         .setAttachData(attachData)
230         .setMatchStrategy(MatchStrategy.ISOMORPHISM)
231         .build();
232       break;
233     case ALGO_ISO_EXP_BC_HASH_FIRST:
234       op = new ExplorativePatternMatching.Builder()
235         .setQuery(query)
236         .setAttachData(attachData)
237         .setMatchStrategy(MatchStrategy.ISOMORPHISM)
238         .setTraverserStrategy(TraverserStrategy.SET_PAIR_BULK_ITERATION)
239         .setEdgeStepJoinStrategy(BROADCAST_HASH_FIRST)
240         .setVertexStepJoinStrategy(BROADCAST_HASH_FIRST)
241         .build();
242       break;
243     case ALGO_CYPHER:
244       op = new CypherPatternMatching(query, attachData,
245         MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM,
246         statistics);
247       break;
248     default:
249       throw new IllegalArgumentException(algorithm + " not supported");
250     }
251 
252     return op.execute(databaseGraph);
253   }
