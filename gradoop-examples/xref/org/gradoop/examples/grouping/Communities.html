16  package org.gradoop.examples.grouping;
17  
18  import org.gradoop.examples.AbstractRunner;
19  import org.gradoop.flink.algorithms.gelly.labelpropagation.GellyLabelPropagation;
20  import org.gradoop.flink.io.api.DataSink;
21  import org.gradoop.flink.io.api.DataSource;
22  import org.gradoop.flink.io.impl.csv.CSVDataSource;
23  import org.gradoop.flink.io.impl.dot.DOTDataSink;
24  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
25  import org.gradoop.flink.model.impl.operators.aggregation.functions.count.Count;
26  import org.gradoop.flink.model.impl.operators.grouping.GroupingStrategy;
27  import org.gradoop.flink.util.GradoopFlinkConfig;
28  
29  import static java.util.Collections.emptyList;
30  import static java.util.Collections.singletonList;
31  
32  /**
33   * Demo program that uses {@link GellyLabelPropagation} to compute communities in a social network
34   * and groups the vertices by their community identifier. The result is a summary graph where each
35   * vertex represents a community including its user count while each edge represents all friendships
36   * between communities.
37   */
38  public class Communities extends AbstractRunner {
39  
40    /**
41     * Loads a social network graph from the specified location, applies label propagation to extract
42     * communities and computes a summary graph using the community id. The resulting summary graph is
43     * written using the DOT format and also converted into an PNG image using GraphViz.
44     *
45     * args[0] - input path (CSV)
46     * args[1] - output path
47     *
48     * @param args arguments
49     * @throws Exception if something goes wrong
50     */
51    public static void main(String[] args) throws Exception {
52      String inputPath = args[0];
53      String outputPath = args[1];
54  
55      // instantiate a default gradoop config
56      GradoopFlinkConfig config = GradoopFlinkConfig.createConfig(getExecutionEnvironment());
57  
58      // define a data source to load the graph
59      DataSource dataSource = new CSVDataSource(inputPath, config);
60  
61      // property key used for label propagation
62      final String communityKey = "c_id";
63  
64      // load the graph and set initial community id
65      LogicalGraph graph = dataSource.getLogicalGraph();
66      graph = graph.transformVertices((current, transformed) -> {
67        current.setProperty(communityKey, current.getId());
68        return current;
69      });
70  
71      // apply label propagation to compute communities
72      graph = graph.callForGraph(new GellyLabelPropagation(10, communityKey));
73  
74      // group the vertices of the graph by their community, count vertices per group and edges
75      // between groups
76      LogicalGraph communities = graph.groupBy(singletonList(communityKey), // vertex grouping keys
77        singletonList(new Count("count")), // vertex aggregate functions
78        emptyList(), // edge grouping keys
79        singletonList(new Count("count")), // edge aggregate functions
80        GroupingStrategy.GROUP_REDUCE);
81  
82      // extract vertex induced subgraph only containing communities with more than 10 users
83      communities = communities
84        .vertexInducedSubgraph((vertex) -> vertex.getPropertyValue("count").getLong() > 10);
85  
86      // instantiate a data sink for the DOT format
87      DataSink dataSink = new DOTDataSink(outputPath, false);
88      dataSink.write(communities, true);
89  
90      // run the job
91      getExecutionEnvironment().execute();
92  
93      // convert to png
94      convertDotToPNG(outputPath, outputPath + ".png");
95    }
96  }
