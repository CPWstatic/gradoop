24  import org.gradoop.flink.model.impl.operators.aggregation.functions.count.Count;
25  import org.gradoop.flink.model.impl.operators.aggregation.functions.max.MaxVertexProperty;
26  import org.gradoop.flink.model.impl.operators.aggregation.functions.min.MinVertexProperty;
27  import org.gradoop.flink.model.impl.operators.grouping.GroupingStrategy;
28  import org.gradoop.flink.util.GradoopFlinkConfig;
29  
30  import java.time.LocalDate;
31  import java.util.Arrays;
32  import java.util.Collections;

34  /**
35   * Demo program that combines the grouping operator with
36   *
37   * (1) the subgraph operator extract vertices and edges by specified predicate functions and
38   * (2) the transformation operator to modify vertex properties which are used for grouping
39   */
40  public class Composition extends AbstractRunner {
41  
42    /**
43     * Loads a social network graph from the specified location, applies vertex and edge predicates
44     * and groups the resulting graph by vertex properties.
45     *
46     * args[0] - input path (CSV)
47     * args[1] - output path
48     *
49     * @param args arguments
50     * @throws Exception if something goes wrong
51     */
52    public static void main(String[] args) throws Exception {
53      String inputPath = args[0];
54      String outputPath = args[1];
55  
56      // instantiate a default gradoop config
57      GradoopFlinkConfig config = GradoopFlinkConfig.createConfig(getExecutionEnvironment());
58  
59      // define a data source to load the graph
60      DataSource dataSource = new CSVDataSource(inputPath, config);
61  
62      // load the graph
63      LogicalGraph socialNetwork = dataSource.getLogicalGraph();
64  
65      // use the subgraph operator to filter the graph
66      LogicalGraph subgraph =
67        socialNetwork.subgraph(v -> v.getLabel().equals("person"), e -> e.getLabel().equals("knows"));
68  
69      // use the transformation operator to classify the 'birthday' property for the users
70  
71      LogicalGraph transformed = subgraph.transformVertices((current, modified) -> {
72        LocalDate birthDayDate = current.getPropertyValue("birthday").getDate();
73        current.setProperty("yob", birthDayDate.getYear());
74        current.setProperty("decade", birthDayDate.getYear() - birthDayDate.getYear() % 10);
75        return current;
76      });
77  
78      // group the transformed graph by users decade and apply several aggregate functions
79      LogicalGraph summary = transformed.groupBy(
80        Collections.singletonList("decade"),
81        Arrays.asList(
82          new Count("count"),
83          new MinVertexProperty("yob", "min_yob"),
84          new MaxVertexProperty("yob", "max_yob")),
85        Collections.emptyList(),
86        Collections.singletonList(new Count("count")),
87        GroupingStrategy.GROUP_COMBINE);
88  
89      // use the decade as label information for the DOT sink
90      summary = summary.transformVertices((current, modified) -> {
91        current.setLabel(current.getPropertyValue("decade").toString());
92        return current;
93      });
94  
95      // instantiate a data sink for the DOT format
96      DataSink dataSink = new DOTDataSink(outputPath, false);
97      dataSink.write(summary, true);
98  
99      getExecutionEnvironment().execute();
100 
101     // convert DOT to PNG image
102     convertDotToPNG(outputPath, outputPath + ".png");
103   }
