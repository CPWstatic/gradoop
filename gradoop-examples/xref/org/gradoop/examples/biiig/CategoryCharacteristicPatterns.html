44  /**
45   * Example workflow of paper "Scalable Business Intelligence with Graph
46   * Collections" submitted to IT special issue on Big Data Analytics
47   *
48   * To execute the example:
49   * 1. checkout Gradoop
50   * 2. mvn clean install
51   * 3. run main method
52   */
53  public class CategoryCharacteristicPatterns implements ProgramDescription {
54  
55    /**
56     * main method
57     *
58     * @param args arguments (none required)
59     * @throws Exception on failure
60     */
61    public static void main(String[] args) throws Exception {
62  
63      ExampleOutput out = new ExampleOutput();
64  
65      LogicalGraph iig = getIntegratedInstanceGraph();
66  
67      out.add("Integrated Instance Graph", iig);
68  
69      // extract business transaction graphs (BTGs)
70      GraphCollection btgs = iig
71        .callForCollection(new BusinessTransactionGraphs());
72  
73      btgs = btgs
74        // determine closed/open BTGs
75        .apply(new ApplyAggregation(new IsClosedAggregateFunction()))
76        // select closed BTGs
77        .select(g -> g.getPropertyValue("isClosed").getBoolean())
78        // count number of sales orders per BTG
79        .apply(new ApplyAggregation(new CountSalesOrdersAggregateFunction()));
80  
81      out.add("Business Transaction Graphs with Measures", btgs);
82  
83      btgs = btgs.apply(new ApplyTransformation((graph, copy) -> {
84        // Transformation function to categorize graphs
85        String category = graph.getPropertyValue("soCount").getInt() > 0 ? "won" : "lost";
86        copy.setProperty(CATEGORY_KEY, PropertyValue.create(category));
87        return copy;
88      }, (vertex, copy) -> {
89        // Transformation function to relabel vertices and to drop properties
90        String superType = vertex.getPropertyValue(SUPERTYPE_KEY).toString();
91  
92        if (superType.equals(SUPERCLASS_VALUE_TRANSACTIONAL)) {
93          copy.setLabel(vertex.getLabel());
94        } else { // master data
95          copy.setLabel(vertex.getPropertyValue(SOURCEID_KEY).toString());
96        }
97        return copy;
98      }, (edge, copy) -> {
99        copy.setLabel(edge.getLabel());
100       return copy;
101     })
102     );
103 
104     out.add("Business Transaction Graphs after Transformation", btgs);
105 
106     FSMConfig fsmConfig = new FSMConfig(0.8f, true, 1, 3);
107 
108     GraphCollection patterns = btgs
109       .callForCollection(new CategoryCharacteristicSubgraphs(fsmConfig, 2.0f));
110 
111     out.add("Category characteristic graph patters", patterns);
112 
113     out.print();
114   }

116   /**
117    * Returns example integrated instance graph from GDL input.
118    *
119    * @return integrated instance graph
120    * @throws IOException on failure
121    */
122   public static LogicalGraph getIntegratedInstanceGraph() throws IOException {
123 
124     ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();
125 
126     GradoopFlinkConfig gradoopConf = GradoopFlinkConfig.createConfig(env);
127 
128     FlinkAsciiGraphLoader loader = new FlinkAsciiGraphLoader(gradoopConf);
129 
130     String gdl = IOUtils.toString(CategoryCharacteristicPatterns.class
131       .getResourceAsStream("/data/gdl/itbda.gdl"));
132 
133     gdl = gdl
134       .replaceAll("SOURCEID_KEY",
135         SOURCEID_KEY)
136       .replaceAll("SUPERTYPE_KEY",
137         SUPERTYPE_KEY)
138       .replaceAll("SUPERCLASS_VALUE_MASTER",
139         SUPERCLASS_VALUE_MASTER)
140       .replaceAll("SUPERCLASS_VALUE_TRANSACTIONAL",
141         SUPERCLASS_VALUE_TRANSACTIONAL);
142 
143     loader.initDatabaseFromString(gdl);
144 
145     return loader.getLogicalGraphByVariable("iig");
146   }

148   /**
149    * Aggregate function to determine "isClosed" measure.
150    */
151   private static class IsClosedAggregateFunction implements Or, VertexAggregateFunction {
152 
153     @Override
154     public String getAggregatePropertyKey() {
155       return "isClosed";
156     }
157 
158     @Override
159     public PropertyValue getIncrement(EPGMElement vertex) {
160       boolean isClosedQuotation =
161         vertex.getLabel().equals("Quotation") &&
162           !vertex.getPropertyValue("status").toString().equals("open");
163 
164       return PropertyValue.create(isClosedQuotation);
165     }
166   }
167 
168   /**
169    * Aggregate function to count sales orders per graph.
170    */
171   private static class CountSalesOrdersAggregateFunction extends VertexCount {
172 
173     /**
174      * Creates a new instance of a CountSalesOrderAggregateFunction aggregate function.
175      */
176     CountSalesOrdersAggregateFunction() {
177       super("soCount");
178     }
179 
180     @Override
181     public PropertyValue getIncrement(EPGMElement vertex) {
182       return PropertyValue.create(vertex.getLabel().equals("SalesOrder") ? 1 : 0);
183     }
184   }

188     return  CategoryCharacteristicPatterns.class.getName();
