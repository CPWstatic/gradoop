34  /**
35   * A self contained example on how to use the rollUp operator on Gradoop's LogicalGraph class.
36   *
37   * The example uses our own telephone call example graph.
38   */
39  public class RollUpExample extends AbstractRunner {
40  
41    /**
42     * Path to the example data graph
43     */
44    private static final String EXAMPLE_DATA_FILE =
45      RollUpExample.class.getResource("/data/gdl/telephonecalls.gdl").getFile();
46  
47    /**
48     * The example provides an overview over the usage of the different rollUp methods.
49     * It showcases the application of rollUp functions that are already provided by
50     * Gradoop. Documentation for all available rollUp functions can be found in the projects wiki.
51     *
52     * Using the telephone call graph in the resources directory, the program will:
53     * 1. transform the long value of the "time" property into the separate properties "year",
54     * "month", "day", "hour" and "minute"
55     * 2. run a vertex based rollUp using the vertex label as well as the "country", "state", "city"
56     * properties
57     * 3. run an edge based rollUp using the "year", "month", "day", "hour" and "minute" properties
58     * 4. write both results into a separate GDL-File
59     *
60     * args[0] - the base path for the generated png files
61     *
62     * @param args arguments
63     *
64     * @see <a href="https://github.com/dbs-leipzig/gradoop/wiki/Unary-Logical-Graph-Operators">
65     * Gradoop Wiki</a>
66     * @throws Exception on failure
67     */
68    public static void main(String[] args) throws Exception {
69      // init execution environment
70      ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();
71  
72      // create loader
73      FlinkAsciiGraphLoader loader = new FlinkAsciiGraphLoader(GradoopFlinkConfig.createConfig(env));
74  
75      // load data
76      loader.initDatabaseFromFile(
77        URLDecoder.decode(EXAMPLE_DATA_FILE, StandardCharsets.UTF_8.name()));
78  
79      // get LogicalGraph representation of the telephone call graph
80      LogicalGraph callGraph = loader.getLogicalGraph();
81  
82      // create year, month, day, hour and minute properties out of the time stamp
83      callGraph = callGraph.transformEdges(new TimePropertyTransformationFunction<>("time"));
84  
85      // path prefix for dot file
86      String pathPrefix = appendSeparator(args[0]);
87  
88      // group by rollUp on vertices
89      String firstDotPath = pathPrefix + "vertexRollUp.dot";
90      GraphCollection graphGroupedByRollUpOnVertices =
91        callGraph.groupVerticesByRollUp(
92          Arrays.asList(Grouping.LABEL_SYMBOL, "country", "state", "city"),
93          Collections.singletonList(new Count("count")),
94          Arrays.asList(Grouping.LABEL_SYMBOL, "year", "month"),
95          Collections.singletonList(new Count("count")));
96  
97      new DOTDataSink(firstDotPath, true).write(graphGroupedByRollUpOnVertices, true);
98  
99      // group by rollUp on edges
100     String secondDotPath = pathPrefix + "edgeRollUp.dot";
101     GraphCollection graphGroupedByRollUpOnEdges =
102       callGraph.groupEdgesByRollUp(
103         Arrays.asList(Grouping.LABEL_SYMBOL, "country", "state"),
104         Collections.singletonList(new Count("count")),
105         Arrays.asList(Grouping.LABEL_SYMBOL, "year", "month", "day", "hour", "minute"),
106         Collections.singletonList(new Count("count")));
107 
108     new DOTDataSink(secondDotPath, true).write(graphGroupedByRollUpOnEdges, true);
109 
110     env.execute();
111 
112     convertDotToPNG(firstDotPath, firstDotPath.replace("dot", "png"));
113     convertDotToPNG(secondDotPath, secondDotPath.replace("dot", "png"));
114   }
