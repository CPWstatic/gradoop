16  package org.gradoop.examples.aggregation;
17  
18  import java.net.URLDecoder;
19  import java.nio.charset.StandardCharsets;
20  
21  import org.apache.flink.api.java.ExecutionEnvironment;
22  import org.gradoop.examples.aggregation.functions.AddPropertyMeanAgeToGraphHead;
23  import org.gradoop.examples.aggregation.functions.AggregateListOfNames;
24  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
25  import org.gradoop.flink.model.impl.functions.epgm.ByLabel;
26  import org.gradoop.flink.model.impl.operators.aggregation.functions.count.VertexCount;
27  import org.gradoop.flink.model.impl.operators.aggregation.functions.sum.SumVertexProperty;
28  import org.gradoop.flink.util.FlinkAsciiGraphLoader;
29  import org.gradoop.flink.util.GradoopFlinkConfig;
30  
31  /**
32   * A self contained example on how to use the aggregate operator on Gradoop's LogicalGraph class.
33   *
34   * The example uses the graph in dev-support/social-network.pdf
35   */
36  public class AggregationExample {
37  
38    /**
39     * Path to the example data graph
40     */
41    private static final String EXAMPLE_DATA_FILE =
42      AggregationExample.class.getResource("/data/gdl/sna.gdl").getFile();
43    /**
44     * Property key 'birthday'
45     */
46    private static final String PROPERTY_KEY_BIRTHDAY = "birthday";
47    /**
48     * Property key 'person'
49     */
50    private static final String LABEL_PERSON = "Person";
51  
52    /**
53     * Runs the program on the example data graph.
54     *
55     * The example provides an overview over the usage of the aggregate() method.
56     * It both showcases the application of aggregation functions that are already provided by
57     * Gradoop, as well as the definition of custom ones.
58     * Documentation for all available aggregation functions as well as a detailed description of the
59     * aggregate method can be found in the projects wiki.
60     *
61     * Using the social network graph in the resources directory, the program will:
62     * 1. extract a subgraph only containing vertices which are labeled "person"
63     * 2. concatenate the values of the vertex property "name" of each vertex in the subgraph
64     * 3. count the amount of vertices in the subgraph (aka the amount of persons)
65     * 4. sum up the values of the vertex property "birthday"
66     * 5. add the property "meanAge" to the graph head using a graph head transformation
67     *
68     * @param args arguments
69     *
70     * @see <a href="https://github.com/dbs-leipzig/gradoop/wiki/Unary-Logical-Graph-Operators">
71     * Gradoop Wiki</a>
72     * @throws Exception on failure
73     */
74    public static void main(String[] args) throws Exception {
75      // init execution environment
76      ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();
77  
78      // create loader
79      FlinkAsciiGraphLoader loader = new FlinkAsciiGraphLoader(GradoopFlinkConfig.createConfig(env));
80  
81      // load data
82      loader.initDatabaseFromFile(
83        URLDecoder.decode(EXAMPLE_DATA_FILE, StandardCharsets.UTF_8.name()));
84  
85      // get LogicalGraph representation of the social network graph
86      LogicalGraph networkGraph = loader.getLogicalGraph();
87  
88      // execute aggregations and graph-head transformations
89      LogicalGraph result = networkGraph
90        // extract subgraph with edges and vertices which are of interest
91        .vertexInducedSubgraph(new ByLabel<>(LABEL_PERSON))
92        // apply custom VertexAggregateFunction
93        .aggregate(
94          new AggregateListOfNames(),
95          // aggregate sum of vertices in order to obtain total amount of persons
96          new VertexCount(),
97          // sum up values of the vertex property "birthday"
98          new SumVertexProperty(PROPERTY_KEY_BIRTHDAY))
99        // add computed property "meanAge" to the graph head
100       .transformGraphHead(new AddPropertyMeanAgeToGraphHead());
101 
102     // print graph, which now contains the newly aggregated properties in the graph head
103     result.print();
104   }
105 }
