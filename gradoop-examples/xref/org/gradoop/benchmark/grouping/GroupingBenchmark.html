204   /**
205    * Main program to run the benchmark. Arguments are the available options.
206    *
207    * @param args program arguments
208    * @throws Exception on failure
209    */
210   @SuppressWarnings("unchecked")
211   public static void main(String[] args) throws Exception {
212     CommandLine cmd = parseArguments(args,
213       GroupingBenchmark.class.getName());
214     if (cmd == null) {
215       return;
216     }
217     // test if minimum arguments are set
218     performSanityCheck(cmd);
219 
220     // read cmd arguments
221     readCMDArguments(cmd);
222 
223     // initialize EPGM database
224     LogicalGraph graphDatabase = readLogicalGraph(INPUT_PATH, INPUT_FORMAT);
225 
226     // initialize grouping keys
227     List<String> vertexKeys = Lists.newArrayList();
228     if (VERTEX_GROUPING_KEYS != null) {
229       vertexKeys = getKeys(VERTEX_GROUPING_KEYS);
230     }
231 
232     List<String> edgeKeys = Lists.newArrayList();
233     if (EDGE_GROUPING_KEYS != null) {
234       edgeKeys = getKeys(EDGE_GROUPING_KEYS);
235     }
236 
237     // initialize aggregators
238     List<AggregateFunction> vAggregators = Lists.newArrayList();
239     List<AggregateFunction> eAggregators = Lists.newArrayList();
240 
241     if (cmd.hasOption(OPTION_VERTEX_AGGREGATION_KEYS)) {
242       vAggregators =
243         getAggregators(VERTEX_AGGREGATORS, VERTEX_AGGREGATOR_KEYS,
244           VERTEX_AGGREGATOR_RESULT_KEYS);
245     }
246 
247     if (cmd.hasOption(OPTION_EDGE_AGGREGATION_KEYS)) {
248       eAggregators = getAggregators(EDGE_AGGREGATORS, EDGE_AGGREGATOR_KEYS,
249         EDGE_AGGREGATOR_RESULT_KEYS);
250     }
251     // build grouping operator
252     Grouping grouping = getOperator(STRATEGY,
253       vertexKeys, edgeKeys, USE_VERTEX_LABELS, USE_EDGE_LABELS, vAggregators,
254       eAggregators);
255 
256     // call grouping on whole database graph
257     LogicalGraph summarizedGraph = graphDatabase.callForGraph(grouping);
258     if (summarizedGraph != null) {
259       writeLogicalGraph(summarizedGraph, OUTPUT_PATH);
260       writeCSV();
261     } else {
262       System.err.println("wrong parameter constellation");
263     }
264   }

267   /**
268    * Checks if the minimum of arguments is provided
269    *
270    * @param cmd command line
271    */
272   private static void performSanityCheck(final CommandLine cmd) {
273     if (!cmd.hasOption(OPTION_INPUT_PATH)) {
274       throw new IllegalArgumentException("Define a graph input directory.");
275     }
276     if (!cmd.hasOption(OPTION_CSV_PATH)) {
277       throw new IllegalArgumentException("Path to CSV-File need to be set");
278     }
279     if (!cmd.hasOption(OPTION_VERTEX_GROUPING_KEY) &&
280       !cmd.hasOption(OPTION_USE_VERTEX_LABELS)) {
281       throw new IllegalArgumentException(
282         "Chose at least one vertex grouping key or use vertex labels.");
283     }
284     if (!cmd.hasOption(OPTION_VERTEX_AGGREGATION_FUNCS)) {
285       throw new IllegalArgumentException("Vertex aggregator need to be set! " +
286         "(max, min, count, none (or list of these)");
287     }
288     if (!cmd.hasOption(OPTION_EDGE_AGGREGATION_FUNCS)) {
289       throw new IllegalArgumentException("Edge aggregator need to be set! " +
290         "(max, min, count, none (or list of these)");
291     }
292

295   /**
296    * Reads the given arguments from command line
297    *
298    * @param cmd command line
299    */
300   private static void readCMDArguments(final CommandLine cmd) {
301     // read input output paths
302     INPUT_PATH = cmd.getOptionValue(OPTION_INPUT_PATH);
303     OUTPUT_PATH = cmd.getOptionValue(OPTION_OUTPUT_PATH);
304     CSV_PATH = cmd.getOptionValue(OPTION_CSV_PATH);
305 
306     // input format
307     INPUT_FORMAT = cmd.hasOption(OPTION_INPUT_FORMAT) ?
308       cmd.getOptionValue(OPTION_INPUT_FORMAT).toLowerCase() : DEFAULT_FORMAT;
309 
310     // initialize grouping strategy
311     if (cmd.hasOption(OPTION_GROUPING_STRATEGY)) {
312       String value = cmd.getOptionValue(OPTION_GROUPING_STRATEGY);
313       if (value.toUpperCase().equals("GC")) {
314         STRATEGY = GroupingStrategy.GROUP_COMBINE;
315       }
316     }
317 
318     // read if vertex or edge keys should be used
319     boolean useVertexKey = cmd.hasOption(OPTION_VERTEX_GROUPING_KEY);
320     VERTEX_GROUPING_KEYS =
321       useVertexKey ? cmd.getOptionValue(OPTION_VERTEX_GROUPING_KEY) : null;
322     boolean useEdgeKey = cmd.hasOption(OPTION_EDGE_GROUPING_KEY);
323     EDGE_GROUPING_KEYS =
324       useEdgeKey ? cmd.getOptionValue(OPTION_EDGE_GROUPING_KEY) : null;
325 
326     // read vertex and edge labels
327     USE_VERTEX_LABELS = cmd.hasOption(OPTION_USE_VERTEX_LABELS);
328     USE_EDGE_LABELS = cmd.hasOption(OPTION_USE_EDGE_LABELS);
329 
330     // read aggregators
331     VERTEX_AGGREGATORS = cmd.getOptionValue(OPTION_VERTEX_AGGREGATION_FUNCS);
332     boolean vertexAggKeys = cmd.hasOption(OPTION_VERTEX_AGGREGATION_KEYS);
333     if (vertexAggKeys) {
334       VERTEX_AGGREGATOR_KEYS =
335         cmd.getOptionValue(OPTION_VERTEX_AGGREGATION_KEYS);
336       VERTEX_AGGREGATOR_RESULT_KEYS =
337         cmd.getOptionValue(OPTION_VERTEX_AGGREGATION_RESULT_KEYS);
338     }
339 
340     EDGE_AGGREGATORS = cmd.getOptionValue(OPTION_EDGE_AGGREGATION_FUNCS);
341     boolean edgeAggKeys = cmd.hasOption(OPTION_EDGE_AGGREGATION_KEYS);
342     if (edgeAggKeys) {
343       EDGE_AGGREGATOR_KEYS = cmd.getOptionValue(OPTION_EDGE_AGGREGATION_KEYS);
344       EDGE_AGGREGATOR_RESULT_KEYS =
345         cmd.getOptionValue(OPTION_EDGE_AGGREGATION_RESULT_KEYS);
346     }
347   }

360   /**
361    * Method to build aggregators
362    *
363    * @param aggs        aggregators as whole string
364    * @param keys        aggregator keys as whole string
365    * @param resultKeys  aggregator result keys as whole string
366    * @return List of PropertyValueAggregators
367    */
368   private static List<AggregateFunction> getAggregators(String
369     aggs, String keys, String resultKeys) {
370 
371     List<AggregateFunction> aggregatorList = Lists.newArrayList();
372 
373     aggs = aggs.replace("\\s", "");
374     keys = keys.replace("\\s", "");
375     resultKeys = resultKeys.replace("\\s", "");
376 
377     List<String> aggsList = Arrays.asList(TOKEN_SEPARATOR.split(aggs));
378     List<String> keyList = Arrays.asList(TOKEN_SEPARATOR.split(keys));
379     List<String> resultKeyList = Arrays.asList(TOKEN_SEPARATOR.split
380       (resultKeys));
381 
382     if (!aggs.equals("none")) {
383       for (int i = 0; i < aggsList.size(); i++) {
384         switch (aggsList.get(i)) {
385         case "count" :
386           aggregatorList.add(new Count());
387           break;
388         case "max" :
389           aggregatorList.add(new MaxProperty(keyList.get(i),
390             resultKeyList.get(i)));
391           break;
392         case "min" :
393           aggregatorList.add(new MinProperty(keyList.get(i),
394             resultKeyList.get(i)));
395           break;
396         default:
397           aggregatorList.add(null);
398           break;
399         }
400       }
401     }
402     return aggregatorList;
403   }

405   /**
406    * Returns the grouping operator implementation based on the given strategy.
407    *
408    * @param strategy        grouping strategy to use
409    * @param vertexKeys      vertex property keys used for grouping
410    * @param edgeKeys        edge property keys used for grouping
411    * @param useVertexLabels use vertex label for grouping, true/false
412    * @param useEdgeLabels   use edge label for grouping, true/false
413    * @param vAggs           used vertex aggregators
414    * @param eAggs           used edge aggregators
415    * @return grouping operator implementation
416    */
417   private static Grouping getOperator(GroupingStrategy strategy,
418     List<String> vertexKeys, List<String> edgeKeys,
419     boolean useVertexLabels, boolean useEdgeLabels,
420     List<AggregateFunction> vAggs, List<AggregateFunction> eAggs) {
421 
422     Grouping.GroupingBuilder builder =
423       new Grouping.GroupingBuilder()
424         .setStrategy(strategy)
425         .useVertexLabel(useVertexLabels)
426         .useEdgeLabel(useEdgeLabels);
427 
428     if (vAggs.size() > 0) {
429       for (AggregateFunction agg:vAggs) {
430         if (agg != null) {
431           builder.addVertexAggregateFunction(agg);
432         }
433       }
434     }
435 
436     if (eAggs.size() > 0) {
437       for (AggregateFunction agg: eAggs) {
438         if (agg != null) {
439           builder.addEdgeAggregateFunction(agg);
440         }
441       }
442     }
443 
444     if (vertexKeys.size() > 0) {
445       for (String vKey : vertexKeys) {
446         builder.addVertexGroupingKey(vKey);
447       }
448     }
449 
450     if (edgeKeys.size() > 0) {
451       for (String eKey : edgeKeys) {
452         builder.addEdgeGroupingKey(eKey);
453       }
454     }
455     return builder.build();
456   }

459   /**
460    * Method to create and add lines to a csv-file
461    * @throws IOException on failure
462    */
463   private static void writeCSV() throws IOException {
464 
465     String head = String.format("%s|%s|%s|%s|%s|%s|%s|%s|%s|%s|%s%n",
466       "Parallelism", "dataset", "vertexKeys", "edgeKeys", "USE_VERTEX_LABELS",
467       "USE_EDGE_LABELS", "Vertex Aggregators", "Vertex-Aggregator-Keys",
468       "EPGMEdge-Aggregators", "EPGMEdge-Aggregator-Keys", "Runtime(s)");
469 
470     String tail = String.format("%s|%s|%s|%s|%s|%s|%s|%s|%s|%s|%s%n",
471       getExecutionEnvironment().getParallelism(), INPUT_PATH,
472       VERTEX_GROUPING_KEYS, EDGE_GROUPING_KEYS, USE_VERTEX_LABELS,
473       USE_EDGE_LABELS, VERTEX_AGGREGATORS, VERTEX_AGGREGATOR_KEYS,
474       EDGE_AGGREGATORS, EDGE_AGGREGATOR_KEYS,
475       getExecutionEnvironment().getLastJobExecutionResult()
476         .getNetRuntime(TimeUnit.SECONDS));
477 
478     File f = new File(CSV_PATH);
479     if (f.exists() && !f.isDirectory()) {
480       FileUtils.writeStringToFile(f, tail, true);
481     } else {
482       PrintWriter writer = new PrintWriter(CSV_PATH, "UTF-8");
483       writer.print(head);
484       writer.print(tail);
485       writer.close();
486     }
487   }
