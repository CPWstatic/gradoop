38  /**
39   * Used to benchmark {@link TransactionalPatternMatching} implementation.
40   * <p>
41   * The benchmarks expects the graph to be stored in two different directories:
42   * <p>
43   * [inputDir]/vertices -> [unique vertex-id]
44   * [inputDir]/edges    -> [unique edge-id,source-vertex-id,target-vertex-id]
45   * <p>
46   * All identifiers must be of type {@link Long}.
47   */
48  public class TransactionalBenchmark extends AbstractRunner {
49    /**
50     * Option to declare path to input graph
51     */
52    private static final String OPTION_INPUT_PATH = "i";
53    /**
54     * Option to declare path to input graph
55     */
56    private static final String OPTION_QUERY = "q";
57    /**
58     * Option to set the traverser
59     */
60    private static final String OPTION_RETURN_EMBEDDINGS = "e";
61  
62    static {
63      OPTIONS.addOption(OPTION_INPUT_PATH, "input", true, "Graph directory");
64      OPTIONS.addOption(OPTION_QUERY, "query", true, "Pattern or fixed query");
65      OPTIONS.addOption(OPTION_RETURN_EMBEDDINGS, "embeddings", false,
66        "if embeddings should be returned");
67    }
68  
69    /**
70     * Given a query pattern, the benchmark computes the number of matching
71     * subgraphs in the given data graph.
72     * <p>
73     * This benchmark currently supports structure only pattern. For semantic
74     * patterns use {@link PatternMatching}.
75     * <p>
76     * usage: org.gradoop.benchmark.patternmatching.TraverserBenchmark
77     * [-i <arg>] [-q <arg>] [-t <arg>]
78     * -i,--input <arg>       Graph directory
79     * -q,--query <arg>       Pattern or fixed query (e.g. q2 or "(a)-->(b)")
80     * -t,--traverser <arg>   [loop|bulk]
81     *
82     * @param args program arguments
83     * @throws Exception on failure
84     */
85    public static void main(String[] args) throws Exception {
86      CommandLine cmd =
87        parseArguments(args, TransactionalBenchmark.class.getName());
88      if (cmd == null) {
89        return;
90      }
91      performSanityCheck(cmd);
92  
93      String inputPath = cmd.getOptionValue(OPTION_INPUT_PATH);
94      String queryString = "query[" + cmd.getOptionValue(OPTION_QUERY) + "]";
95      boolean returnEmbeddings = cmd.hasOption(OPTION_RETURN_EMBEDDINGS);
96  
97      ExecutionEnvironment env = getExecutionEnvironment();
98  
99      TLFDataSource source =
100       new TLFDataSource(inputPath, GradoopFlinkConfig.createConfig(env));
101 
102     DataSet<GraphWithCandidates> graphs =
103       source.getGraphCollection().getGraphTransactions()
104         .map(new GraphTransactionMatcher(queryString));
105 
106     if (returnEmbeddings) {
107       DataSet<Tuple4<GradoopId, GradoopId, GradoopIdSet, GradoopIdSet>> embeddings =
108         graphs.flatMap(
109           new FindEmbeddings(new DepthSearchMatching(), queryString));
110 
111       long embeddingCount = embeddings.count();
112       System.out.println("embeddingCount = " + embeddingCount);
113 
114     } else {
115       DataSet<Tuple2<GradoopId, Boolean>> containment =
116         graphs.map(new HasEmbeddings(
117           new DepthSearchMatching(), queryString))
118           .filter(new SecondFieldTrue<>());
119 
120       long containmentCount = containment.count();
121       System.out.println(
122         "containmentCount = " + containmentCount);
123     }
124 
125     System.out.println(String.format("Net runtime [s]: %d",
126       env
127         .getLastJobExecutionResult()
128         .getNetRuntime(TimeUnit.SECONDS)));
129   }
130 
131   /**
132    * Checks if the minimum of arguments is provided
133    *
134    * @param cmd command line
135    */
136   private static void performSanityCheck(final CommandLine cmd) {
137     if (!cmd.hasOption(OPTION_INPUT_PATH)) {
138       throw new IllegalArgumentException("Define a graph input directory.");
139     }
140     if (!cmd.hasOption(OPTION_QUERY)) {
141       throw new IllegalArgumentException("Define an graph output directory.");
142     }
143   }
144 
145   /**
146    * Filter that checks if the second field is True.
147    * @param <T> any type
148    */
149   private static class SecondFieldTrue<T> implements
150     FilterFunction<Tuple2<T, Boolean>> {
151     @Override
152     public boolean filter(Tuple2<T, Boolean> tuple2) {
153       return tuple2.f1;
154     }
155   }
