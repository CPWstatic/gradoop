62        "[set-pair-for|set-pair-bulk|triple-for]");
63      OPTIONS.addOption(OPTION_CSV_PATH, "csv-path", true, "Path to output CSV file");

100   /**
101    * Constructor
102    *
103    * @param cmd commandline
104    */
105   TraverserBenchmark(CommandLine cmd) {
106     this.inputPath = cmd.getOptionValue(OPTION_INPUT_PATH);
107     this.query = cmd.getOptionValue(OPTION_QUERY);
108 
109     String traverserStrategyString = cmd.getOptionValue(OPTION_TRAVERSER).toLowerCase();
110 
111     switch (traverserStrategyString) {
112     case "set-pair-bulk":
113       this.traverserStrategy = TraverserStrategy.SET_PAIR_BULK_ITERATION;
114       break;
115     case "set-pair-for":
116       this.traverserStrategy = TraverserStrategy.SET_PAIR_FOR_LOOP_ITERATION;
117       break;
118     case "triple-for":
119       this.traverserStrategy = TraverserStrategy.TRIPLES_FOR_LOOP_ITERATION;
120       break;
121     default:
122       throw new IllegalArgumentException("Unknown traverser strategy: " + traverserStrategyString);
123     }
124 
125     if (cmd.hasOption(OPTION_CSV_PATH)) {
126       csvPath = cmd.getOptionValue(OPTION_CSV_PATH);
127     }
128     initialize();
129   }
130 
131   /**
132    * Initialize the benchmark using a given query. The query can be a predefined one (e.g. q0) or
133    * a GDL pattern (e.g. (a)-->(b)).
134    */
135   private void initialize() {
136     if (query.toLowerCase().startsWith("q")) {
137       // fixed query
138       Queries.Query q = getQuery(query.toLowerCase());
139       vertexCount = q.getVertexCount();
140       edgeCount = q.getEdgeCount();
141       tc = q.getTraversalCode();
142     } else {
143       // GDL query pattern
144       QueryHandler queryHandler = new QueryHandler(query);
145       vertexCount = queryHandler.getVertexCount();
146       edgeCount = queryHandler.getEdgeCount();
147       Traverser traverser = new DFSTraverser();
148       traverser.setQueryHandler(queryHandler);
149       tc = traverser.traverse();
150     }
151   }

153   /**
154    * Returns a string containing information about the benchmark run.
155    *
156    * @return benchmark result string
157    */
158   private String getResultString() {
159     return String.format("%s|%s|%s|%s|%s|%s",
160       inputPath,
161       getExecutionEnvironment().getParallelism(),
162       traverserStrategy.name(),
163       query,
164       embeddingCount,
165       getExecutionEnvironment().getLastJobExecutionResult().getNetRuntime());
166   }

168   /**
169    * Writes the results of the benchmark into the given csv file. If the file already exists,
170    * the results are appended.
171    *
172    * @param csvFile path to csv file
173    * @throws IOException on failure
174    */
175   private void writeResults(String csvFile) throws IOException {
176     String header = "Input|Parallelism|Strategy|Query|Embeddings|Runtime[ms]";
177     String line = getResultString();
178 
179     File f = new File(csvFile);
180     if (f.exists() && !f.isDirectory()) {
181       FileUtils.writeStringToFile(f, String.format("%s%n", line), true);
182     } else {
183       PrintWriter writer = new PrintWriter(csvFile, "UTF-8");
184       writer.println(header);
185       writer.println(line);
186       writer.close();
187     }
188   }
189

195   }

287   /**
288    * Initializes {@link IdWithCandidates}
289    */
290   @FunctionAnnotation.ForwardedFields("f0")
291   public static class GetIdWithCandidates
292     implements MapFunction<Tuple1<Long>, IdWithCandidates<Long>> {
293     /**
294      * Reduce object instantiations
295      */
296     private final IdWithCandidates<Long> reuseTuple;
297 
298     /**
299      * Constructor
300      *
301      * @param vertexCount number of query vertices
302      */
303     GetIdWithCandidates(int vertexCount) {
304       reuseTuple = new IdWithCandidates<>();
305       boolean[] candidates = new boolean[vertexCount];
306       for (int i = 0; i < vertexCount; i++) {
307         candidates[i] = true;
308       }
309       reuseTuple.setCandidates(candidates);
310     }
311 
312     @Override
313     public IdWithCandidates<Long> map(Tuple1<Long> value) {
314       reuseTuple.setId(value.f0);
315       return reuseTuple;
316     }
317   }
