117   /**
118    * Main program to run the benchmark. Required arguments are a path to CSVDataSource compatible
119    * files that define a graph, an integer defining the sampling algorithm to be tested and a list
120    * of parameters for the constructor of the sampling class.
121    * Other arguments are the available options.
122    *
123    * @param args program arguments
124    * @throws Exception on failure
125    */
126   public static void main(String[] args) throws Exception {
127     CommandLine cmd = parseArguments(args, SamplingBenchmark.class.getName());
128 
129     if (cmd == null) {
130       System.exit(1);
131     }
132 
133     readCMDArguments(cmd);
134 
135     LogicalGraph graph = readLogicalGraph(INPUT_PATH, INPUT_FORMAT);
136 
137     // instantiate selected sampling algorithm and create sample
138     SamplingAlgorithm algorithm = SamplingBuilder.buildSelectedSamplingAlgorithm(
139       SELECTED_ALGORITHM, CONSTRUCTOR_PARAMS);
140     LogicalGraph graphSample = algorithm.execute(graph);
141 
142     // write graph sample and benchmark data
143     writeLogicalGraph(graphSample, OUTPUT_PATH + OUTPUT_PATH_GRAPH_SAMPLE_SUFFIX);
144     writeBenchmark(graphSample.getConfig().getExecutionEnvironment(), algorithm);
145   }
146 
147   /**
148    * Reads the given arguments from command line.
149    *
150    * @param cmd command line
151    */
152   private static void readCMDArguments(CommandLine cmd) {
153     INPUT_PATH          = cmd.getOptionValue(OPTION_INPUT_PATH);
154     SELECTED_ALGORITHM  = Integer.parseInt(cmd.getOptionValue(OPTION_SELECTED_ALGORITHM));
155     CONSTRUCTOR_PARAMS  = cmd.getOptionValues(OPTION_CONSTRUCTOR_PARAMS);
156     OUTPUT_PATH         = cmd.getOptionValue(OPTION_OUTPUT_PATH, OUTPUT_PATH_DEFAULT);
157     INPUT_FORMAT        = cmd.getOptionValue(OPTION_INPUT_FORMAT, INPUT_FORMAT_DEFAULT);
158   }
159 
160   /**
161    * Method to crate and add lines to a benchmark file.
162    *
163    * @param env given ExecutionEnvironment
164    * @param sampling sampling algorithm under test
165    * @throws IOException exception during file writing
166    */
167   private static void writeBenchmark(ExecutionEnvironment env, SamplingAlgorithm sampling)
168       throws IOException {
169     String head = String.format("%s|%s|%s|%s|%s%n",
170       "Parallelism", "Dataset", "Algorithm", "Params", "Runtime [s]");
171 
172     // build log
173     String samplingName = sampling.getClass().getSimpleName();
174     String tail = String.format("%s|%s|%s|%s|%s%n",
175       env.getParallelism(),
176       INPUT_PATH.substring(INPUT_PATH.lastIndexOf(File.separator) + 1),
177       samplingName,
178       String.join(", ", CONSTRUCTOR_PARAMS),
179       env.getLastJobExecutionResult().getNetRuntime(TimeUnit.SECONDS));
180 
181     File f = new File(OUTPUT_PATH + OUTPUT_PATH_BENCHMARK_SUFFIX);
182     if (f.exists() && !f.isDirectory()) {
183       FileUtils.writeStringToFile(f, tail, true);
184     } else {
185       PrintWriter writer = new PrintWriter(OUTPUT_PATH + OUTPUT_PATH_BENCHMARK_SUFFIX, "UTF-8");
186       writer.print(head);
187       writer.print(tail);
188       writer.close();
189     }
190   }
