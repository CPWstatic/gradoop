44  /**
45   * A dedicated program to evaluate a more complex social network analytics example.
46   */
47  public class SocialNetworkAnalyticsExample extends AbstractRunner implements ProgramDescription {

69     * Used output path

71    private static String OUTPUT_PATH;

83        "Path to source files.");
84      OPTIONS.addOption(OPTION_OUTPUT_PATH, "output", true,
85        "Path to output file");
86      OPTIONS.addOption(OPTION_CSV_PATH, "csv", true,

89        "Path to graph statistics");

92    /**
93     * Main program to run the benchmark. Arguments are the available options.
94     *
95     * @param args program arguments
96     * @throws Exception IO or execution Exception
97     */
98    public static void main(String[] args) throws Exception {
99      CommandLine cmd = parseArguments(args, SocialNetworkAnalyticsExample.class.getName());
100 
101     if (cmd == null) {
102       System.exit(1);
103     }
104 
105     // test if minimum arguments are set
106     performSanityCheck(cmd);
107 
108     // read cmd arguments
109     readCMDArguments(cmd);
110 
111     ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();
112     GradoopFlinkConfig conf = GradoopFlinkConfig.createConfig(env);
113 
114     // read logical graph
115     DataSource source = new IndexedCSVDataSource(INPUT_PATH, conf);
116     LogicalGraph graph = source.getLogicalGraph();
117 
118     // create subgraph containing vertices with label: person, tag, country, city and post
119     // and its incident edges
120     graph = graph.vertexInducedSubgraph(new LabelIsIn<>(
121       "person", "tag", "country", "city", "post"));
122 
123     // call cypher function (with or without statistics)
124     if (cmd.hasOption(OPTION_STATISTICS_PATH)) {
125       GraphStatistics statistics = GraphStatisticsHDFSReader
126         .read(STATISTICS_PATH, new Configuration());
127 
128       graph = graph
129         .query(getQuery(), getConstruction(), statistics)
130         .reduce(new ReduceCombination());
131 
132     } else {
133       graph = graph
134         .query(getQuery(), getConstruction())
135         .reduce(new ReduceCombination());
136     }
137 
138     // group on vertex and edge labels + count grouped edges
139     LogicalGraph groupedGraph  = new Grouping.GroupingBuilder()
140       .setStrategy(GroupingStrategy.GROUP_COMBINE)
141       .addVertexGroupingKey("name")
142       .useEdgeLabel(true).useVertexLabel(true)
143       .addEdgeAggregateFunction(new Count())
144       .build().execute(graph);
145 
146     // filter all edges below a fixed threshold
147     groupedGraph = groupedGraph.edgeInducedSubgraph(new CountFilter());
148 
149     // write data to sink
150     DataSink sink = new CSVDataSink(OUTPUT_PATH, conf);
151     sink.write(groupedGraph);
152 
153     // execute and write job statistics
154     env.execute();
155     writeCSV(env);
156   }

158   /**
159    * Reads the given arguments from command line
160    *
161    * @param cmd command line
162    */
163   private static void readCMDArguments(CommandLine cmd) {
164     INPUT_PATH      = cmd.getOptionValue(OPTION_INPUT_PATH);
165     OUTPUT_PATH     = cmd.getOptionValue(OPTION_OUTPUT_PATH);
166     CSV_PATH        = cmd.getOptionValue(OPTION_CSV_PATH);
167     STATISTICS_PATH = cmd.getOptionValue(OPTION_STATISTICS_PATH);
168   }
169 
170   /**
171    * Returns used cypher query
172    *
173    * @return used cypher query
174    */
175   private static String getQuery() {
176     return
177       "MATCH (p1:person)<-[:hasCreator]-(po:post)<-[:likes]-(p2:person)\n" +
178       "(p1)-[:isLocatedIn]->(c1:city)\n" +
179       "(p2)-[:isLocatedIn]->(c2:city)" +
180       "(po)-[:hasTag]->(t:tag)\n" +
181       "(c1)-[:isPartOf]->(ca:country)<-[:isPartOf]-(c2)\n" +
182       "WHERE p1 != p2";
183   }
184 
185   /**
186    * Returns used construction pattern
187    *
188    * @return used construction pattern
189    */
190   private static String getConstruction() {
191     return "(ca)-[new:hasInterest]->(t)";
192   }

194   /**
195    * Checks if the minimum of arguments is provided
196    *
197    * @param cmd command line
198    */
199   private static void performSanityCheck(CommandLine cmd) {
200     if (!cmd.hasOption(OPTION_INPUT_PATH)) {
201       throw new IllegalArgumentException("Define a graph input directory.");
202     }
203     if (!cmd.hasOption(OPTION_CSV_PATH)) {
204       throw new IllegalArgumentException("Path to CSV-File need to be set.");
205     }
206     if (!cmd.hasOption(OPTION_OUTPUT_PATH)) {
207       throw new IllegalArgumentException("Define a graph output directory.");
208     }
209   }

211   /**
212    * Method to create and add lines to a csv-file
213    *
214    * @param env given ExecutionEnvironment
215    * @throws IOException exception during file writing
216    */
217   private static void writeCSV(ExecutionEnvironment env) throws IOException {
218 
219     String head = String.format("%s|%s|%s%n",
220       "Parallelism",
221       "dataset",
222       "Runtime(s)");
223 
224     String tail = String.format("%s|%s|%s%n",
225       env.getParallelism(),
226       INPUT_PATH,
227       env.getLastJobExecutionResult().getNetRuntime(TimeUnit.SECONDS));
228 
229     File f = new File(CSV_PATH);
230     if (f.exists() && !f.isDirectory()) {
231       FileUtils.writeStringToFile(f, tail, true);
232     } else {
233       PrintWriter writer = new PrintWriter(CSV_PATH, "UTF-8");
234       writer.print(head);
235       writer.print(tail);
236       writer.close();
237     }
238   }
