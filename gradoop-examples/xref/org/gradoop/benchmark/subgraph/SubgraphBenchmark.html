37  /**
38   * A dedicated program for parametrized subgraph benchmark.
39   */
40  public class SubgraphBenchmark extends AbstractRunner implements ProgramDescription {
41    /**
42     * Option to declare path to indexed input graph
43     */
44    private static final String OPTION_INPUT_PATH = "i";
45    /**
46     * Option to declare path to output graph
47     */
48    private static final String OPTION_OUTPUT_PATH = "o";
49    /**
50     * Option to declare output path to statistics csv file
51     */
52    private static final String OPTION_CSV_PATH = "c";
53    /**
54     * Option to declare verification
55     */
56    private static final String OPTION_VERIFICATION = "v";
57    /**
58     * Option to declare used vertex label
59     */
60    private static final String OPTION_VERTEX_LABEL = "vl";
61    /**
62     * Option to declare used edge label
63     */
64    private static final String OPTION_EDGE_LABEL = "el";
65    /**
66     * Used input path
67     */
68    private static String INPUT_PATH;
69    /**
70     * Used output path
71     */
72    private static String OUTPUT_PATH;
73    /**
74     * Used csv path
75     */
76    private static String CSV_PATH;
77    /**
78     * Used vertex label
79     */
80    private static String VERTEX_LABEL;
81    /**
82     * Used edge label
83     */
84    private static String EDGE_LABEL;
85    /**
86     * Used verification flag
87     */
88    private static boolean VERIFICATION;
89  
90    static {
91      OPTIONS.addOption(OPTION_INPUT_PATH, "input", true,
92        "Path to indexed source files.");
93      OPTIONS.addOption(OPTION_OUTPUT_PATH, "output", true,
94        "Path to output file");
95      OPTIONS.addOption(OPTION_CSV_PATH, "csv", true,
96        "Path to csv statistics");
97      OPTIONS.addOption(OPTION_VERIFICATION, "verification", false,
98        "Verify Subgraph with join.");
99      OPTIONS.addOption(OPTION_VERTEX_LABEL, "vertex-label", true,
100       "Used vertex label");
101     OPTIONS.addOption(OPTION_EDGE_LABEL, "edge-label", true,
102       "Used edge label");
103   }

105   /**
106    * Main program to run the benchmark. Arguments are the available options.
107    *
108    * @param args program arguments
109    * @throws Exception in case of Error
110    */
111   public static void main(String[] args) throws Exception {
112     CommandLine cmd = parseArguments(args, SubgraphBenchmark.class.getName());
113 
114     if (cmd == null) {
115       System.exit(1);
116     }
117 
118     // test if minimum arguments are set
119     performSanityCheck(cmd);
120 
121     // read cmd arguments
122     readCMDArguments(cmd);
123 
124     // create gradoop config
125     ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();
126     GradoopFlinkConfig conf = GradoopFlinkConfig.createConfig(env);
127 
128     // read graph
129     DataSource source = new IndexedCSVDataSource(INPUT_PATH, conf);
130     LogicalGraph graph = source.getLogicalGraph();
131 
132     // compute subgraph -> verify results (join) vs no verify (filter)
133     if (VERIFICATION) {
134       graph = graph.subgraph(new ByLabel<>(VERTEX_LABEL), new ByLabel<>(EDGE_LABEL),
135         Subgraph.Strategy.BOTH_VERIFIED);
136     } else {
137       graph = graph.subgraph(new ByLabel<>(VERTEX_LABEL), new ByLabel<>(EDGE_LABEL),
138         Subgraph.Strategy.BOTH);
139     }
140 
141     // write graph
142     DataSink sink = new CSVDataSink(OUTPUT_PATH, conf);
143     sink.write(graph);
144 
145     // execute and write job statistics
146     env.execute();
147     writeCSV(env);
148   }
149 
150   /**
151    * Reads the given arguments from command line
152    *
153    * @param cmd command line
154    */
155   private static void readCMDArguments(CommandLine cmd) {
156     INPUT_PATH   = cmd.getOptionValue(OPTION_INPUT_PATH);
157     OUTPUT_PATH  = cmd.getOptionValue(OPTION_OUTPUT_PATH);
158     CSV_PATH     = cmd.getOptionValue(OPTION_CSV_PATH);
159     VERTEX_LABEL = cmd.getOptionValue(OPTION_VERTEX_LABEL);
160     EDGE_LABEL   = cmd.getOptionValue(OPTION_EDGE_LABEL);
161     VERIFICATION = cmd.hasOption(OPTION_VERIFICATION);
162   }
163 
164   /**
165    * Checks if the minimum of arguments is provided
166    *
167    * @param cmd command line
168    */
169   private static void performSanityCheck(CommandLine cmd) {
170     if (!cmd.hasOption(OPTION_INPUT_PATH)) {
171       throw new IllegalArgumentException("Define a graph input directory.");
172     }
173     if (!cmd.hasOption(OPTION_CSV_PATH)) {
174       throw new IllegalArgumentException("Path to CSV-File need to be set.");
175     }
176     if (!cmd.hasOption(OPTION_OUTPUT_PATH)) {
177       throw new IllegalArgumentException("Define a graph output directory.");
178     }
179   }
180 
181   /**
182    * Method to create and add lines to a csv-file
183    *
184    * @param env given ExecutionEnvironment
185    * @throws IOException exeption during file writing
186    */
187   private static void writeCSV(ExecutionEnvironment env) throws IOException {
188 
189     String head = String
190       .format("%s|%s|%s|%s|%s|%s%n", "Parallelism", "dataset", "vertex-label", "edge-label",
191         "verification", "Runtime(s)");
192 
193     String tail = String
194       .format("%s|%s|%s|%s|%s|%s%n", env.getParallelism(), INPUT_PATH, VERTEX_LABEL, EDGE_LABEL,
195         VERIFICATION, env.getLastJobExecutionResult().getNetRuntime(TimeUnit.SECONDS));
196 
197     File f = new File(CSV_PATH);
198     if (f.exists() && !f.isDirectory()) {
199       FileUtils.writeStringToFile(f, tail, true);
200     } else {
201       PrintWriter writer = new PrintWriter(CSV_PATH, "UTF-8");
202       writer.print(head);
203       writer.print(tail);
204       writer.close();
205     }
206   }

211   }
