90    /**
91     * Main program to run the benchmark. Arguments are the available options.
92     *
93     * @param args program arguments
94     * @throws Exception IO or execution Exception
95     */
96    public static void main(String[] args) throws Exception {
97      CommandLine cmd = parseArguments(args, CAPFBenchmark.class.getName());
98  
99      if (cmd == null) {
100       System.exit(1);
101     }
102 
103     // test if minimum arguments are set
104     performSanityCheck(cmd);
105 
106     // read cmd arguments
107     readCMDArguments(cmd);
108 
109     // read graph
110     LogicalGraph graph = readLogicalGraph(INPUT_PATH);
111 
112     // prepare collection
113     GraphCollection collection;
114 
115     System.out.println(QUERY);
116     // execute cypher with or without statistics
117     CAPFQueryResult result = graph.cypher(QUERY);
118 
119 
120     if (result.containsGraphs()) {
121       collection = result.getGraphs();
122       List<Vertex> vertices = new ArrayList<>();
123       collection.getVertices().output(new LocalCollectionOutputFormat<>(vertices));
124       System.out.println(collection.getConfig().getExecutionEnvironment().getExecutionPlan());
125 //      System.out.println(collection.getGraphHeads().count());
126     } else {
127       result.getTable().printSchema();
128     }
129 
130     // execute and write job statistics
131     writeCSV(graph.getConfig().getExecutionEnvironment());
132   }

134   /**
135    * Reads the given arguments from command line
136    *
137    * @param cmd command line
138    */
139   private static void readCMDArguments(CommandLine cmd) {
140     INPUT_PATH = cmd.getOptionValue(OPTION_INPUT_PATH);
141     CSV_PATH = cmd.getOptionValue(OPTION_CSV_PATH);
142     FIRST_NAME = cmd.getOptionValue(OPTION_FIRST_NAME);
143 
144     if (cmd.hasOption(OPTION_CUSTOM_QUERY)) {
145       StringBuilder queryBuilder = new StringBuilder();
146       for (String val : cmd.getOptionValues(OPTION_CUSTOM_QUERY)) {
147         queryBuilder.append(val).append(" ");
148       }
149       QUERY = queryBuilder.toString();
150     } else if (cmd.hasOption(OPTION_PREDEFINED_QUERY)) {
151       QUERY = getPredefinedQuery(cmd.getOptionValue(OPTION_PREDEFINED_QUERY));
152     }
153   }
154 
155   /**
156    * Checks if the minimum of arguments is provided
157    *
158    * @param cmd command line
159    */
160   private static void performSanityCheck(CommandLine cmd) {
161     if (!cmd.hasOption(OPTION_INPUT_PATH)) {
162       throw new IllegalArgumentException("Define a graph input directory.");
163     }
164     if (!cmd.hasOption(OPTION_CSV_PATH)) {
165       throw new IllegalArgumentException("Path to CSV-File need to be set.");
166     }
167     if (!(cmd.hasOption(OPTION_PREDEFINED_QUERY) || cmd.hasOption(OPTION_CUSTOM_QUERY))) {
168       throw new IllegalArgumentException("Define a query to run.");
169     }
170     if (cmd.hasOption(OPTION_PREDEFINED_QUERY) && cmd.hasOption(OPTION_CUSTOM_QUERY)) {
171       throw new IllegalArgumentException("Specify either a custom or predefined query, not both!");
172     }
173     if (cmd.hasOption(OPTION_PREDEFINED_QUERY)) {
174       String predefQuery = cmd.getOptionValue(OPTION_PREDEFINED_QUERY);
175       if (predefQuery.equals("q1") || predefQuery.equals("q2") || predefQuery.equals("q3")) {
176         if (!cmd.hasOption(OPTION_FIRST_NAME)) {
177           throw new IllegalArgumentException("Queries 1, 2 and 3 require a first name.");
178         }
179       }
180     }
181   }

183   /**
184    * Method to create and add lines to a csv-file
185    *
186    * @param env given ExecutionEnvironment
187    * @throws IOException exeption during file writing
188    */
189   private static void writeCSV(ExecutionEnvironment env) throws IOException {
190 
191     String head = String.format("%s|%s|%s|%s%n",
192       "Parallelism",
193       "dataset",
194       "query",
195       "Runtime(s)");
196 
197     String tail = String.format("%s|%s|%s|%s%n",
198       env.getParallelism(),
199       INPUT_PATH,
200       QUERY,
201       env.getLastJobExecutionResult().getNetRuntime(TimeUnit.SECONDS));
202 
203     File f = new File(CSV_PATH);
204     if (f.exists() && !f.isDirectory()) {
205       FileUtils.writeStringToFile(f, tail, true);
206     } else {
207       PrintWriter writer = new PrintWriter(CSV_PATH, "UTF-8");
208       writer.print(head);
209       writer.print(tail);
210       writer.close();
211     }
212   }

214   /**
215    * Returns used query for benchmark
216    *
217    * @param query argument input
218    * @return used query
219    */
220   private static String getPredefinedQuery(String query) {
221     switch (query) {
222     case "q1":
223       return CAPFQueries.q1(FIRST_NAME);
224     case "q2":
225       return CAPFQueries.q2(FIRST_NAME);
226     case "q3":
227       return CAPFQueries.q3(FIRST_NAME);
228     case "q4":
229       return CAPFQueries.q4();
230     case "q5":
231       return CAPFQueries.q5();
232     case "q6":
233       return CAPFQueries.q6();
234     default:
235       throw new IllegalArgumentException("Unsupported query: " + query);
236     }
237   }
