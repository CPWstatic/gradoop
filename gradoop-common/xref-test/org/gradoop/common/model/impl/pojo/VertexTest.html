40    @Test
41    public void createVertexPojoTest() {
42      GradoopId vertexID = GradoopId.get();
43      String label = "A";
44      Properties props = Properties.create();
45      props.set("k1", "v1");
46      props.set("k2", "v2");
47  
48      GradoopId graphId1 = GradoopId.get();
49      GradoopId graphId2 = GradoopId.get();
50  
51      GradoopIdSet graphIds = new GradoopIdSet();
52      graphIds.add(graphId1);
53      graphIds.add(graphId2);
54  
55      EPGMVertex vertex = new VertexFactory()
56        .initVertex(vertexID, label, props, graphIds);
57  
58      assertThat(vertex.getId(), is(vertexID));
59      assertEquals(label, vertex.getLabel());
60      assertThat(vertex.getPropertyCount(), is(2));
61      assertThat(vertex.getPropertyValue("k1").getString(), Is.<Object>is("v1"));
62      assertThat(vertex.getPropertyValue("k2").getString(), Is.<Object>is("v2"));
63      assertThat(vertex.getGraphCount(), is(2));
64      assertTrue(vertex.getGraphIds().contains(graphId1));
65      assertTrue(vertex.getGraphIds().contains(graphId2));
66    }
