16  package org.gradoop.common.model.impl.properties.strategies;
17  
18  import org.apache.flink.core.memory.DataInputView;
19  import org.apache.flink.core.memory.DataInputViewStreamWrapper;
20  import org.apache.flink.core.memory.DataOutputViewStreamWrapper;
21  import org.gradoop.common.model.impl.properties.PropertyValue;
22  import java.io.ByteArrayOutputStream;
23  import java.io.DataOutputStream;
24  import java.io.IOException;
25  import java.util.HashMap;
26  import java.util.Map;
27  
28  /**
29   * Strategy class for handling {@code PropertyValue} operations with a value of the type
30   * {@code Map}.
31   */
32  public class MapStrategy
33    extends AbstractVariableSizedPropertyValueStrategy<Map<PropertyValue, PropertyValue>> {
34  
35    @Override
36    public Map<PropertyValue, PropertyValue> read(DataInputView inputView, byte typeByte)
37        throws IOException {
38      byte[] rawBytes = readVariableSizedData(inputView, typeByte);
39      DataInputViewStreamWrapper internalInputView = createInputView(rawBytes);
40      return createMap(internalInputView);
41    }
42  
43    @Override
44    public int compare(Map value, Object other) {
45      throw new UnsupportedOperationException("Method compare() is not supported for Map.");
46    }
47  
48    @Override
49    public boolean is(Object value) {
50      if (!(value instanceof Map)) {
51        return false;
52      }
53      for (Map.Entry<?, ?> entry : ((Map<?, ?>) value).entrySet()) {
54        // Map is not a valid property value if it contains any object
55        // that is not a property value itself.
56        if (!(entry.getKey() instanceof PropertyValue) ||
57          !(entry.getValue() instanceof PropertyValue)) {
58          return false;
59        }
60      }
61  
62      return true;
63    }
64  
65    @Override
66    public Class<Map<PropertyValue, PropertyValue>> getType() {
67      return (Class) Map.class;
68    }
69  
70    /**
71     * {@inheritDoc}
72     * @throws IOException if converting the byte array to a Map fails.
73     */
74    @Override
75    public Map<PropertyValue, PropertyValue> get(byte[] bytes) throws IOException {
76      DataInputViewStreamWrapper inputView = createInputView(bytes);
77      Map<PropertyValue, PropertyValue> map;
78  
79      try {
80        if (inputView.skipBytes(PropertyValue.OFFSET) != PropertyValue.OFFSET) {
81          throw new IOException("Malformed entry in PropertyValue Map.");
82        }
83        map = createMap(inputView);
84      } catch (IOException e) {
85        throw new IOException("Error while processing DataInputViewStreamWrapper.");
86      }
87  
88      return map;
89    }
90  
91    @Override
92    public byte getRawType() {
93      return PropertyValue.TYPE_MAP;
94    }
95  
96    /**
97     * {@inheritDoc}
98     * @throws IOException if converting the value to a byte array fails.
99     */
100   @Override
101   public byte[] getRawBytes(Map<PropertyValue, PropertyValue> value) throws IOException {
102     int size = value.keySet().stream().mapToInt(PropertyValue::byteSize).sum() +
103         value.values().stream().mapToInt(PropertyValue::byteSize).sum() +
104         PropertyValue.OFFSET;
105 
106     try (ByteArrayOutputStream byteStream = new ByteArrayOutputStream(size);
107         DataOutputStream outputStream = new DataOutputStream(byteStream);
108         DataOutputViewStreamWrapper outputView = new DataOutputViewStreamWrapper(outputStream)) {
109 
110       outputStream.write(PropertyValue.TYPE_MAP);
111       for (Map.Entry<PropertyValue, PropertyValue> entry : value.entrySet()) {
112         entry.getKey().write(outputView);
113         entry.getValue().write(outputView);
114       }
115 
116       return byteStream.toByteArray();
117     } catch (IOException e) {
118       throw new IOException("Error writing PropertyValue with MapStrategy.", e);
119     }
120   }
121 
122   /**
123    * Creates a map with data read from an {@link DataInputViewStreamWrapper}.
124    *
125    * @param inputView {@link DataInputViewStreamWrapper} containing data
126    * @return a map containing the deserialized data
127    */
128   private Map<PropertyValue, PropertyValue> createMap(DataInputViewStreamWrapper inputView)
129     throws IOException {
130     PropertyValue key;
131     PropertyValue value;
132 
133     Map<PropertyValue, PropertyValue> map = new HashMap<>();
134 
135     try {
136       while (inputView.available() > 0) {
137         key = new PropertyValue();
138         key.read(inputView);
139 
140         value = new PropertyValue();
141         value.read(inputView);
142 
143         map.put(key, value);
144       }
145     } catch (IOException e) {
146       throw new IOException("Error reading PropertyValue with MapStrategy.", e);
147     }
148     return map;
149   }
