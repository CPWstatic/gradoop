171     temporalGraph.getEdges().output(new LocalCollectionOutputFormat<>(loadedEdges));
172 
173     Collection<EPGMGraphHead> epgmGraphHeads = new ArrayList<>();
174     Collection<EPGMVertex> epgmVertices = new ArrayList<>();
175     Collection<EPGMEdge> epgmEdges = new ArrayList<>();
176 
177     testLogicalGraph.getGraphHead().output(new LocalCollectionOutputFormat<>(epgmGraphHeads));
178     testLogicalGraph.getVertices().output(new LocalCollectionOutputFormat<>(epgmVertices));
179     testLogicalGraph.getEdges().output(new LocalCollectionOutputFormat<>(epgmEdges));
180 
181     getExecutionEnvironment().execute();
182 
183     assertFalse(loadedGraphHeads.isEmpty());
184     assertFalse(loadedVertices.isEmpty());
185     assertFalse(loadedEdges.isEmpty());
186 
187     validateElementCollections(epgmGraphHeads, loadedGraphHeads);
188     validateElementCollections(epgmVertices, loadedVertices);
189     validateElementCollections(epgmEdges, loadedEdges);
190     validateGraphElementCollections(epgmVertices, loadedVertices);
191     validateGraphElementCollections(epgmEdges, loadedEdges);
192 
193     loadedGraphHeads.forEach(this::checkDefaultTemporalElement);
194     loadedVertices.forEach(this::checkDefaultTemporalElement);
195     loadedEdges.forEach(this::checkDefaultTemporalElement);
