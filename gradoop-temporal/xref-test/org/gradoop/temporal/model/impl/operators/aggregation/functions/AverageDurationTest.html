46    /**
47     * Test the implementation of {@link AverageDuration#getIncrement(TemporalElement)} for transaction time.
48     */
49    @Test
50    public void testGetIncrementForTxTime() {
51      AverageDuration function = new AverageDuration("", TRANSACTION_TIME);
52      TemporalElement testElement = getVertexFactory().createVertex();
53      testElement.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
54      // This will produce an overflow, which is currently ignored.
55      assertEquals(IGNORED_VALUE, function.getIncrement(testElement));
56      testElement.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 7L));
57      assertEquals(IGNORED_VALUE, function.getIncrement(testElement));
58      testElement.setTransactionTime(Tuple2.of(4L, DEFAULT_TIME_TO));
59      assertEquals(IGNORED_VALUE, function.getIncrement(testElement));
60      testElement.setTransactionTime(Tuple2.of(1L, 11L));
61      assertEquals(create(Arrays.asList(create(10L), create(1L))),
62        function.getIncrement(testElement));
63    }

83    /**
84     * Test the {@link AverageDuration} aggregate function and its subclasses in an aggregation.
85     *
86     * @throws Exception when the execution in Flink fails.
87     */
88    @Test
89    public void testWithAggregation() throws Exception {
90      TemporalGraphFactory graphFactory = getConfig().getTemporalGraphFactory();
91      VertexFactory<TemporalVertex> vertexFactory = graphFactory.getVertexFactory();
92      EdgeFactory<TemporalEdge> edgeFactory = graphFactory.getEdgeFactory();
93      TemporalVertex v1 = vertexFactory.createVertex();
94      v1.setTransactionTime(Tuple2.of(1L, 2L));
95      v1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
96      TemporalVertex v2 = vertexFactory.createVertex();
97      v2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 5L));
98      v2.setValidTime(Tuple2.of(-3L, DEFAULT_TIME_TO));
99      TemporalVertex v3 = vertexFactory.createVertex();
100     v3.setTransactionTime(Tuple2.of(0L, DEFAULT_TIME_TO));
101     v3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, 0L));
102     TemporalVertex v4 = vertexFactory.createVertex();
103     v4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
104     v4.setValidTime(Tuple2.of(-5L, -2L));
105     TemporalVertex v5 = vertexFactory.createVertex();
106     v5.setTransactionTime(Tuple2.of(1L, 4L));
107     v5.setValidTime(Tuple2.of(-10L, -3L));
108 
109     TemporalEdge e1 = edgeFactory.createEdge(v1.getId(), v2.getId());
110     e1.setTransactionTime(Tuple2.of(0L, 30L));
111     e1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
112     TemporalEdge e2 = edgeFactory.createEdge(v2.getId(), v3.getId());
113     e2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 7L));
114     e2.setValidTime(Tuple2.of(-100L, DEFAULT_TIME_TO));
115     TemporalEdge e3 = edgeFactory.createEdge(v3.getId(), v1.getId());
116     e3.setTransactionTime(Tuple2.of(-1L, DEFAULT_TIME_TO));
117     e3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, -80L));
118     TemporalEdge e4 = edgeFactory.createEdge(v1.getId(), v3.getId());
119     e4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
120     e4.setValidTime(Tuple2.of(-120L, -107L));
121     TemporalEdge e5 = edgeFactory.createEdge(v2.getId(), v4.getId());
122     e5.setTransactionTime(Tuple2.of(10L, 50L));
123     e5.setValidTime(Tuple2.of(-301L, -276L));
124 
125     DataSet<TemporalVertex> vertices = getExecutionEnvironment().fromElements(v1, v2, v3, v4, v5);
126     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(e1, e2, e3, e4, e5);
127     TemporalGraph result = graphFactory.fromDataSets(vertices, edges)
128       .aggregate(new AverageDuration("avgDurTx", TRANSACTION_TIME),
129         new AverageDuration("avgDurValid", VALID_TIME),
130         new AverageVertexDuration("avgVertexDurTx", TRANSACTION_TIME),
131         new AverageVertexDuration("avgVertexDurValid", VALID_TIME),
132         new AverageEdgeDuration("avgEdgeDurTx", TRANSACTION_TIME),
133         new AverageEdgeDuration("avgEdgeDurValid", VALID_TIME));
134     Properties headProperties = result.getGraphHead().collect().get(0).getProperties();
135     assertEquals(18.5d, headProperties.get("avgDurTx").getDouble(), 1e-7d);
136     assertEquals(12.d, headProperties.get("avgDurValid").getDouble(), 1e-7d);
137     assertEquals(2.d, headProperties.get("avgVertexDurTx").getDouble(), 1e-7d);
138     assertEquals(5.d, headProperties.get("avgVertexDurValid").getDouble(), 1e-7d);
139     assertEquals(35.d, headProperties.get("avgEdgeDurTx").getDouble(), 1e-7d);
140     assertEquals(19.d, headProperties.get("avgEdgeDurValid").getDouble(), 1e-7d);
141   }
