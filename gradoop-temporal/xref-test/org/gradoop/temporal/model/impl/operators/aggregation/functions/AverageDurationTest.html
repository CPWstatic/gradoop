145   /**
146    * Test the {@link AverageDuration} aggregate function and its subclasses in an aggregation.
147    *
148    * @throws Exception when the execution in Flink fails.
149    */
150   @Test
151   public void testWithAggregation() throws Exception {
152     TemporalGraphFactory graphFactory = getConfig().getTemporalGraphFactory();
153     VertexFactory<TemporalVertex> vertexFactory = graphFactory.getVertexFactory();
154     EdgeFactory<TemporalEdge> edgeFactory = graphFactory.getEdgeFactory();
155     TemporalVertex v1 = vertexFactory.createVertex();
156     v1.setTransactionTime(Tuple2.of(1L, 2L));
157     v1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
158     TemporalVertex v2 = vertexFactory.createVertex();
159     v2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 5L));
160     v2.setValidTime(Tuple2.of(-3L, DEFAULT_TIME_TO));
161     TemporalVertex v3 = vertexFactory.createVertex();
162     v3.setTransactionTime(Tuple2.of(0L, DEFAULT_TIME_TO));
163     v3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, 0L));
164     TemporalVertex v4 = vertexFactory.createVertex();
165     v4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
166     v4.setValidTime(Tuple2.of(-5L, -2L));
167     TemporalVertex v5 = vertexFactory.createVertex();
168     v5.setTransactionTime(Tuple2.of(1L, 4L));
169     v5.setValidTime(Tuple2.of(-10L, -3L));
170 
171     TemporalEdge e1 = edgeFactory.createEdge(v1.getId(), v2.getId());
172     e1.setTransactionTime(Tuple2.of(0L, 30L));
173     e1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
174     TemporalEdge e2 = edgeFactory.createEdge(v2.getId(), v3.getId());
175     e2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 7L));
176     e2.setValidTime(Tuple2.of(-100L, DEFAULT_TIME_TO));
177     TemporalEdge e3 = edgeFactory.createEdge(v3.getId(), v1.getId());
178     e3.setTransactionTime(Tuple2.of(-1L, DEFAULT_TIME_TO));
179     e3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, -80L));
180     TemporalEdge e4 = edgeFactory.createEdge(v1.getId(), v3.getId());
181     e4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
182     e4.setValidTime(Tuple2.of(-120L, -107L));
183     TemporalEdge e5 = edgeFactory.createEdge(v2.getId(), v4.getId());
184     e5.setTransactionTime(Tuple2.of(10L, 50L));
185     e5.setValidTime(Tuple2.of(-301L, -276L));
186 
187     DataSet<TemporalVertex> vertices = getExecutionEnvironment().fromElements(v1, v2, v3, v4, v5);
188     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(e1, e2, e3, e4, e5);
189     TemporalGraph result = graphFactory.fromDataSets(vertices, edges)
190       .aggregate(new AverageDuration("avgDurTx", TRANSACTION_TIME),
191         new AverageDuration("avgDurValid", VALID_TIME),
192         new AverageVertexDuration("avgVertexDurTx", TRANSACTION_TIME),
193         new AverageVertexDuration("avgVertexDurValid", VALID_TIME),
194         new AverageEdgeDuration("avgEdgeDurTx", TRANSACTION_TIME),
195         new AverageEdgeDuration("avgEdgeDurValid", VALID_TIME));
196     Properties headProperties = result.getGraphHead().collect().get(0).getProperties();
197     assertEquals(18.5d, headProperties.get("avgDurTx").getDouble(), 1e-7d);
198     assertEquals(12.d, headProperties.get("avgDurValid").getDouble(), 1e-7d);
199     assertEquals(2.d, headProperties.get("avgVertexDurTx").getDouble(), 1e-7d);
200     assertEquals(5.d, headProperties.get("avgVertexDurValid").getDouble(), 1e-7d);
201     assertEquals(35.d, headProperties.get("avgEdgeDurTx").getDouble(), 1e-7d);
202     assertEquals(19.d, headProperties.get("avgEdgeDurValid").getDouble(), 1e-7d);
203   }
