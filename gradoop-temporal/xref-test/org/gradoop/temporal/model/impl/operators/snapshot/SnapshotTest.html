47  /**
48   * Test for the snapshot operator for temporal graphs.
49   */
50  @RunWith(Parameterized.class)
51  public class SnapshotTest extends TemporalGradoopTestBase {
52  
53    /**
54     * The temporal predicate to test.
55     */
56    @Parameterized.Parameter
57    public TemporalPredicate predicate;
58  
59    /**
60     * The temporal dimension to test.
61     */
62    @Parameterized.Parameter(1)
63    public TimeDimension dimension;
64  
65    /**
66     * The variable name of the input graph.
67     */
68    @Parameterized.Parameter(2)
69    public String[] expectedVertices;
70  
71    /**
72     * The expected result graph for the ascii graph loader.
73     */
74    @Parameterized.Parameter(3)
75    public String[] expectedEdges;
76  
77    /**
78     * Run the test. Calls the snapshot operator using a predicate and compares results with the expected
79     * result graph.
80     *
81     * @throws Exception when the Execution in Flink fails.
82     */
83    @Test
84    public void runTest() throws Exception {
85      TemporalGraph inputGraph = getTestGraphWithValues();
86  
87      Collection<TemporalVertex> resultVertices = new HashSet<>();
88      Collection<TemporalEdge> resultEdges = new HashSet<>();
89  
90      TemporalGraph resultGraph = inputGraph.callForGraph(new Snapshot(predicate, dimension));
91  
92      resultGraph.getVertices().output(new LocalCollectionOutputFormat<>(resultVertices));
93      resultGraph.getEdges().output(new LocalCollectionOutputFormat<>(resultEdges));
94  
95      getConfig().getExecutionEnvironment().execute();
96  
97      List<String> resultVertexLabels =
98        resultVertices.stream().map(EPGMElement::getLabel).collect(Collectors.toList());
99      List<String> resultEdgeLabels =
100       resultEdges.stream().map(EPGMElement::getLabel).collect(Collectors.toList());
101 
102     assertEquals(expectedVertices.length, resultVertices.size());
103     assertEquals(expectedEdges.length, resultEdges.size());
104 
105     Arrays.stream(expectedVertices).map(resultVertexLabels::contains).forEach(Assert::assertTrue);
106     Arrays.stream(expectedEdges).map(resultEdgeLabels::contains).forEach(Assert::assertTrue);
107   }
