37  /**
38   * Test for the snapshot operator for temporal graphs.
39   */
40  @RunWith(Parameterized.class)
41  public class SnapshotTest extends TemporalGradoopTestBase {
42  
43    /**
44     * The temporal predicate to test.
45     */
46    @Parameterized.Parameter
47    public TemporalPredicate predicate;
48  
49    /**
50     * The variable name of the input graph.
51     */
52    @Parameterized.Parameter(1)
53    public String inputGraph;
54  
55    /**
56     * The expected result graph for the ascii graph loader.
57     */
58    @Parameterized.Parameter(2)
59    public String resultGraph;
60  
61    /**
62     * Run the test. Calls the snapshot operator using a predicate and compares results with the expected
63     * result graph.
64     *
65     * @throws Exception when the Execution in Flink fails.
66     */
67    @Test
68    public void runTest() throws Exception {
69      FlinkAsciiGraphLoader loader = getTemporalSocialNetworkLoader();
70      loader.appendToDatabaseFromString(resultGraph);
71      TemporalGraph input = toTemporalGraphWithDefaultExtractors(loader.getLogicalGraphByVariable(inputGraph));
72      LogicalGraph expected = loader.getLogicalGraphByVariable("expected");
73      TemporalGraph result = input.callForGraph(new Snapshot(predicate));
74      collectAndAssertTrue(result.toLogicalGraph().equalsByElementData(expected));
75    }
76  
77    /**
78     * Parameters for this parametrized test.
79     *
80     * @return An array containing arrays in the form of {@code {predicate, inputGraph,
81     * extraGraphData}}.
82     */
83    @Parameterized.Parameters(name = "{1} {0}")
84    public static Iterable<Object[]> parameters() {
85      // Another test graph with more validTo times.
86      final String g4 = "g4[(t1:A {__valFrom: 1543700000000L, __valTo: 1543900000000L})-->" +
87        "(t2:B {__valFrom: 1543700000000L})-->(t3:C {__valTo: 1543800000000L})-->" +
88        "(t4:D {__valFrom: 1543600000000L, __valTo: 1543800000000L})-->" +
89        "(t5:F {__valFrom: 1543700000000L, __valTo: 1544000000000L})]";
90      return Arrays.asList(new Object[][] {
91        {new All(), "g0",
92          "expected[(eve)-[eka]->(alice)-[akb]->(bob)(eve)-[ekb]->(bob)-[bka]->(alice)]"},
93        {new AsOf(1543600000000L), "g0", "expected[(alice)-[akb]->(bob)-[bka]->(alice)]"},
94        {new Between(1543500000000L, 1543800000000L), "g0",
95          "expected[(eve)-[eka]->(alice)-[akb]->(bob)-[bka]->(alice)]"},
96        {new ContainedIn(1543700000000L, 1543900000000L), "g4", g4 + "expected[(t1)]"},
97        {new CreatedIn(1543500000000L, 1543800000000L), "g0", "expected[(eve)(bob)]"},
98        {new DeletedIn(1543900000000L, 1543900000000L), "g4", g4 + "expected[(t1)]"},
99        {new FromTo(1543500000000L, 1543800000000L), "g0",
100         "expected[(alice)-[akb]->(bob)-[bka]->(alice)]"},
101       {new ValidDuring(1543600000000L, 1543800000000L), "g3",
102         "expected[(gps)-[:hasMember]->(carol)]"}
103     });
