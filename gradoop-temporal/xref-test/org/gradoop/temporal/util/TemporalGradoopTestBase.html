292     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(e1, e2, e3, e4, e5);
293     return getConfig().getTemporalGraphFactory().fromDataSets(vertices, edges);

313   /**
314    * Create a temporal edge with temporal attributes set.
315    *
316    * @param label     The label of the edge.
317    * @param source    The element used as a source for the edge.
318    * @param target    The element used as a target for the edge.
319    * @param txFrom    The start of the transaction time.
320    * @param txTo      The end of the transaction time.
321    * @param validFrom The start of the valid time.
322    * @param validTo   The end of the valid time.
323    * @return A temporal edge with those times set.
324    */
325   private TemporalEdge createEdge(String label, Identifiable source, Identifiable target, long txFrom,
326                                   long txTo, long validFrom, long validTo) {
327     TemporalEdge edge = getConfig().getTemporalGraphFactory().getEdgeFactory()
328       .createEdge(source.getId(), target.getId());
329     edge.setLabel(label);
330     edge.setTransactionTime(Tuple2.of(txFrom, txTo));
331     edge.setValidTime(Tuple2.of(validFrom, validTo));
332     return edge;
333   }

337    *
338    * @param label     The label of the vertex.
339    * @param txFrom    The start of the transaction time.
340    * @param txTo      The end of the transaction time.
341    * @param validFrom The start of the valid time.
342    * @param validTo   The end of the valid time.
343    * @return A temporal vertex with those times set.

345   private TemporalVertex createVertex(String label, long txFrom, long txTo, long validFrom, long validTo) {
346     TemporalVertex vertex = getConfig().getTemporalGraphFactory().getVertexFactory().createVertex();
347     vertex.setLabel(label);
348     vertex.setTransactionTime(Tuple2.of(txFrom, txTo));
349     vertex.setValidTime(Tuple2.of(validFrom, validTo));

353   /**
354    * Convert a formatted date/time string to the time format (milliseconds since unix epoch) used by Gradoop.
355    * The date string is expected to be formatted according to {@link #DATE_FORMAT}.
356    *
357    * @param dateTimeString The date string.
358    * @return The time in milliseconds since unix epoch.
359    */
360   protected long asMillis(String dateTimeString) {
361     try {
362       return dateFormatter.parse(dateTimeString).getTime();
363     } catch (ParseException pe) {
364       throw new IllegalArgumentException("Failed to parse date.", pe);
365     }
