51  /**
52   * A temporal (logical) graph is a base concept of the Temporal Property Graph Model (TPGM) that
53   * extends the Extended Property Graph Model (EPGM). The temporal graph inherits the main concepts
54   * of the {@link org.gradoop.flink.model.impl.epgm.LogicalGraph} and extends them by temporal
55   * attributes. These attributes are two temporal intervals: the valid-time and transaction time.
56   * Both are represented by a {@link Tuple2} of Long values that specify the beginning and end time as unix
57   * timestamp in milliseconds.
58   *
59   * <ul>
60   *   <li>{@code transactionTime}: {@code (tx-from [ms], tx-to [ms])}</li>
61   *   <li>{@code validTime}: {@code (val-from [ms], val-to [ms])}</li>
62   * </ul>
63   *
64   * Furthermore, a temporal graph provides operations that are performed on the underlying data.
65   * These operations result in either another temporal graph or in a {@link TemporalGraphCollection}.
66   *
67   * Analogous to a logical graph, a temporal graph is wrapping a layout which defines, how the graph
68   * is represented in Apache Flink.<br>
69   * Note that the {@link TemporalGraph} also implements that interface and just forwards the calls to
70   * the layout. This is just for convenience and API synchronicity.
71   *
72   * @see TemporalGraphOperators
73   */
74  public class TemporalGraph implements BaseGraph<TemporalGraphHead, TemporalVertex, TemporalEdge,
75    TemporalGraph, TemporalGraphCollection>, TemporalGraphOperators {

115   /**
116    * Returns a 1-element dataset containing a {@link Boolean} value which indicates if the graph is empty.
117    *
118    * A graph is considered empty, if it contains no vertices.
119    *
120    * @return  1-element dataset containing {@link Boolean#TRUE}, if the collection is
121    *          empty or {@link Boolean#FALSE} if not
122    */
123   public DataSet<Boolean> isEmpty() {
124     return getVertices().map(new True<>()).distinct()
125       .union(getConfig().getExecutionEnvironment().fromElements(false)).reduce(new Or())
126       .map(new Not());
127   }

227   @Override
228   public LogicalGraph toLogicalGraph() {
229     final LogicalGraphFactory logicalGraphFactory = getConfig().getLogicalGraphFactory();
230     return logicalGraphFactory.fromDataSets(
231       getGraphHead().map(new TemporalGraphHeadToGraphHead(logicalGraphFactory.getGraphHeadFactory())),
232       getVertices().map(new TemporalVertexToVertex(logicalGraphFactory.getVertexFactory())),
233       getEdges().map(new TemporalEdgeToEdge(logicalGraphFactory.getEdgeFactory())));
234   }

236   /**

238    * default values for the temporal attributes.

241    * @return a temporal graph with default temporal values
242    * @see TemporalGraphFactory#fromNonTemporalGraph(BaseGraph)

244   public static TemporalGraph fromLogicalGraph(LogicalGraph logicalGraph) {
245     return TemporalGradoopConfig.fromGradoopFlinkConfig(logicalGraph.getConfig()).getTemporalGraphFactory()
246       .fromNonTemporalGraph(logicalGraph);
247   }
